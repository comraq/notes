\newglossaryentry{cpo}
{
  name={$\omega$-Complete Partial Order},
  text={$\omega$-complete partial order},
  description={is a poset in which every \textit{$\omega$-chain} has a
  \textit{supremum}.
  \begin{note}
    If such a poset also has minimum element, it is called a \textit{strict
    $\omega$-CPO} and such an element is commonly known as \textit{bottom},
    $\bot$.
  \end{note}
  \begin{note}
    A strict $\omega$-CPO is a \textit{flat CPO} if for any $d$ and $d'$,
    $d\neq d'$, $d\neq\bot$, $d'\neq\bot$ and $d\not\leq d'$.
  \end{note}
  },
  symbol={$\omega$-CPO}
}

\newglossaryentry{scottcontinuous}
{
  name=Scott-continuous Function,
  text=Scott-continuous function,
  description={For two posets $S$ and $T$, a function $f:S\rightarrow T$ is
    \textbf{Scott-continuous} if whenever $s$ is the supremum of a chain
    $\mathscr{C}=(c_0,c_1,c_2,\dots)\in S$, then $f(s)$ is the supremum of the
    image $f(\mathscr{C})=\{f(c_i)|i\in\mathbb{N}\}$ (ie: Scott-continuous
    functions preserves the supremum).
    \begin{note}
      Such a function is \textit{strict} if it also preserves the bottom
      element between strict $\omega$-CPOs.
    \end{note}
  }
}

  name=Kleene Closure Induced Homormophism,
  text=kleene closure induced homormophism,
  description={For a function $f:A\rightarrow B$, between sets $A$ and $B$,
    $f^*:A^*\rightarrow B^*$ is a monoid homomorphism defined by
    $f^*((a_1,a_2,\dots,a_k))=(f(a_1),f(a_2),\dots,f(a_k))$.

    \end{note}
  }
}

\longnewglossaryentry{rewritesystem}
{
  name=Rewrite System,
  text=rewrite system
} {
  A \textbf{rewrite system} $G$ consists of a finite set of
  \textit{productions} which are symbols of the form $y\rightarrow z$ where
  $y,z\in A^*$. Here, $A$ is a finite alphabet and $A^*$ is the kleene
  closure of $A$, consisting of strings formed by letters in $A$.

  \begin{note}[Directly Derive]
    A string $x$ can be \textit{directly derived} from $w$, denoted as
    $w\Rightarrow x$, if for strings $u,v\in A^*$, $y\rightarrow z,w=uyv$
    and $x=uzv$.
  \end{note}
  \begin{note}[Example]
    If $G$ has a production $ab\rightarrow baa$, then $baba\Rightarrow
    bbaaa,abb\Rightarrow baab$ and $ab\Rightarrow baa$.
  \end{note}
  \begin{note}[Derive to]
    is an operation that is the reflexive transitive closure of $\Rightarrow$
    denoted as $\xRightarrow{*}$. That is for any string $w$:
    \begin{itemize}
      \item $w\xRightarrow{*}w$
      \item $(w\xRightarrow{*}x)\Leftrightarrow(\exists
        w'$ st. $w\Rightarrow w'$ and $w'\Rightarrow x)$
    \end{itemize}
  \end{note}
}

\longnewglossaryentry{contextfreerewritesystem}
{
  name=Context-Free Rewrite System,
  text=context-free rewrite system
} {
  is a rewrite system with the property that for every production of the
  form $w\rightarrow x$, $w$ is a string of length 1
  (effectively a single letter).

  \begin{note}[As a 2-category]
    Let $\mathscr{W}$ be a context-free rewrite system on the alphabet $A$. The
    2-category $\mathbf{C}(\mathscr{W})$ is defined as follows:
    \begin{itemize}
      \item $\mathbf{C}(\mathscr{W})$ has one 0-cell.

      \item 1-cells of $\mathbf{C}(\mathscr{W})$ are the strings in $A^*$.

      \item A 2-cell $\alpha:w/rightarrow x$ is the derivation forest of a
        derivation of $x$ from $w$ using productions of $W$.

      \item If $\alpha:w\rightarrow x$ and $\beta:w'\rightarrow x'$ are
        2-cells, then $\alpha\ast\beta:ww'\rightarrow xx'$ is the derivation
        forest corresponding to deriving $ww'$ to $xw'$ using $\alpha$, then
        from $xw'$ to $xx'$ using $\beta$.

      \item If $\alpha:w\rightarrow x$ and $\gamma:x\rightarrow y$ are
        derivation forests, then $\gamma\circ\alpha:w\rightarrow y$ is the
        derivation forest obtained by using $\alpha$ then $\gamma$
    \end{itemize}
  \end{note}
}

\longnewglossaryentry{contextfreegrammar}
{
  name=Context-Free Grammar,
  text=context-free grammar
} {
  A \textbf{context-free grammar} $G$ is a context-free rewrite system
  where:
  \begin{itemize}
    \item The alphabet $A$ is the union of disjoint sets $T$ (set of
      \textit{terminals} denoted by lowercase letters) and $N$ (set of
      \textit{nonterminals} denoted by uppercase letters).

    \item There exists a distinguished nonterminal known as the
      \textit{start symbol}.

    \item Every production has the form $V\rightarrow w$ where $V$ is a
      single nonterminal and $w$ is any string in $A^*$ ($w$ can contain
      terminals, nonterminals or both).

    \item The language generated by $G$ is the set $\{w\in
      T^*|S\xRightarrow{*}w\}$.
  \end{itemize}
}

{
  name=Chain,
  text=chain,
  description={A poset $P$ is a \textbf{chain} (also known as
    \textit{totally} or \textit{linearly ordered set}) if for all $x,y\in P$,
    $x\leq y$ or $y\leq x$.

\newglossaryentry{productlattice}
{
  name=Product Lattice,
  text=product lattice,
  description={If $D_1,D_2,\dots,D_n$ are complete lattices, $D_1\times
    D_2\times\dots\times D_n$ is the \textbf{product lattice} (also a
    complete lattice), whose elements are of the form $\langle
    x_1,x_2,\dots,x_n\rangle$ where $x_i\in D_i$ for $1\leq i\leq n$. Two
    such elements are partially ordered by $\langle
    x_1,x_2,\dots,x_n\rangle\leq\langle y_1,y_2,\dots,y_n\rangle$ if $x_i\leq
  y_i$ for all $1\leq i\leq n$.}
}

\longnewglossaryentry{sumlattice}
{
  name=Sum Lattice,
  text=sum lattice
} {
  If $D_1,D_2,\dots,D_n$ are complete lattices, $D_1+
  D_2+\dots+ D_n$ is the \textbf{sum lattice} (also a complete lattice),
  which is the disjoint union of all the elements of $D_1,D_2,\dots,D_n$
  with a new $\top$ and $\bot$. $\bot\leq x\leq\top$ for all $x\in D_1+
  D_2+\dots+ D_n$. If $x$ is an element of $D_i$ and $y$ is an element of
  $D_j$, then $x\leq y$ in $D_1+ D_2+\dots+ D_n$ if and only
  if $i=j$ and $x\leq y$ in $D_i$.

  \begin{note}
    The sum lattice construction with a new $\top$ and $\bot$ is also called
    a \textit{separated sum}.
  \end{note}

  \begin{note}
    An alternative sum lattice construction is to make $\top$ and $\bot$ in
    the sum domain the images of $\top$ and $\bot$ of each of the component
    domains. This is also known as a \textit{coalesced sum}.
  \end{note}

  \begin{note}
    It is possible for the same lattice to appear as two distinct summands of
    a sum lattice. In this case, each element of this lattice will appear
    twice in the sum lattice.
  \end{note}
}

\longnewglossaryentry{basis}
{
  name=Basis,
  text=basis
} {
  A subset $E$ of a complete lattice $D$ is a \textbf{basis} of $D$ if
  $x=\bigsqcup\{e|e\in E,e\leq x\}$ for all $x\in D$ and $\bigsqcup E'\in E$
  for all finite $E'\subseteq E$.

  \begin{note}
    If the condition $\bigsqcup E'\in E$ for all finite $E'\subseteq E$ is
    absent, the $E$ is a \textit{subbasis} of $D$.
  \end{note}
}

\newglossaryentry{directedset}
{
  name=Directed Set,
  text=directed set,
  description={is a set $X$ if every finite subset of $X$ has an upper bound
    in $X$. That is, if $A\subseteq X$ is finite then there is an element
    $b\in X$ such that $a\leq b$ for all $a\in A$.
    \begin{note}
      Finite subsets of $X$ need not have least upper bounds in $X$.
    \end{note}
    \begin{note}
      Directed sets are not empty since the empty et is one of its finite
      subsets, and thus directed sets must contain an element that is an upper
      bound of the empty set.
    \end{note}
    \begin{note}
      A directed set is \textit{interesting} if it does not contain its least
      upper bound, ie: $\bigsqcup X\notin X$.
    \end{note}
  }
}

\newglossaryentry{continuousfunction}
{
  name=Continuous Function,
  text=continuous function,
  description={A function $f:D\rightarrow D'$ between complete lattices is
    \textbf{continuous} if $f(\bigsqcup X)=\bigsqcup\{f(x)|x\in X\}$ for all
    directed $X\subseteq D$.
    \begin{note}
      All monotonic functions on finite lattices are continuous.
    \end{note}
  }
}

\newglossaryentry{limitpoint}
{
  name=Limit Point,
  text=limit point,
  description={is an element $x\in D$ of a complete lattice if that
  $x=\bigsqcup X$ for some interesting directed $X\subseteq D$.}
}

\newglossaryentry{wellbelow}
{
  name=Well Below,
  text=well below,
  symbol={$\ll$},
  description={For $x,y$ in a continous lattice $D$, $x$ is \textbf{well
    below} $y$, $x\ll y$ if whenever $y\leq\bigsqcup Z$ for some directed
  $Z\subseteq D$ then $x\leq z$ for some $z\in Z$.}
}

\newglossaryentry{isolatedpoint}
{
  name=Isolated Point,
  text=isolated point,
  description={A point $a\in D$ of a complete lattice is \textbf{isolated}
    if $a\ll a$.
    \begin{note}
      An isolated point cannot be the limit point of any interesting
      directed set.
    \end{note}
  }
}

\newglossaryentry{continouslattice}
{
  name=Continuous Lattice,
  text=continuous lattice,
  description={is a complete lattice $D$ such that for all $y\in D$,
    $y=\bigsqcup\{x\in D|x\ll y\}$.
    \begin{note}
      Every point in a continuous lattice that is not isolated is a limit
      point.
    \end{note}
  }
}

\newglossaryentry{domaintheorydomain}

    \end{note}
  }

\newglossaryentry{fixedpointoperator}
{
  name=Fixed Point Operator,
  text=fixed point operator,
  symbol={$fix$},
  description={For any domain $D$, the function $fix:(D\rightarrow
    D)\rightarrow D$ is a fixed point operator defined by
    $fix(f)=\bigsqcup_{n=0}^\infty f^n(\bot)$.
    \begin{note}
      If $f:D\rightarrow D$ is a continuous function, $fix(f)$ is the
      least fixed point of $f$.
    \end{note}
    \begin{note}
      When applied to a function, fixed point operators produce a fixed
      point of the function, ie: $fix(f)$ is the fixed point of $f$.
    \end{note}
  }
}
