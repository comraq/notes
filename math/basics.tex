\longnewglossaryentry{set}
{
  name=Set,
  text=set,
} {
  is a mathematical entity that is distinct from, but
  completely determined by, its elements (if any). For every entity $x$
  and set $S$, the statement $x\in S$, read "\textit{$x$ is an element of
  $S$}", is a sentence that is either true or false.
  \begin{note}
    When members of a set $A$ are sets themselves, $A$ may be called a \textit{class, collection}, or \textit{family}.
  \end{note}
  \begin{note}[Set Equality]
    \item[]
    \begin{itemize}
      \item
        Two sets $A$ and $B$ are equal, written $A=B$, if they consist of the same elements. Note that sets do not depend on the way they are displayed, a set remains the same if its elements are repeated or rearraged.
      \item
        Two sets $A$ and $B$ are equal if and only if $A\subseteq B$ and $B\subseteq A$.
    \end{itemize}
    Note that this is different from the concept of \gls{setequivalence}.
  \end{note}
}

\longnewglossaryentry{setequivalence}
{
  name=Set Equivalence,
  text=set equivalence,
  symbol={$\sim$}
} {
  A set $A$ is called \textbf{equivalent} to a set $B$, written $A\sim B$, if there exists a function $f:A\rightarrow B$ which is both \textit{injective} and \textit{surjective}. The function $f$ is then said to define a \textit{one-to-one correspondence} between the sets $A$ and $B$. By this definition, two finite sets are equivalent iff they contain the same number of elements.
  \begin{note}
    For finite sets, equivalence corresponds to the usual meaning of two sets containing the same number of elements.
  \end{note}
  \begin{note}
    An infinite set can be equivalent to a proper subset of itself, this property is tru of infinite sets in general.
  \end{note}
}

\longnewglossaryentry{setcardinality}
{
  name=Set Cardinality,
  text=set cardinality,
  symbol={$\#$},
} {
  If $A$ is equivalent to $B$ ($A\sim B$), then we say that $A$ and $B$ have the same \textbf{cardinal number} or \textbf{cardinality}. We write $\#(A)$ for "the cardinal number (or cardinality) of $A$".
  \begin{equation*}
    \#(A)=\#(B)\text{ iff }A\sim B
  \end{equation*}
  \begin{note}
    The cardinal numbers of $\mathbb{N}$ and the interval $[0,1]$ are:
    \begin{equation*}
      \aleph_0\text{(read: \textit{aleph-null})}=\#(\mathbb{N})\text{, }\textbf{c}=\#([0,1])
    \end{equation*}
    A set $X$ is said to have cardinality $\aleph_0$  iff it is equivalent to $\mathbb{N}$. And $0<1<2<\dots<\aleph_0<\textbf{c}$.
  \end{note}
  \begin{note}[Finite]
    A set is \textbf{finite} iff it is empty or equivalent to $\{1,2,\dots,n\}$ for some $n\in\mathbb{N}$. Otherwise it is said to be \textbf{infinite}.
  \end{note}
  \begin{note}[Denumerable]
    A set is called \textbf{denumerable} if it has cardinality $\aleph_0$. Note that every infinite set contains a denumerable subset.
  \end{note}
  \begin{note}[Countable]
    A set is called \textbf{countable} iff it is finite or denumerable. Note that Every subset of a countable set is countable.
  \end{note}
  \begin{note}
    A set which is neither finite nor denumerable is said to be \textit{non-denumerable} or \textit{non-countable}.
  \end{note}
  \begin{note}[Cantor's Theorem]
    The power set $\mathscr{P}(A)$ of any set $A$ has cardinality greater than $A$.
  \end{note}
}

\newglossaryentry{relation}
{
  name=Relation,
  text=relation,
  description={A \textbf{relation} $\alpha$ from a set $S$ to a set $T$ is a subset
    of $S\times T$. Any such subset is a relation from $S$ to $T$. Extreme
    examples of relations are the $\emptyset$ and the set $S\times
    T$.
    \begin{note}[Domain and Range]
      The \textit{domain} of a relation $R$ from $A$ to $B$ is the set of first coodinates of the pairs in $R$ and its \textit{range} is the set of second coordinates. i.e., domain of $R=\{a:\langle a,b\rangle\in R\}$, range of $R=\{b:\langle a,b\rangle\in R\}$.
    \end{note}
    \begin{note}[Inverse]
      The \textit{inverse} of $R$, denoted by $R^{-1}$, is the relation from $B$ to $A$ defined by domain of $R=\{a:\langle a,b\rangle\in R\}$, range of $R=\{b:\langle a,b\rangle\in R\}$.
    \end{note}
    \begin{note}[Composition]
      Let $U$ be a relation from $A$ to $B$ and let $V$ be a relation from $B$ to $C$ ($U\subseteq A\times B$ and $V\subseteq B\times C$), the \textbf{composition} of $U$ and $V$ is denoted $V\circ U$. Which consists of all ordered pairs $\langle a,c\rangle\in A\times C$ such that for some $b\in B$:
      \begin{equation*}
      \langle a,b\rangle\in U\text{ and }\langle b,c\rangle\in V
      \end{equation*}
    \end{note}
  }
}

\longnewglossaryentry{equivalencerelation}
{
  name=Equivalence Relation,
  text=equivalence relation
} {
  A relation $R$ in a set $A$ (a subset $R$ of $A\times A$), is an \textbf{equivalence relation} iff it satisfies the following axioms:
    \begin{note}[Reflexive]
      For every $a\in A$, $\langle a,a\rangle\in R$
    \end{note}
    \begin{note}[Symmetric]
      If $\langle a,b\rangle\in R$, then $\langle b,a\rangle\in R$
    \end{note}
    \begin{note}[Transitive]
      If $\langle a,b\rangle\in R$ and $\langle b,c\rangle\in R$, then $\langle a,c\rangle\in R$
    \end{note}
}

\longnewglossaryentry{function}
{
  name=Function,
  text=function,
  symbol={$\rightarrow$}
}{
  A \textbf{function} $f$ is a mathematical entity with the
  following properities:
  \begin{enumerate}
    \item $f$ has a \textbf{domain} and \textbf{codomain}, each of
      which must be a set.
    \item For every element $x$ of the domain, $f$ has a
        \textbf{value} at $x$, which is an element of the codomain and is
      denoted $f(x)$.
    \item The domain, the codomain, and the value $f(x)$ for each
      $x$ in the domain are all determined completely by the function.
    \item Conversely, the data consisting of the domain, the
        codomain, and the value $f(x)$ for each element $x$ of the domain
        completely determine the function $f$.
  \end{enumerate}
  The domain and codomain are often called the \textbf{source} and
  \textbf{target} of $f$.
  \begin{note}[Range]
    of $f:A\rightarrow B$, denoted by $f[A]$, is the set of images: $f[A]=\{f(a):A\in A$.
  \end{note}
  \begin{note}[Equivalence]
    Two functions $f:A\rightarrow B$ and $g:A\rightarrow B$ are equal, written $f=g$, iff $f(a)=g(a)$ for every $a\in A$ (i.e. iff they have the same graph).
  \end{note}
  \begin{note}[Negation]
    of $f=g$ is written $f\neq g$ and is the statement $\exists a\in A$ for which $f(a)\neq g(a)$.
  \end{note}
  \begin{note}[Relation]
    A subset $f$ of $A\times B$ (a relation from $A$ to $B$) is a function iff for each $a\in A$ appears as the first coordinate in exactly one ordered pair $\langle a,b\rangle\in f$.
  \end{note}
  \begin{note}[Composition]
    Given functions $f:A\rightarrow B$ and $g:B\rightarrow C$, the function $g\circ f:A\rightarrow C$ (called the \textbf{composition} of and $f$ and $g$ maps the element $a\in A$ into the element $g(f(a))\in C$.
  \end{note}
  \begin{note}[Inverse]
    In general, the inverse relation $f^{-1}$ of a function $f\subseteq A\times B$ need not be a function. However, if $f$ is both \textit{injective} and \textit{sujective}, then $f^{-1}$ is a function from $B$ to $A$ and is the \textbf{inverse function}.
    \begin{equation*}
      f^{-1}\circ f=id_A\text{ and }f\circ f^{-1}=id_B
    \end{equation*}
  \end{note}
}

\longnewglossaryentry{homomorphism}
{
  name=Homomorphism,
  text=homomorphism
} {
  is a structure preserving map between two \textit{algebraic
  structures} of the same type (such as \textit{groups, graphs, rings, vector
  spaces})\\

  \begin{note}[Graph Homomorphism]
  \item[]
    For a graph $\mathscr{G}$, let $G_0$ denote the set of nodes and $G_1$
    denote the set of paths/arrows. A \textit{graph homomorphism} $\phi$
    from a graph $\mathscr{G}$ to a graph $\mathscr{H}$ denoted
    $\phi:\mathscr{G}\rightarrow\mathscr{H}$, is a pair of functions
    ${\phi}_0:\mathscr{G}_0\rightarrow\mathscr{H}_0$ and
    ${\phi}_1:\mathscr{G}_1\rightarrow\mathscr{H}_1$ with the property that
    if $u:m\rightarrow n$ is an arrow of $\mathscr{G}$, then
    ${\phi}_1(u):{\phi}_0(m)\rightarrow{\phi}_0(n)$ in $\mathscr{H}$.
  \end{note}

  \begin{note}[Monoid Homomorphism]
  \item[]
    A \textit{monoid homomorphism} between two monoids $(M, \ast)$ and $(N,
    \bullet)$ is a function $f:M\rightarrow N$ such that $f(x\ast
    y)=f(x)\bullet f(y)$ for all $x,y\in M$ and $f(e_M)=e_N$.
  \end{note}
}

\longnewglossaryentry{induction}

  text=induction

  \begin{itemize}[align=left]
    \item[Weak Induction] Prove
        $\forall n\in\mathbb{N}P(n)$.
        \begin{itemize}[align=left]

      $\mathbb{N}$

    equivalent to each other. In general, attempt a proof with weak induction,
    if a stronger inductive hypthothesis is required, use strong induction.
    It is important that the initial step must lead to the induction step (ie:
    $P(1)$ leads to $P(2)$, $P(1)$ is true leads to $P(2)$ being true).
  \end{note}

\longnewglossaryentry{identityelement}
{
  name=Identity Element,
  text=identity element,
} {
  An element $e\in\Omega$ is said to be an \textbf{identity
  element} under the binary operator $o$ if and only if $\forall
  a\in\Omega$:
  \begin{equation*}
    a\circ e=e\circ a=a
  \end{equation*}
  In order for $e$ to be an \textbf{identity element}, $e$ must statisfy
  the following:
  \begin{enumerate}
    \item $e\in\Omega$
    \item $\forall a\in\Omega, a\circ e=a$
    \item $\forall a\in\Omega, e\circ a=a$
  \end{enumerate}
  ie: the identity element $e$ must be in $\Omega$ and commute with every
  element in $\Omega$.
  \begin{note}
    if the binary operator $\circ$ is abelian, only conditions
    $1.$ and $2.$ or $3.$ is necessary.
  \end{note}
  To prove, let $a\in\Omega$ be arbitrary but fixed. Compute $a\circ x$,
  $x\circ a$ and solve for $x$ where $a\circ x=x\circ a$. Show that
  $x\in\Omega$ does not depend on $a$. Conclude $e=x$ is an identity
  element in $\Omega$ under the binary operator $\circ$.
  \begin{note}[Theorem]
    If $\Omega$ is closed under the $\circ$ binary operation and $e$ is an
    identity element under $\circ$, then $e$ is unique
  \end{note}
}

\longnewglossaryentry{inverseelement}
{
  name=Inverse Element,
  text=inverse element
} {
  An element $a\in\Omega$ is said to have an \textbf{inverse element}
  $a^{-1}\in\Omega$ under the binary operator $\circ$ if and only if:
  \begin{equation*}
    a\circ a^{-1}=a^{-1}\circ a=e
  \end{equation*}
  where $e$ is the identity element in $\Omega$.\\

  To prove, determine the identity element $e$ and let $a\in\Omega$ be
  arbitrary but fixed. Compute $a\circ x$, $x\circ a$, solve for $x$
  where $a\circ x=e=x\circ a$ and show that $x\in\Omega$. Conc
  that $a^{-1}=x$ is the inverse of the element $a$ under the binary
  operator $\circ$.
  \begin{note}[Theorem]
    Let $o$ be an associative binary operator. If $\Omega$ is closed under
    $\circ$ and $a^{-1}\in\Omega$ whenever $a\in\Omega$, then $a^{-1}$ is
    unique
  \end{note}
  \begin{note}[Theorem]
    If $\Omega$ is closed under $\circ$ and $a^{-1}\in\Omega$ whenever
    $a\in\Omega$, then $(a^{-1})^{-1}=a$
  \end{note}
}

\longnewglossaryentry{orderedset}

  \end{note}

    An element $a_0\in X$ is \textbf{maximal} iff $a_0\preceq x$ implies $x=a$, i.e. if no element follows $a_0$ except itself. Similarly, an element $b_0\in X$ is \textbf{minimal} iff $x\preceq b_0$ implies $x=b_0$, i.e. if no element precedes $b_0$ except itself.

\longnewglossaryentry{partiallyorderedset}
{
  name=Partially Ordered Set/Poset,
  text=partially ordered set
} {
  is a \textit{preordered set} $(X,\preceq)$ where $\preceq$ is
  \textit{antisymmetric}, (ie: a set $X$ with a partial order).
  \begin{note}[Inverse Order]
    If a relation $R$ in a set $A$ defines a partial order, then the inverse relation $R^{-1}$ is also a partial order; it is called the \textbf{inverse order}.
  \end{note}
  \begin{note}[Upper and Lower Bounds]
    Let $A\subseteq X$. An element $m\in X$ is a \textbf{lower bound} of $A$ iff $m\preceq x$ for all $x\in A$, i.e. if $m$ precedes every element in $A$. If some lower bound of $A$ follows every other lower bound of $A$, then it is called the \gls{infimum} or \textit{greatest lower bound}. Similarly, an element $M\in X$ is an \textbf{upper bound} of $A$ iff $x\preceq M$ for all $x\in A$, i.e. if $M$ follows every element in $A$. If some upper bound of $A$ precedes every other upper bound of $A$, then it is called the \gls{supremum} or \textit{least upper bound}.
  \end{note}
  \begin{note}
    $A$ is said to be \textit{bounded above} if it has an upper bound, and \textit{bounded below} if it has a lower bound. If $A$ has both an upper and lower bound, then it is said to be \textit{bounded}.
  \end{note}

    and join. Thus, every \textbf{complete} lattice is bounded.
    \begin{note}
      Every finite lattice is complete.
    \end{note}
  }
}

\newglossaryentry{semilattice}
{
  name=Semilattice,
  text=semilattice,
  description={A poset with a join for any nonempty finite subset is a
    \textbf{join-semilatice/upper semilattice}. Dually, a poset with a meet
    for any nonempty finite subset is a {meet-semilattice/lower
  semilattice}.}
}

\longnewglossaryentry{booleanalgebra}
{
  name=Boolean Algebra,
  text=boolean algebra
} {
  is a poset $B$ that satisfies the following:
  \begin{itemize}
    \item $\forall x,y\in B$, there exists a \textit{supremum}, $x\vee y$.
    \item $\forall x,y\in B$, there exists a \textit{infimum}, $x\wedge y$.
    \item $\wedge$ distributes over $\vee$ such that $\forall x,y,z\in B,
      x\wedge(y\vee z)=(x\wedge y)\vee(x\wedge z)$.
    \item $B$ contains the least element $0$ and the greatest element $1$ of
      $B$.
    \item Each element $x$ has a \textit{complement} $\neg x$, such that
      $x\wedge\neg x=0$ and $x\vee\neg x=1$.
  \end{itemize}

  \begin{note}
    By definition, a Boolean Algebra is a lattice (thus also known as Boolean
    lattice).
  \end{note}
}

\longnewglossaryentry{semidirectproduct}
{
  name=Semidirect Product,
  text=semidirect product
} {
  The \textbf{semidirect product} of monoids $M$ and $T$ with the action
  $\alpha:M\times T\rightarrow T$ is the monoid with the underlying set
  $T\times M$ where $\alpha$ is defined as:
  \begin{itemize}
    \item $\forall m\in M, \alpha(m,1_T)=1_T$
    \item $\forall m\in M, \forall t,t'\in T,
      \alpha(m,tt')=\alpha(m,t)\alpha(m,t')$
    \item $\forall t\in T, \alpha(1_M,t)=t$
    \item $\forall m,m'\in M, \forall t\in T,
      \alpha(mm',t)=\alpha(m,\alpha(m',t))$
  \end{itemize}
  and multiplication of the semidirect product is:
  \begin{equation*}
    (t,m)(t',m')=(t\alpha(m,t'),mm')
  \end{equation*}
  for all $t,t'\in T$ and $m,m'\in M$.

  \begin{note}
    The action $\alpha$ can be thought of a monoid homomorphism
    $\phi:M\rightarrow\mathbf{End}(T)$ where $\mathbf{End}(T)$ is the monoid
    of endomorphisms of $T$.
  \end{note}
}

\longnewglossaryentry{heytingalgebra}
{
  name=Heyting Algebra,
  text=heyting algebra
} {
  is a bounded lattice where $\vee$ and $\wedge$ are the join and meet
  operations and has an implication binary operation
  $a\Rightarrow b$ that satisifies the condition $(x\wedge a)\leq b$ if and
  only if $x\leq(a\Rightarrow b)$.

  \begin{note}
    A heyting algebra is \textit{complete} if its lattice structure is
    complete.
  \end{note}
}

    \begin{equation*}
      A\setminus B=\{x:x\in A, x\notin B\}

  }

\newglossaryentry{absolutecomplement}

\newglossaryentry{indexedset}
{
  name=Indexed Set,
  text=indexed set,
  description={
    An \textbf{indexed class of sets} $\mathscr{A}$, denoted by:
    \begin{equation*}
      \{A_i:i\in I\}\text{, }\{A_i\}_{i\in I}\text{ or simply }\{A_i\}
    \end{equation*}
    assigns a set $A_i$ to each $i\in I$ (a function from $I$ into a class of sets). The set $I$ is called the \textit{index set}, the sets $A_i$ are called \textbf{indexed sets}, and each $i\in I$ is called an \textit{index}.
    \begin{note}[Sequence]
      When the index set $I$ is the set of positive integers, the indexed class $\{A_1, A_2, \dots\}$ is called a \textbf{sequence} (of sets).
      We write
      \begin{equation*}
        \bigcup^{\infty}_{i=1}A_i=A_1\cup A_2\cup\dots\text{ and }\bigcap^{\infty}_{i=1}A_i=A_1\cap A_2\cap\dots
      \end{equation*}
      for the union and intersection respectively.
    \end{note}
    \begin{note}[Cartesian Product]
      for an indexed class of sets $\mathscr{A}$ is denoted by:
      \begin{equation*}
        \prod\{A_i:i\in I\}\text{ or }\prod_{i\in I}A_i\text{ or simply }\prod_iA_i
      \end{equation*}
      is the set of all functions $p:I\rightarrow\cup_iA_i$ such that $p(i)=a_i\in A_i$. We denote such an element of the Cartesian product by $p=\langle a_i:i\in I\rangle$. For each $i_0\in I$ there exists an $i_0$th \textit{projection function}, $\pi_{i_0}$, from the product set $\prod_iA_i$ into the $i_0$th \textit{coordinate set} $A_{i_0}$ defined by:
      \begin{equation*}
        \pi_{i_0}(\langle a_i:i\in I\rangle)=a_{i_0}
      \end{equation*}
    \end{note}
    \begin{note}[Union]
      for an indexed class of sets $\mathscr{A}$ is denoted by:
      \begin{equation*}
        \bigcup\{A_i:i\in I\}\text{, }\bigcup_{i\in I}A_i\text{ or }\cup_iA_i
      \end{equation*}
    \end{note}
    \begin{note}[Intersection]
      for an indexed class of sets $\mathscr{A}$ is denoted by:
      \begin{equation*}
        \bigcap\{A_i:i\in I\}\text{, }\bigcap_{i\in I}A_i\text{ or }\cap_iA_i
      \end{equation*}
    \end{note}
  }
}

\newglossaryentry{setunion}
{
  name=Set Union,
  text=set union,
  description={
    Let $\mathscr{A}$ be any class of subsets of the universal set $U$. The \textbf{union} of the sets in $\mathscr{A}$, denoted by $\bigcup\{A:A\in\mathscr{A}\}$, is the set of elements which belong to at least one set in $\mathscr{A}$:
    \begin{equation*}
      \bigcup\{A:A\in\mathscr{A}\}=\{x:x\in U,\exists A\in\mathscr{A}\text{ s.t. }x\in A\}
    \end{equation*}
    \begin{note}
      $\bigcup\{A:A\in\emptyset\}=\emptyset$ and $\bigcup\{A_i:i\in\emptyset\}=\emptyset$. Ref: \cite{empty_intersection_empty_union}
    \end{note}
  },
  symbol={$\bigcup$}
}

\newglossaryentry{setintersection}
{
  name=Set Intersection,
  text=set intersection,
  description={
    Let $\mathscr{A}$ be any class of subsets of the universal set $U$. The \textbf{intersection} of the sets in $\mathscr{A}$, denoted by $\bigcap\{A:A\in\mathscr{A}\}$, is the set of elements which belong to every set in $\mathscr{A}$:
    \begin{equation*}
      \bigcap\{A:A\in\mathscr{A}\}=\{x:x\in U,\forall A\in\mathscr{A},x\in A\}
    \end{equation*}
    \begin{note}
      $\bigcap\{A:A\in\emptyset\}=U$ and $\bigcap\{A_i:i\in\emptyset\}=U$. Ref: \cite{empty_intersection_empty_union}
    \end{note}
  },
  symbol={$\bigcap$}
}

\newglossaryentry{continuum}
{
  name=Continuum,
  text=continuum,
  description={
    A set $X$ is said to have the \textbf{power of the continuum} or is said to have \textbf{cardinality c} iff it is equivalent to the unit interval $[0,1]$ (note that not all infinite sets are denumerable, e.g. the unit interval $[0,1]$ is non-denumerable).
    \begin{note}
      $\mathbb{R}$, the set of real numbers, has cardinality \textbf{c}. Every interval in $\mathbb{R}$, open or closed, has cardinality \textbf{c}. And hence, any interval is equivalent to $\mathbb{R}$.
    \end{note}
    \begin{note}[Continuum Hypothesis]
      There does not exist a set $A$ with the property that $\aleph_0<\#(A)<\textbf{c}$.
    \end{note}
  }
}
\longnewglossaryentry{schroederbernstein}
{
  name=Schroeder-Bernstein Theorem,
  text=schroeder-bernstein theorem
} {
  We write $A\preceq B$ if $A$ is equivalent to a subset of $B$,
  \begin{equation*}
    A\preceq B\text{ iff }\exists B^*\subseteq B\text{ s.t. }A\sim B^*
  \end{equation*}
  The \textbf{Schroeder-Berstein Theorem} states that $A\preceq B$ and $B\preceq A$, then $A\sim B$. This can also be restated as:
  \begin{equation*}
    \text{Let }X\supseteq Y\supseteq X_1\text{ and let }X\sim X_1\text{, then }X\sim Y
  \end{equation*}
  In terms of cardinality, this can also be stated as: If $\#(A)\leq\#(B)$ and $\#(B)\leq\#(A)$, then $\#(A)=\#(B)$.

  \begin{note}
    We also write $A\prec B$ if $A\preceq B$ but $A\nsim B$, if $A$ is not equivalent to $B$.
  \end{note}
  \begin{note}[Example]
    Since $\mathbb{N}\subset\mathbb{R}$, $\mathbb{N}\preceq\mathbb{R}$. In fact, $\mathbb{R}$ is not denumerable, i.e. $\mathbb{R}\nsim\mathbb{N}$, and thus $\mathbb{N}\prec\mathbb{R}$.
  \end{note}
  \begin{note}[Law of Trichotomy]
    Given any pair of sets $A$ and $B$, either $A\prec B$, $A\sim B$ or $B\prec A$.
  \end{note}
  \begin{note}
    If $A\prec B$ then we say that $A$ has cardinality less than $B$ or $B$ has cardinality greater than $A$.
    \begin{equation*}
      \#(A)<\#(B)\text{ iff }A\prec B
    \end{equation*}
  \end{note}
}

\newglossaryentry{zornslemma}
{
  name=Zorn's Lemma,
  text=zorn's lemma,
  description={Let $X$ be a non-empty poset in which every totally ordered subset has an upper bound. Then $X$ contains at least one maximal element.
    \begin{note}[Remark]
      This is equivalent to the classical \textit{Axiom of Choice} and the \textit{Well-ordering Principle}.
    \end{note}
  }
}
