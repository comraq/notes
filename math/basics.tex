\longnewglossaryentry{function}
{
  name=Function,
  text=function,
  symbol={$\rightarrow$}
}{A \textbf{function} $f$ is a mathematical entity with the
  following properities:
  \begin{enumerate}
    \item $f$ has a \textbf{domain} and \textbf{codomain}, each of
      which must be a set.
    \item For every element $x$ of the domain, $f$ has a
        \textbf{value} at $x$, which is an element of the codomain and is
      denoted $f(x)$.
    \item The domain, the codomain, and the value $f(x)$ for each
      $x$ in the domain are all determined completely by the function.
    \item Conversely, the data consisting of the domain, the
        codomain, and the value $f(x)$ for each element $x$ of the domain
        completely determine the function $f$.
  \end{enumerate}
  The domain and codomain are often called the \textbf{source} and
\textbf{target} of $f$.}

\longnewglossaryentry{homomorphism}
{
  name=Homomorphism,
  text=homomorphism
} {
  is a structure preserving map between two \textit{algebraic
  structures} of the same type (such as \textit{groups, graphs, rings, vector
  spaces})\\

  \begin{note}[Graph Homomorphism]
  \item[]
    For a graph $\mathscr{G}$, let $G_0$ denote the set of nodes and $G_1$
    denote the set of paths/arrows. A \textit{graph homomorphism} $\phi$
    from a graph $\mathscr{G}$ to a graph $\mathscr{H}$ denoted
    $\phi:\mathscr{G}\rightarrow\mathscr{H}$, is a pair of functions
    ${\phi}_0:\mathscr{G}_0\rightarrow\mathscr{H}_0$ and
    ${\phi}_1:\mathscr{G}_1\rightarrow\mathscr{H}_1$ with the property that
    if $u:m\rightarrow n$ is an arrow of $\mathscr{G}$, then
    ${\phi}_1(u):{\phi}_0(m)\rightarrow{\phi}_0(n)$ in $\mathscr{H}$.
  \end{note}

  \begin{note}[Monoid Homomorphism]
  \item[]
    A \textit{monoid homomorphism} between two monoids $(M, \ast)$ and $(N,
    \bullet)$ is a function $f:M\rightarrow N$ such that $f(x\ast
    y)=f(x)\bullet f(y)$ for all $x,y\in M$ and $f(e_M)=e_N$.
  \end{note}
}

  \begin{itemize}[align=left]

    \item [Proof by Contrapositive] State the contrapositive to prove; if

  \end{itemize}

\longnewglossaryentry{induction}
{
  name=Mathematical Induction,
  text=induction
} {A special type of direct proof that can often be used with theorems of
  the nature:\\
  The statement $P(n)$ holds for every natural number $n$ (ie: holds for a
  statement indexed by $n$ for all $n\in\mathbb{N}$).
  \begin{itemize}[align=left]
    \item[Weak Induction] Prove
        $\forall n\in\mathbb{N}P(n)$.
        \begin{itemize}[align=left]
          \item[Initial Step (Base Case)] show $P(1)$ is true
          \item[Induction Step] if $P(k)$ is true for an arbitrary but fixed
            (ABF) value of $k$ in $\mathbb{N}$, then it follows that $P(k+1)$
            is also true
        \end{itemize}
    \item[Strong Induction] Same as \textit{Weak Induction} except in the
      \textit{Induction Step}: instead of assuming only $P(k)$ is true,
      assume that $P(1),\dots,P(k)$ are all true for an ABF value of $k$ in
      $\mathbb{N}$
  \end{itemize}
  \begin{note}
    \textit{Weak} and \textit{Strong} inductions are logically
    equivalent to each other. In general, attempt a proof with weak induction,
    if a stronger inductive hypthothesis is required, use strong induction.
    It is important that the initial step must lead to the induction step (ie:
    $P(1)$ leads to $P(2)$, $P(1)$ is true leads to $P(2)$ being true).
  \end{note}
}

\longnewglossaryentry{identityelement}
{
  name=Identity Element,
  text=identity element,
} {
  An element $e\in\Omega$ is said to be an \textbf{identity
  element} under the binary operator $o$ if and only if $\forall
  a\in\Omega$:
  \begin{equation*}
    a\circ e=e\circ a=a
  \end{equation*}
  In order for $e$ to be an \textbf{identity element}, $e$ must statisfy
  the following:
  \begin{enumerate}
    \item $e\in\Omega$
    \item $\forall a\in\Omega, a\circ e=a$
    \item $\forall a\in\Omega, e\circ a=a$
  \end{enumerate}
  ie: the identity element $e$ must be in $\Omega$ and commute with every
  element in $\Omega$.
  \begin{note}
    if the binary operator $\circ$ is abelian, only conditions
    $1.$ and $2.$ or $3.$ is necessary.
  \end{note}
  To prove, let $a\in\Omega$ be arbitrary but fixed. Compute $a\circ x$,
  $x\circ a$ and solve for $x$ where $a\circ x=x\circ a$. Show that
  $x\in\Omega$ does not depend on $a$. Conclude $e=x$ is an identity
  element in $\Omega$ under the binary operator $\circ$.
  \begin{note}[Theorem]
    If $\Omega$ is closed under the $\circ$ binary operation and $e$ is an
    identity element under $\circ$, then $e$ is unique
  \end{note}
}

\longnewglossaryentry{inverseelement}
{
  name=Inverse Element,
  text=inverse element
} {
  An element $a\in\Omega$ is said to have an \textbf{inverse element}
  $a^{-1}\in\Omega$ under the binary operator $\circ$ if and only if:
  \begin{equation*}
    a\circ a^{-1}=a^{-1}\circ a=e
  \end{equation*}
  where $e$ is the identity element in $\Omega$.\\

  To prove, determine the identity element $e$ and let $a\in\Omega$ be
  arbitrary but fixed. Compute $a\circ x$, $x\circ a$, solve for $x$
  where $a\circ x=e=x\circ a$ and show that $x\in\Omega$. Conc
  that $a^{-1}=x$ is the inverse of the element $a$ under the binary
  operator $\circ$.
  \begin{note}[Theorem]
    Let $o$ be an associative binary operator. If $\Omega$ is closed under
    $\circ$ and $a^{-1}\in\Omega$ whenever $a\in\Omega$, then $a^{-1}$ is
    unique
  \end{note}
  \begin{note}[Theorem]
    If $\Omega$ is closed under $\circ$ and $a^{-1}\in\Omega$ whenever
    $a\in\Omega$, then $(a^{-1})^{-1}=a$
  \end{note}
}

\newglossaryentry{transitive}

  text=transitive,
  description={is a binary relation $alpha$ over a set $S$ if whenever an
    element $a$ is related to an element $b$, and $b$ is related to an
    element $c$, then $a$ is also related to $c$. Thus, $\forall a,b,c\in S,
  (a\alpha b\wedge b\alpha c)\Rightarrow a\alpha c$.}

\longnewglossaryentry{supremum}
{
  name=Supremum,
  text=supremum
} {
  is the \textit{least upper bound} of a subset $T$ of a poset $(S,\leq)$.
  That is, for a subset $T\subseteq S$, a \textbf{supremum} is an element
  $v\in S$ such that:
  \begin{itemize}
    \item $\forall t\in T, t\leq v$
    \item If $w\in S$ has the property that $t\leq w$ for every $t\in T$,
      then $v\leq w$
  \end{itemize}
}

\newglossaryentry{infimum}
{
  name=Infimum,
  text=infimum,
  description={is the \textit{greatest lower bound} of a subset $T$ of a
  poset $S$, if such an element exists.}
}

\longnewglossaryentry{path}
{
  name=Path,
  text=path
} {
  In a graph $\mathscr{G}$, a \textbf{path} from a node $x$ to
  a node $y$ of length $k$ is a sequence $(f_1,f_2,\dots,f_k)$ of directed
  edges for which:
  \begin{itemize}
    \item $source(f_k)=x$
    \item $target(f_i)=source(f_{i-1})$ for $i=2,\dots,k$
    \item $target(f_1)=y$
  \end{itemize}
  The empty path is denoted as $()$.
}

    \end{note}
  }

\longnewglossaryentry{groupaction}
{
  name=Group Action,
  text=group action
} {
  is a way of interpreting elements of group ``acting'' on some space, thus
  if $G$ is a group and $X$ is a set, the action of $G$ on $X$ is a group
  homomorphism from $G$ to the symmetry group on $X$. The group action
  $\alpha:G\times X\rightarrow X$ may be defined as follows:
  \begin{itemize}
    \item $\alpha(1, x)=x$ where $1$ is the identity in $G$ for any $x\in
      X$\\
      Also written as: $1x=x$
    \item $\alpha(gh,x)=\alpha(g,\alpha(h,x))$ for all $g,h\in G, x\in X$\\
      Also written as: $(gh)x=g(hx)$
  \end{itemize}

  \begin{note}
    A similar construct can be defined for monoids.
  \end{note}

  \begin{note}
    One way to think of group actions would be the set $X$ is a
    \textit{state space} and elements of $G$ acting on $X$ induces
    \textit{transitions} from one state to another.
  \end{note}
}

\newglossaryentry{lattice}
{
  name=Lattice,
  text=lattice,
  description={is a poset in which every pair of elements have an unique
    supremum and infimum.
    \begin{note}
      A lattice $L$ is \textit{bounded} if it has a greatest element $1$ and least
      element $0$ such that for every element $x\in L$, $0\leq x\geq 1$.
    \end{note}
    \begin{note}
      A lattice is \textit{complete} if any subset of elements have a meet
      and join. Thus, every complete lattice is bounded.
    \end{note}
  }
}

    for any nonempty finite subset is a {meet-semilattice/lower
  semilattice}.}
}

\longnewglossaryentry{booleanalgebra}
{
  name=Boolean Algebra,
  text=boolean algebra
} {
  is a poset $B$ that satisfies the following:
  \begin{itemize}
    \item $\forall x,y\in B$, there exists a \textit{supremum}, $x\vee y$.
    \item $\forall x,y\in B$, there exists a \textit{infimum}, $x\wedge y$.
    \item $\wedge$ distributes over $\vee$ such that $\forall x,y,z\in B,
      x\wedge(y\vee z)=(x\wedge y)\vee(x\wedge z)$.
    \item $B$ contains the least element $0$ and the greatest element $1$ of
      $B$.
    \item Each element $x$ has a \textit{complement} $\neg x$, such that
      $x\wedge\neg x=0$ and $x\vee\neg x=1$.
  \end{itemize}

  \begin{note}
    By definition, a Boolean Algebra is a lattice (thus also known as Boolean
    lattice).
  \end{note}
}

\longnewglossaryentry{semidirectproduct}
{
  name=Semidirect Product,
  text=semidirect product
} {
  The \textbf{semidirect product} of monoids $M$ and $T$ with the action
  $\alpha:M\times T\rightarrow T$ is the monoid with the underlying set
  $T\times M$ where $\alpha$ is defined as:
  \begin{itemize}
    \item $\forall m\in M, \alpha(m,1_T)=1_T$
    \item $\forall m\in M, \forall t,t'\in T,
      \alpha(m,tt')=\alpha(m,t)\alpha(m,t')$
    \item $\forall t\in T, \alpha(1_M,t)=t$
    \item $\forall m,m'\in M, \forall t\in T,
      \alpha(mm',t)=\alpha(m,\alpha(m',t))$
  \end{itemize}
  and multiplication of the semidirect product is:
  \begin{equation*}
    (t,m)(t',m')=(t\alpha(m,t'),mm')
  \end{equation*}
  for all $t,t'\in T$ and $m,m'\in M$.

  \begin{note}
    The action $\alpha$ can be thought of a monoid homomorphism
    $\phi:M\rightarrow\mathbf{End}(T)$ where $\mathbf{End}(T)$ is the monoid
    of endomorphisms of $T$.
  \end{note}
}

\longnewglossaryentry{heytingalgebra}
{
  name=Heyting Algebra,
  text=heyting algebra
} {
  is a bounded lattice where $\vee$ and $\wedge$ are the join and meet
  operations and has an implication binary operation
  $a\Rightarrow b$ that satisifies the condition $(x\wedge a)\leq b$ if and
  only if $x\leq(a\Rightarrow b)$.

  \begin{note}
    A heyting algebra is \textit{complete} if its lattice structure is
    complete.
  \end{note}
}
