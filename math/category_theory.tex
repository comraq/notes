\newglossaryentry{homfunction}
{
  name=Hom Function,
  text=hom function,
  description={
    Let $S$ be an object and $f:T\rightarrow V$ an arrow in a
    category, a \textit{covariant} \textbf{hom function}
    $\Hom(S,f):\Hom(S,T)\rightarrow\Hom(S,V)$, if defined by:
    \begin{center}
      $\Hom(S,f)(g)=f\circ g$
    \end{center}
    for some $g\in\Hom(S,T)$.\\
%
    Similarily, a \textit{contravariant} \textbf{hom function}
    $\Hom(f,S):\Hom(V,S)\rightarrow\Hom(T,S)$, if defined by:
    \begin{center}
      $\Hom(f,S)(g)=g\circ f$
    \end{center}
    for some $g\in\Hom(V,S)$.
  }
}

\longnewglossaryentry{category}
{
  name=Category,
  text=category
} {
  A \textbf{category} $\mathscr{C}$ contains a collection of
  \textit{objects} (denoted as $\mathscr{C}_0$) and a collection of
  \textit{morphisms} (denoted as $\mathscr{C}_1$) where every morphism has
  $source, target:\mathscr{C}_1\rightarrow\mathscr{C}_0$ functions mapping
  each arrow to its \textit{source} and \textit{target} objects respectively
  into the collection of objects.\\

  For any two arrows $f,g$ where the $target(f)=source(g)$, the
  \textit{composition} of $f$ and $g$ are written as $g\circ
  f:source(f)\rightarrow target(g)\in\mathscr{C}_1$. Composition is
  associative, $(h\circ g)\circ f=h\circ (g\circ f)$ when either side is
  defined.\\

  $\forall A\in\mathscr{C}_0$, there exists the \textit{identity} arrow,
  $id_A:A\rightarrow A$ and for any $f:A\rightarrow B$, $f\circ id_A=id_B\circ
  f=f$.\\

  A category is \textit{small} if its objects and arrows constitute sets;
  otherwise it is \textit{large}.
}

\longnewglossaryentry{subcategory}
{
  name=Subcategory,
  text=subcategory
} {
  A \textbf{subcategory} $\mathscr{D}$ of a category
  $\mathscr{C}$ is a category for which:
  \begin{itemize}
    \item All objects and arrows in $\mathscr{D}$ are objects and arrows in
      $\mathscr{C}$ (ie: $\mathscr{D}_0\subseteq\mathscr{C}_0$ and
      $\mathscr{D}_1\subseteq\mathscr{C}_1$).

    \item The source and target of an arrow in $\mathscr{D}$ are the same as
      its souce and target in $\mathscr{C}$ (ie: source and target maps in
      $\mathscr{D}$ are restrictions of those in $\mathscr{C}$). Thus,
      $\forall A,B\in\mathscr{C}_0$,
      $\Hom_{\mathscr{D}}(A,B)\subseteq\Hom_{\mathscr{C}}(A,B)$.

    \item If $A$ is an object in $\mathscr{C}$, its identity arrow $id_A$ in
      $\mathscr{C}$ is in $\mathscr{D}$.

    \item If $f:A\rightarrow B$ and $g:B\rightarrow C$ in $\mathscr{D}$,
      then the composite $g\circ f\in\mathscr{C}_1$ is also the composite in
      $\mathscr{D}$.
  \end{itemize}
}

\newglossaryentry{productofcategories}
{
  name=Product of Categories,
  text=product of categories,
  description={If $\mathscr{C}$ and $\mathscr{D}$ are categories their
    \textbf{product} $\mathscr{C}\times\mathscr{D}$ is the category whose
    objects are all ordered pairs $(C,D)$ where $C\in\mathscr{C}_0,
    D\in\mathscr{D}_0$ and arrows are $(f,g):(C,D)\rightarrow (C',D')$ where
    $f:C\rightarrow C'\in\mathscr{C}_1, g:D\rightarrow D'\in\mathscr{D}_1$.
    The identity arrow of $(C,D)$ is $(id_C,id_D)$ and composition of arrows is
  defined component wise.}
}

\longnewglossaryentry{categoricaldual}
{
  name=Categorical Dual,
  text=categorical dual,
  symbol={$^{op}$}
} {
  Given any category $\mathscr{C}$, $\mathscr{C}^{op}$ is the
  \textbf{dual category} of $\mathscr{C}$ where all the arrows are reversed.
  $\mathscr{C}^{op}$ is defined as follows:
  \begin{itemize}
    \item The objects and arrows of $\mathscr{C}^{op}$ are the objects and
      arrows of $\mathscr{C}$.

    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $f:B\rightarrow A$ in
      $\mathscr{C}^{op}$.

    \item If $h=g\circ f$ in $\mathscr{C}$, then $h=f\circ g$ in
      $\mathscr{C}^{op}$.
  \end{itemize}
  Since identity arrows have the same source and target objects, identity
  arrows in $\mathscr{C}$ are the same as in $\mathscr{C}^{op}$ and thus
  $\mathscr{C}^{op}$ is also a category.
}

\longnewglossaryentry{slicecategory}
{
  name=Slice Category,
  text=slice category,
  symbol={$/$}
} {
  If $\mathscr{C}$ is a category, for any object $A\in\mathscr{C}_0$, the
  \textbf{slice category} $\mathscr{C}/A$ is defined as follows:
  \begin{itemize}
    \item An object of $\mathscr{C}/A$ is an arrow $f:C\rightarrow A$ in
      $\mathscr{C}$ for some $C\in\mathscr{C}_0$.

    \item An arrow of $\mathscr{C}/A$ from $f:C\rightarrow A$ to
      $f':C'\rightarrow A$ is an arrow $h:C\rightarrow C'$ such that
      $f=f'\circ h$.

    \item The composite of $h:f\circ f'$ and $h':f'\circ f''$ is $h'\circ
      h$.
  \end{itemize}

  \begin{note}
    Since the same $h$ can satisfy both $f=f'\circ h$ and $g=g'\circ h$
    where $f\neq g$ or $f'\neq g'$, $h:f\rightarrow f'$ and $h:g\rightarrow
    g'$ are regarded as different arrows in $\mathscr{C}/A$.
  \end{note}

  \begin{note}
    The \textit{indexed function} of an \textit{indexed set} indexed
    by $S$, is precisely an arrow in the slice category \textbf{Set}$/S$.
  \end{note}
}

\newglossaryentry{monomorphism}
{
  name=Monomorphism,
  text=monomorphism,
  description={is an arrow $f:A\rightarrow B$ such that for any object $T$
    and arrows $x,y:T\rightarrow A$, if $x\neq y$, then $f\circ x\neq f\circ
    y$. Monomorphisms can be denoted by $f:A\rightarrowtail B$.
    \begin{note}
      In this context, $x,y:T\rightarrow A$ may be regarded as
      \textit{variable elements} and can be written as if $x\neq y$, then
      $f(x)\neq f(y)$.
    \end{note}
    \begin{note}[Properties]
      Let $f:A\rightarrow B$ and $g:B\rightarrow C$ be arrows:
      \begin{itemize}
        \item If $f$ and $g$ are monomorphisms, so is $g\circ f$.
        \item If $g\circ f$ is a monomorphism, so is $f$.
      \end{itemize}
    \end{note}
    \begin{note}
      In \textbf{Set}, a function is injective if and only if it is a
      monomorphism.
    \end{note}
    \begin{note}
      In the category determined by a Monoid, a monic element is left
      cancellable.
    \end{note}
  },
  symbol={$\rightarrowtail$}
}

\newglossaryentry{epimorphism}
{
  name=Epimorphism,
  text=epimorphism,
  description={is the dual to monomorphisms. An arrow $f:S\rightarrow T$ is
    an \textbf{epimorphism} if for any arrows $g,h:T\rightarrow X$, $g\circ
    f=h\circ f$ implies $g=h$. Epimorphisms can be denoted by
    $f:S\twoheadrightarrow T$.
    \begin{note}[Properties]
      Let $f:A\rightarrow B$ and $g:B\rightarrow C$ be arrows:
      \begin{itemize}
        \item If $f$ and $g$ are epimorphisms, so is $g\circ f$.
        \item If $g\circ f$ is an epimorphism, so is $g$.
      \end{itemize}
    \end{note}
    \begin{note}
      In \textbf{Set}, a function is surjective if and only if it is an
      epimorphism.
    \end{note}
    \begin{note}
      In the category determined by a Monoid, an epic element is right
      cancellable.
    \end{note}
  },
  symbol={$\twoheadrightarrow$}
}

\longnewglossaryentry{subobject}
{
  name=Subobject,
  text=subobject
} {
  In a category $\mathscr{C}$, a \textbf{subobject} of an object
  $C\in\mathscr{C}$ is an equivalence class of monomorphisms under the
  relation $\sim$. The subobject is a \textbf{proper subobject} if it does not
  contain $id_C$. The equivalence relation $\sim$ is defined as follows:
  \begin{center}
    For monomorphisms $f_0:C_0\rightarrowtail C$ and $f_1:C_1\rightarrowtail
    C$, $f_0\sim f_1$ if both $f_0$ and $f_1$ factors through each other.
  \end{center}
}

\longnewglossaryentry{functor}
{
  name=Functor,
  text=functor
} {
  is a ``structure preserving'' map between categories, similar to
  homomorphisms between groups, graphs and etc$\dots$ A \textbf{functor}
  $F:\mathscr{C}\rightarrow\mathscr{D}$ from the category $\mathscr{C}$ to
  the category $\mathscr{D}$ defined as follows:
  \begin{itemize}
    \item $\forall A\in\mathscr{C}$, $F(A)\in\mathscr{D}$.
    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $F(f):F(A)\rightarrow
      F(B)$ is an arrow in $\mathscr{D}$ such that:
      \begin{itemize}
        \item $F(id_A)=id_{F(A)}$ for every object $A\in\mathscr{C}$.
        \item $F(g\circ f)=F(g)\circ F(f)$ for all arrows $f:X\rightarrow
          Y, g:Y\rightarrow Z$ in $\mathscr{C}$.
      \end{itemize}
  \end{itemize}

  \begin{note}
    All functors preserve isomorphisms.
  \end{note}
}

\longnewglossaryentry{categoryexamples}
{
  name={Examples of Categories:}
} {
  \begin{itemize}[label={}]
    \item
  \begin{itemize}[align=left]
    \item[\textbf{Set}]
      is the category whose objects are sets and arrows are functions
      between sets. Composition is function composition and the identity
      arrow is the identity function $id_S$ for a set $S$.

    \item[\textbf{Fin}]
      is the category whose objects are finite sets and arrows are all the
      functions between finite sets.

    \item[\textbf{Pfn}]
      is the category whose objects are sets but all arrows are all partial
      functions. If $f:S\rightarrow T, g:T\rightarrow V$ are partial
      functions with $f$ defined on $S_0\subseteq S$ and $g$ defined on
      $T_0\subseteq T$, then the composite $g\circ f:S\rightarrow V$ is the
      partial function from $S$ to $V$ defined on the subset $\{x\in
      S_0|f(x)\in T_0\}$ of $S$ by the requirement $(g\circ f)(x)=g(f(x))$.

    \item[\textbf{Rel}]
      is the category whose objects are sets and arrows are relations
      relations between sets. The composite $\beta\circ\alpha$ of relations
      $\alpha$ from sets $S$ to $T$ and $\beta$ from $T$ to $U$ is a
      relation from $S$ to $U$ defined as follows:
      \begin{center}
        If $x\in S$ and $z\in U$, $(x,z)\in\beta\circ\alpha$ if and only if
        there is an element $y\in T$ for which $(x,y)\in\alpha$ and
        $(y,z)\in\beta$.
      \end{center}
      The identity arrow for a set $S$ is the \textit{diagonal relation}
      $\Delta_S=\{(x,x)|x\in S\}$

    \item [\textbf{Poset}]
      is the category whose objects are elements of a poset $P$ and arrows
      are the partially ordered relation between those elements. The
      reflexive and transitive properties provides the identity and
      composition of arrows.
      \begin{note}
        Every hom-set in this category has at most ONE element.
        Equivalently, any category where every hom-set has at most one element
        is a poset.
      \end{note}

    \item [\textbf{Monoid}]
      is the category  whose only object is the Monoid $M$ itself and arrows
      are the elements of the monoid (this category is denoted as $C(M)$).
      The identity arrow is the identity element and composition being the
      monoid binary operation.
      \begin{note}
        Any category with one object is a monoid.
      \end{note}

    \item [\textbf{Group}]
      is the category whose only object is the Group itself and arrows are
      elements of the group, same as the category determined by a Monoid.
      \begin{note}
        All arrows in this category are isomorphisms.
      \end{note}

    \item [\textbf{Grf}]
      is the category whose objects are graphs and arrows are graph
      homomorphisms. Identity arrow being the identity homormophism and
      composition being composition of homomorphisms.

    \item [\textbf{Posets and Monotone Functions}]
      is the category whose objects are posets and arrows are monotone
      functions between posets.

    \item [\textbf{Sem}]
      is the category whose objects are semigroups and arrows are semigroup
      homomorphisms.
      \begin{note}
        This can be extended to monoids, denoted as \textbf{Mon}.
      \end{note}

    \item[\textbf{$\omega$-CPOs and Scott-continuous Functions}]
      is the category whose objects are $\omega$-CPOs and arrows are
      Scott-continuous functions.
      \begin{note}
        A category is also formed with \textit{strict} $\omega$-CPOs and
        \textit{strict} Scott-continous functions.
      \end{note}

    \item[\textbf{Path Category}]
      is the free category $F(\mathscr{G})$ generated by a graph
      $\mathscr{G}$ whose objects are the nodes of $\mathscr{G}$ and arrows
      are paths in $\mathscr{G}$. For each node/object $A$, the identity
      arrow is the empty path from $A$ to $A$. Composition is defined by
      $(f_1,\dots,f_k)\circ(f_{k+1},\dots,f_n)=(f_1,\dots,f_n)$

    \item[\textbf{Cat}]
      is the category whose objects are small categories and arrows are
      functors between small categories.

    \item[$M$-\textbf{Act}]
      is the category whose objects are monoid actions and arrows are
      equivariant maps. Since equivariant maps are functions, the identity
      arrow is the identity equivariant map and composition is function
      composition.

    \item[\textbf{FBool}]
      is the category whose objects are finite Boolean algebras and arrows
      are homormophisms between them.

    \item[\textbf{Pts}]
      is the category whose objects are pointed sets and arrows are
      functions that preserves the pointed object.

    \item[$\mathbf{Func}(\mathscr{C},\mathscr{D})$]
      is the category whose objects are functors from categories
      $\mathscr{C}$ to $\mathscr{D}$ and arrows are natural transformations.

  \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{functorexamples}
{
  name={Examples of Functors:}
} {
  \begin{itemize}[label={}]
    \item
  \begin{itemize}[align=left]
    \item[\textbf{Monoid Homomorphism}]
      $f:M\rightarrow N$ where $M$ and $N$ are monoids is a functor from the
      category $C(M)$ to the category $C(N)$.

    \item[\textbf{Monotone Map}]
      between two posets is a functor between the category determined by the
      posets.

    \item[\textbf{Projection}]
      The first and second projections from the product category to its
      first or second parts,
      $P_1:\mathscr{C}\times\mathscr{D}\rightarrow\mathscr{C}$ and
      $P_2:\mathscr{C}\times\mathscr{D}\rightarrow\mathscr{D}$ for
      categories $\mathscr{C}$ and $\mathscr{D}$ are functors.

    \item[\textbf{Inclusion Map}]
      of a subcategory is a functor.
      \begin{note}
        By the definition of \textit{subcategory}, objects and arrows of the
        subcategory need not be objects and arrows in the bigger category.
      \end{note}
      \begin{note}[Example]
        \textbf{Set} is a subcategory of \textbf{Rel} and the functor is
        defined by taking sets to the sets as objects in \textbf{Rel} and
        set functions $f:S\rightarrow T$ to $\{(s,t)|t=f(s)\}$.
      \end{note}

    \item[\textbf{Monoid Actions}]
      For an action $\alpha$ of a monoid $M$ acting on a set $S$, let $C(M)$
      be the category determined by $M$ and the action $\alpha$ determines a
      functor $F_\alpha:C(M)\rightarrow\mathbf{Set}$ defined as:
      \begin{itemize}[nosep, label={-}]
        \item $F_\alpha(*)=S$ where $*$ denotes the single object in $C(M)$.
        \item $F_\alpha(m)=s\mapsto\alpha(m,s)$ for $m\in M$, $s\in S$.
      \end{itemize}

    \item[\textbf{Quotient Functor}]
      $Q:\mathscr{C}\rightarrow\mathscr{C}/\sim$\\
      is the functor thats takes a category to its quotient category by
      where $Q(A)=A$ for any object $A$ and $Q(f)=[f]$ for arrows $f$ in the
      category $\mathscr{C}$.

  \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{forgetfulfunctor}
{
  name=Forgetful/Underlying Functor,
  text=forgetful functor
} {
  is the functor which ``forgets'' some of the structure in a
  category of structures and structure-preserving functions.
  \begin{itemize}[label={}, align=left, nolistsep]
    \item[\textit{Examples}:]
    \item
    \begin{itemize}
      \item $U:\mathbf{Mon}\rightarrow\mathbf{Sem}$

      \item $U:\mathbf{Sem}\rightarrow\mathbf{Set}$

      \item $U:\mathbf{Grf}\rightarrow\mathbf{Set}\times\mathbf{Set}$\\
        that takes a graph $\mathscr{G}$ to the set of nodes and set of
        arrows, $U(\mathscr{G})=(\mathscr{G}_0,\mathscr{G}_1)$. It follows
        that there is also an arrowset functor
        $A:\mathbf{Grf}\rightarrow\mathbf{Set}$ and nodeset functor
        $N:\mathbf{Grf}\rightarrow\mathbf{Set}$.

      \item $U:\mathbf{Cat}\rightarrow\mathbf{Grf}$\\
        by forgetting the identity arrows and composition, the remains of a
        category becomes a graph.
        \begin{note}
          If $F$ is a functor, then $F$ is a graph homomorphism. Note that
          the converse is not true.
        \end{note}
        Simiarily, there is the functor
        $A:\mathbf{Cat}\rightarrow\mathbf{Set}$ and
        $O:\mathbf{Cat}\rightarrow\mathbf{Set}$ which takes a category to
        its set of arrows and objects respectively.

      \item $U:\mathscr{C}/A\rightarrow\mathscr{C}$\\
        for an object $A$ in the category $\mathscr{C}$.
        \begin{note}
          If $\mathscr{C}$ is \textbf{Set}, the underlying functor
          $U:\mathbf{Set}/S\rightarrow\mathbf{Set}$ forgets the indexing of
          $S$-indexed sets.
        \end{note}

    \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{freefunctor}
{
  name=Free Functor,
  text=free functor
} {
  Informally, a \textbf{free functor} is left adjoint to a \textbf{forgetful
  functor}.
  \begin{itemize}[label={}, align=left, nolistsep]
    \item[\textit{Examples}:]
    \item
    \begin{itemize}
      \item $F:\mathbf{Set}\rightarrow\mathbf{Mon}$\\
        is the free monoid functor that takes a set $A$ to the free monoid
        $F(A)$, the Kleene closure $A^*$ with concatenation as the monoid
        operation. $F$ takes set functions to the Kleene closure induced
        homormophism.
        \begin{note}
          The Kleene closure itself is a functor from \textbf{Set} to
          \textbf{Set}, which is the composition of $U\circ F$, the
          underlying functor $U:\mathbf{Mon}\rightarrow\mathbf{Set}$ and the
          free monoid functor.
        \end{note}

      \item $F:\mathbf{Grf}\rightarrow\mathbf{Cat}$\\
        takes graphs to the path category of the graph as objects in
        \textbf{Cat} and graph homomorphisms to arrows (functors between
        path categories). For a graph homomorphism
        $\phi:\mathscr{G}\rightarrow\mathscr{H}$,
        $F(\phi)(f_n,f_{n-1},\dots,f_1)=(\phi(f_n),\phi(f_{n-1}),\dots,\phi(f_1))$,
        which is clearly a path in $\mathscr{H}$ (and thus an arrow in the
        path category of $\mathscr{H}$).
    \end{itemize}
  \end{itemize}
}

\newglossaryentry{contravariantfunctor}
{
  name=Contravariant Functor,
  text=contravariant functor,
  description={is a functor $F:\mathscr{C}^{op}\rightarrow\mathscr{D}$ for
    categories $\mathscr{C}$ and $\mathscr{D}$. In contrast to
    \textit{covariant} functors, for an arrow $f$ in $\mathscr{C}$
    \begin{center}
      $F(f:B\rightarrow A)=F(A)\rightarrow F(B)$
    \end{center}
    is an arrow in $\mathscr{D}$. This follows from the fact that $f$ is an
  arrow from $A$ to $B$ in $\mathscr{C}^{op}$.}
}

\longnewglossaryentry{powersetfunctor}
{
  name=Powerset Functor,
  text=powerset functor
} {
  is a functor that takes a set $S$ to its powerset
  $\mathcal{P}(S)$. There are three different powerset functors:

  \begin{itemize}[label={}, nosep]
    \item
      \begin{itemize}[align=left]
        \item[\textbf{Inverse Image}]
          $\mathscr{P}:\mathbf{Set}^{op}\rightarrow\mathbf{Set}$\\
          is a contravariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to its inverse image
          $f:\mathcal{P}(B)\rightarrow\mathcal{P}(A)$.

        \item[\textbf{Direct/Existential Image}]
          is a covariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to the function
          $f_*:\mathcal{P}(A)\rightarrow\mathcal{P}(B)$ where
          $f_*(A_0)=\{f(x)|x\in A_0\}$.

        \item[\textbf{Universal Image}]
          is a covariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to the function
          $f_!:\mathcal{P}(A)\rightarrow\mathcal{P}(B)$ where
          $f_!(A_0)=\{y\in B|(f(x)=y)\Rightarrow(x\in A_0)\}$.
      \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{homfunctor}
{
  name=Hom Functor,
  text=hom functor
} {
  is a functor that takes a category to a hom-set. For any category
  $\mathscr{C}$ with an object $C$, there are three different hom functors:

  \begin{itemize}[label={}, nosep]
    \item
      \begin{itemize}[align=left]
        \item[\textbf{Covariant Hom Functor}]
          $\Hom(C,-):\mathscr{C}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(C,-)(A)=\Hom(C,A)$ for each object
              $A\in\mathscr{C}$.
            \item $\Hom(C,-)(f)=\Hom(C,f):\Hom(C,A)\rightarrow\Hom(C,B)$ for
              an arrow $f:A\rightarrow B$ in $\mathscr{C}$.
          \end{itemize}

        \item[\textbf{Contravariant Hom Functor}]
          $\Hom(-,C):\mathscr{C}^{op}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(-,C)(A)=\Hom(A,C)$ for each object
              $A\in\mathscr{C}$.
            \item $\Hom(-,C)(f)=\Hom(f,C):\Hom(B,C)\rightarrow\Hom(A,C)$ for
              an arrow $f:A\rightarrow B$ in $\mathscr{C}$.
          \end{itemize}

        \item[\textbf{Two-Variable Hom Functor}]
          $\Hom(-,-):\mathscr{C}^{op}\times\mathscr{C}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(-,-)((A,B))=\Hom(A,B)$ for each object
              $(A,B)\in\mathscr{C}^{op}\times\mathscr{C}$.
            \item $\Hom(-,-)((f,g))=\Hom(f,g):\Hom(B,C)\rightarrow\Hom(A,D)$ for
              an arrow $(f:A\rightarrow B, g:C\rightarrow D)$ in
              $\mathscr{C}^{op}\times\mathscr{C}$.
          \end{itemize}

      \end{itemize}
  \end{itemize}

  \begin{note}
    All covariant hom functors preserves products.
  \end{note}
}

\newglossaryentry{fullfunctor}
{
  name=Full,
  text=full,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$ is
    \textbf{full} if the set mapping:
    \begin{center}
      $\Hom_\mathscr{C}(A,B)\rightarrow\Hom_\mathscr{D}(F(A),F(B))$
    \end{center}
    for each pair of objects $A$ and $B$ in $\mathscr{C}$ induced by $F$ is
    \textit{surjective}.
%
    \begin{note}
      A full functor need not be surjective on objects or arrows.
    \end{note}
  }

\newglossaryentry{faithfunctor}
{
  name=Faithful,
  text=faithful,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$ is
    \textbf{faithful} if the set mapping:
    \begin{center}
      $\Hom_\mathscr{C}(A,B)\rightarrow\Hom_\mathscr{D}(F(A),F(B))$
    \end{center}
    for each pair of objects $A$ and $B$ in $\mathscr{C}$ induced by $F$ is
    \textit{injective}.
%
    \begin{note}
      If $f:A\rightarrow B$ and $g:C\rightarrow D$, as long as either $A\neq
      C$ or $C\neq D$, $F(f)=F(g)$ may hold and $F$ can still be faithful.
    \end{note}
%
    \begin{note}
      A faithful functor need not be injective on objects or arrows.
    \end{note}
  }
}

\newglossaryentry{reflect}
{
  name=Reflect,
  text=reflect,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$
    \textbf{reflects} a property $P$ of arrows if whenever $F(f)$ has property
    $P$ then so does $f$ (where $f$ is any arrow for which $F$ takes to
    $F(f)$).
    \begin{note}[Theorem]
      Consider a full and faithful functor
      $F:\mathscr{C}\rightarrow\mathscr{D}$. Let $A$ and $B$ be objects in
      $C$ and $u:F(A)\rightarrow F(B)$ is an isomorphism in $\mathscr{D}$,
      then there is an unique isomorphism $f:A\rightarrow
      B$ in $\mathscr{C}$ for which $F(f)=u$.
    \end{note}
    \begin{note}[Corollary]
      A full and faithful functor reflects isomorphisms.
    \end{note}
    \begin{note}[Corollary]
      For a full an faithful functor $F$, if $F(A)=F(B)$ for objects $A$ and
      $B$ in the domain of $F$, then $A\cong B$.
    \end{note}
  }
}

\longnewglossaryentry{categoryequivalence}
{
  name=Category Equivalence,
  text=category equivalence,
} {
  A functor $F:\mathscr{C}\rightarrow\mathscr{D}$ is an \textbf{equivalence
  of categories} if there is a functor $G:\mathscr{D}\rightarrow\mathscr{C}$
  such that:
  \begin{itemize}[nosep]
    \item A family $u_C:C\rightarrow G(F(C))$ of isomorphisms of
      $\mathscr{C}$ indexed by the objects of $\mathscr{C}$ with the
      property that for every arrow $f:C\rightarrow C'$ in
      $\mathscr{C}$, $G(F(f))=u_{C'}\circ f\circ u_C^{-1}$.

    \item A family $v_D:D\rightarrow F(G(D))$ of isomorphisms of
      $\mathscr{D}$ indexed by the objects of $\mathscr{D}$ with the
      property that for every arrow $g:D\rightarrow D'$ in $\mathscr{D}$,
      $F(G(g))=v_{D'}\circ g\circ v_D^{-1}$.
  \end{itemize}

  \begin{note}
    In this case, the functor $G$ is called a \textit{pseudo-inverse} of $F$
    and $u,v$ are \textit{natural isomorphisms}.
  \end{note}

  \begin{note}
    $F$ and $G$ are both full and faithful.
  \end{note}
}

\longnewglossaryentry{congruence}
{
  name=Congruence Relation,
  text=congruence relation,
  symbol={$\sim$}
} {
  is an equivalence relation $\sim$ on the arrows of a category
  $\mathscr{C}$ if:
  \begin{itemize}
    \item Whenever $f\sim g$, then $f$ and $g$ have the same domain and the
      same codomain.
    \item In the diagram below:
      \begin{mytikzcd}
        A \ar[r,"h"] \& B \ar[r,"f",shift left] \ar[r,"g"',shift right]
                     \& C \ar[r,"k"]
                     \& D
      \end{mytikzcd}
      If $f\sim g$, then $f\circ h\sim g\circ h$ and $k\circ f\sim
      k\circ g$ for arrows $h,k$ where $target(h)=source(f)$ and
      $source(k)=target(f)$.
  \end{itemize}
  \begin{note}
    The congruence class containing the arrow $f$ is denoted by $[f]$.
  \end{note}
}

\longnewglossaryentry{quotientcategory}
{
  name=Quotient Category,
  text=quotient category,
  symbol={$/\sim$}
} {
  For a congruence relation $\sim$ on the arrows of a category
  $\mathscr{C}$, the \textbf{quotient category} $\mathscr{C}/\sim$ is
  defined as follows:
  \begin{itemize}
    \item The objects of $\mathscr{C}/\sim$ are the objects of
      $\mathscr{C}$.
    \item The arrows of $\mathscr{C}/\sim$ are the congruence classes of
      arrows of $\mathscr{C}$.
    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $[f]:A\rightarrow B$
      in $\mathscr{C}/\sim$.
    \item If $f:A\rightarrow B$ and $g:B\rightarrow C$ in $\mathscr{C}$,
      then $[g]\circ[f]=[g\circ f]:A\rightarrow C$ in $\mathscr{C}$.
  \end{itemize}

  \begin{note}
    For any functor $F:\mathscr{C}\rightarrow\mathscr{D}$ with the property
    that if $f\sim g$ then $F(f)=F(g)$, then there is an unique functor
    $F_0:\mathscr{C}/\sim\rightarrow\mathscr{D}$ for which $F_0\circ Q=F$.
    Here, $Q$ is the quotient functor,
    $Q:\mathscr{C}\rightarrow\mathscr{C}/\sim$.
  \end{note}
}

\newglossaryentry{naturalisomorphism}
{
  name=Natural Isomorphism,
  text=natural isomorphism,
  description={is a natural transformation $\alpha:F\rightarrow G$ in which
    there is a natural transformation $\beta:G\rightarrow F$ is an inverse
    to $\alpha$ in $\mathbf{Func}(\mathscr{C},\mathscr{D})$.
    \begin{note}
      Natural isomorphisms are also known as \textit{natural equivalences}.
    \end{note}
    \begin{note}[Theorem]
      Suppose $F:\mathscr{C}\rightarrow\mathscr{D},
      G:\mathscr{D}\rightarrow\mathscr{C}$ are functors and
      $\alpha:F\rightarrow G$ is a natural transformation between them. Then
      $\alpha$ is a \textit{natural isomorphism} if and only if for each
      object $C\in\mathscr{C}$, $\alpha C:F(C)\rightarrow G(C)$ is an
      isomorphism in $\mathscr{D}$.
    \end{note}
  }
}

\longnewglossaryentry{factorizationsystem}
{
  name=Factorization System,
  text=factorization system,
} {
  A \textbf{factorization system} in a category $\mathscr{C}$ consists of
  two subclasses $\mathscr{E}$ and $\mathscr{M}$ of the arrows of
  $\mathscr{C}$ such that:
  \begin{itemize}
    \item If $\mathscr{I}$ is the class of isomorphisms, then
      $\mathscr{M}\circ\mathscr{I}\subseteq\mathscr{M}$ and
      $\mathscr{I}\circ\mathscr{E}\subseteq\mathscr{E}$.

    \item Every arrow $f\in\mathscr{C}$ factors as $f=m\circ e$ where
      $m\in\mathscr{M}$ and $e\in\mathscr{E}$.

    \item In any commutative square
      \begin{mytikzcd}
        A \ar[r,"e",twoheadrightarrow] \ar[d,"f"'] \& B \ar[d,"g"] \\
        C \ar[r,"m"',rightarrowtail]               \& D
      \end{mytikzcd}
      with $e\in\mathscr{E}$ and $m\in\mathscr{M}$, there is a unique
      $h:B\rightarrow C$ such that $h\circ e=f$ and $m\circ h=g$.
  \end{itemize}
}

\longnewglossaryentry{naturaltransformation}
{
  name=Natural Transformation,
  text=natural transformation
} {
  Let $F,G:\mathscr{C}\rightarrow\mathscr{D}$ be 2 functors with the same
  domain and codomain. A \textbf{natural transformation}
  $\alpha:F\rightarrow G$ is given by a family of arrows $\alpha C$ for each
  object $C\in\mathscr{C}$ such that:
  \begin{itemize}
    \item $\alpha C:F(C)\rightarrow G(C)$ for each object $C\in\mathscr{C}$,
      also say that $\alpha$ is natural in $C$.
    \item For any arrow $f:C\rightarrow D$ in $\mathscr{C}$, the following
      diagram commutes.
      \begin{mytikzcd}
        F(C) \ar[r, "\alpha C"]  \ar[d, "F(f)"'] \& G(C) \ar[d, "G(f)"] \\
        F(D) \ar[r, "\alpha D"']                 \& G(D)
      \end{mytikzcd}
  \end{itemize}

  \begin{note}[Composition with Functors]
    Let $H:\mathscr{B}\rightarrow\mathscr{C}$ be a functor and
    $\alpha:F\rightarrow G$ where $F,G:\mathscr{A}\rightarrow\mathscr{B}$ be
    are functors. Then, postcomposing $\alpha$ with the functor $H$ is a
    natural transformation $H\alpha:H\circ F\rightarrow H\circ G$ defined by
    $(H\alpha)A=H(\alpha A)$ for any object $A\in\mathscr{A}$.\\
    A similar result can be shown when precomposing a natural transformation
    with a functor.
  \end{note}
}

\longnewglossaryentry{yonedaembedding}
{
  name=Yoneda Embedding,
  text=yoneda embedding
} {
  For any category $\mathscr{C}$, there is a contravariant \textit{Yoneda
  Functor} $Y:\mathscr{C}^{op}\rightarrow\mathbf{Func}(C,\mathbf{Set})$
  defined as:
  \begin{itemize}
    \item for objects $C\in\mathscr{C}, Y(C)=\Hom(C,-)$
    \item for an arrow $f:D\rightarrow C$ and an object $A\in\mathscr{C},
      Y(f)=\Hom(f,-)$ and $Y(f)(A):\Hom(C,A)\rightarrow\Hom(D,A)$
  \end{itemize}

  Thus, the following diagram commutes
  \begin{equation*}\begin{tikzcd}[column sep=huge, ampersand replacement=\&]
    {\Hom(D,A)} \ar[r, "{\Hom(D,k)}"] \& {\Hom(D,B)} \\
    {\Hom(C,A)} \ar[u, "Y(f)A"] \ar[r, "{\Hom(C,k)}"'] \& {\Hom(C,B)} \ar[u, "Y(f)B"']
  \end{tikzcd}\end{equation*}
  for an arrow $k:A\rightarrow B$.\\

  \begin{note}
    $Y(f):\Hom(C,-)\rightarrow\Hom(D,-)$ is the induced natural
    transformation between hom functors corresponding to $f$. Generalizing
    $\Hom(D,-)$ to an arbitary set valued functor, by the \textit{Yoneda
    Lemma}, $Y$ is both full and faithful.
  \end{note}

  \begin{note}
    By the contravariant \textit{Yoneda Lemma}, there also exists the
    covariant functor
    $J:\mathscr{C}\rightarrow\mathbf{Func}(\mathscr{C}^{op},\mathbf{Set})$.
  \end{note}
}

\longnewglossaryentry{godementcalculus}
{
  name=Godement Calculus,
  text=Godement Calculus
} {
  Let $\mathscr{A,B,C,D,E}$ be categories, $E, F_1, F_2,
  F_3, G_1, G_2, G_3, H$ be functors and $\alpha,\beta,\gamma,\delta$ be
  natrual morphisms as shown in the following diagram:
  \begin{mytikzcd}
       \mathscr{A} \ar[r,"E"]
    \& \mathscr{B} \ar[r,bend left=70,"F_1"]
                   \ar[r,phantom,bend left=70,""{name=F1,below}]
                   \ar[r,"F_2" description]
                   \ar[r,phantom,""{name=F2A,above}]
                   \ar[r,phantom,""{name=F2B,below}]
                   \ar[r,bend right=70,"F_3"']
                   \ar[r,phantom,bend right=70,""{name=F3,above}]
                   \ar[Rightarrow,from=F1,to=F2A,"\alpha"]
                   \ar[Rightarrow,from=F2B,to=F3,"\beta"]
    \& \mathscr{C} \ar[r,bend left=70,"G_1"]
                   \ar[r,phantom,bend left=70,""{name=G1,below}]
                   \ar[r,"G_2" description]
                   \ar[r,phantom,""{name=G2A,above}]
                   \ar[r,phantom,""{name=G2B,below}]
                   \ar[r,bend right=70,"G_3"']
                   \ar[r,phantom,bend right=70,""{name=G3,above}]
                   \ar[Rightarrow,from=G1,to=G2A,"\gamma"]
                   \ar[Rightarrow,from=G2B,to=G3,"\delta"]
    \& \mathscr{D} \ar[r,"H"]
    \& \mathscr{E}
  \end{mytikzcd}
  There are 5 laws/equational rules between composition of functors and
  natural transformations:
  \begin{enumerate}
    \item
      $(\delta\circ\gamma)\ast(\beta\circ\alpha)=(\delta\ast\beta)\circ(\gamma\circ\alpha)$
    \item $(H\circ G_1)\alpha=H(G_1\alpha)$
    \item $\gamma(F_1\circ E)=(\gamma F_1)E$
    \item $G_1(\beta\circ\alpha)E=(G_1\beta E)\circ(G_1\alpha E)$
    \item
      $\gamma\ast\alpha=(\gamma F_2)\circ(G_1\alpha)=(G_2\alpha)\circ(\gamma F_1)$
  \end{enumerate}
  \begin{note}
    The first rule is also known as the \textit{Interchange Law}.
  \end{note}
}

\longnewglossaryentry{yonedalemma}
{
  name=Yoneda Lemma,
  text=yoneda lemma
} {
  Let $F:\mathscr{C}\rightarrow\mathbf{Set}$ be a \textit{set-valued
  functor} and $C$ is an object in $C$. There is a one-to-one correspondance
  between elements of $F(C)$ and $h^C:\Hom(C,-)\rightarrow F$, that is:
  \begin{equation*}
    Nat(h^C,F)\cong F(C)
  \end{equation*}
  Thus, this is an isomorphism that is both an injective and surjective
  mapping between elements of $F(C)$ and $h^C$.

  \begin{note}
    $h^C(X)=\Hom(C,X)\rightarrow F(X)$ for an object $X\in\mathscr{C}$
    is defined as $h^C(X)(f)=F(f)(c)$ where $f:C\rightarrow X$ is an arrow
    and $c\in F(C)$ is an element.
  \end{note}
  \begin{note}
    There is also a contravariant version of Yoneda Lemma involving the
    contravariant hom functor $h_C:\Hom(-,C)\rightarrow F$.
  \end{note}
}

\longnewglossaryentry{universalelement}
{
  name=Universal Element,
  text=universal element
} {
  By the \textit{Yoneda Lemma}, an element $c\in F(C)$ is given by
  $c=h^C(id_C)$. $c$ is an \textbf{universal element} of a functor
  $F:\mathscr{C}\rightarrow\mathbf{Set}$ if and only if the induced
  $h^C:Hom(C,-)\rightarrow F$ is a \textit{natural isomorphism}.
  Consequently, for any object $C'\in\mathscr{C}$ and any element $x\in
  F(C')$, there is an unique arrow $f:C\rightarrow C'$ in $\mathscr{C}$ for
  which $x=F(f)(c)$.

  \begin{note}
    There is a one-to-one correspondance between representations and
    universal elements of $F$.
  \end{note}
  \begin{note}
    A similar argument for contravariant set valued functors of the form
    $F:\mathscr{C}^{op}\rightarrow\mathbf{Set}$ can be made where $c\in
    F(C)$ is an \textbf{universal element} if for any object
    $C'\in\mathscr{C}$ and any element $x\in F(C')$, there is an unique
    arrow $f:C'\rightarrow C$ for which $x=F(f)(c)$.
  \end{note}
}

\longnewglossaryentry{2category}
{
  name=2-Category,
  text=2-category,
} {
  is a category \textbf{C} with the following additional structure:
  \begin{itemize}
    \item A set of objects of \textbf{C} (called \textit{0-cells}).

    \item $\forall A,B\in\mathbf{C}_0$ (for every pair of 0-cells), a small
      category $\mathbf{C}(A,B)$ whose objects are arrows from $A$ to $B$ in
      \textbf{C} (objects and arrows in $\mathbf{C}(A,B)$ are called
      \textit{1-cells} and \textit{2-cells} respectively).

    \item $\forall A,B,C\in\mathbf{C}_0$ (for every triple of 0-cells),
      there is a composition functor:
      \begin{equation*}
        comp:\mathbf{C}(B,C)\times\mathbf{C}(A,B)\longrightarrow\mathbf{C}(A,C)
      \end{equation*}
      and that the following diagram commutes:
      \begin{equation*}\begin{tikzcd}[column sep=huge,ampersand replacement=\&]
           {\mathbf{C}(C,D)\times\mathbf{C}(B,C)\times\mathbf{C}(A,B)}
             \ar[r,"comp\times id"]
             \ar[d,"id\times comp"']
        \& {\mathbf{C}(B,D)\times\mathbf{C}(A,B)}
             \ar[d,"comp"]
        \\
           {\mathbf{C}(C,D)\times\mathbf{C}(A,C)}
             \ar[r,"comp"']
        \& {\mathbf{C}(A,D)}
      \end{tikzcd}\end{equation*}
      ie: $comp$ is associative

    \item For each object $A\in\mathbf{C}$, there is the identity functor:
      \begin{equation*}
        unit_A:1\rightarrow\mathbf{C}(A,A)
      \end{equation*}
      where $1$ is the terminal object in \textbf{Cat} (ie: the category
      with one object denoted as $0$ and its identity arrow) defined as
      $unit_A(0)=id_A$ and that the following diagram commutes:
      \begin{equation*}\begin{tikzcd}[column sep=huge,ampersand replacement=\&]
           \& {\mathbf{C}(A,B)} \ar[ld,"unit_B\times id"']
                               \ar[d,"id"]
                               \ar[rd,"id\times unit_A"]
        \\    {\mathbf{C}(B,B)\times\mathbf{C}(A,B)} \ar[r,"comp"']
           \& {\mathbf{C}(A,B)}
           \& {\mathbf{C}(A,B)\times\mathbf{C}(A,A)} \ar[l,"comp"']
      \end{tikzcd}\end{equation*}
      ie: $unit$ is the identity functor with respect to $comp$

  \end{itemize}

  The structure of a 2-category can be summarized as follows:
  \begin{itemize}
    \item A \textit{horizontal category} whose objects are 0-cells, arrows are
      2-cells.
    \item A \textit{vertical category} whose objects are 1-cells, arrows are
      2-cells.
    \item A \textit{base category} (that can be derived from the horizontal
      and vertical categories) whose objects are 0-cells, arrows are
      1-cells.
  \end{itemize}

  \begin{note}[Visual Depictions]
     \item[]
     \begin{itemize}
       \item[0-cells] $x,y$
       \item[1-cells] $x\rightarrow y$
       \item[2-cells]
         \begin{tikzcd}[ampersand replacement=\&]
              x \ar[r,bend left=70,""{name=F,below}]
                \ar[r,bend right=70,""{name=G,above}]
                \ar[Rightarrow,from=F,to=G]
           \& y
         \end{tikzcd}
     \end{itemize}
  \end{note}

  \textbf{Cat} as a 2-category:
  \begin{itemize}
    \item 0-cells are small categories.
    \item 1-cells are functors.
    \item 2-cells are natural transformations.
    \item $comp$ on 1-cells is composition of functors:
      \begin{equation*}
        comp(G,F)=G\circ F
      \end{equation*}
      for arrows $F:A\rightarrow B,G:B\rightarrow C$ in \textbf{Cat}.
    \item $comp$ on 2-cells is horizontal composition of natural
      transformations:
      \begin{equation*}
        comp(\beta,\alpha)=\beta\ast\alpha
      \end{equation*}
      where $F,F':A\rightarrow B$ and $G,G':B\rightarrow C$ are functors
      $\alpha:F\rightarrow F',\beta:G\rightarrow G'$ are natural
      transformations between them.
    \item By the \textit{Interchange Law}
      \begin{equation*}
        comp((\beta',\alpha')\circ(\beta,\alpha))=comp(\beta',\alpha')\circ comp(\beta,\alpha)
      \end{equation*}
  \end{itemize}
}

\newglossaryentry{posetenrichedcategory}
{
  name=Poset-Enriched Category,
  text=poset-enriched category,
  description={is a category $\mathscr{C}$ with a partial ordering on every
    $\Hom_\mathscr{C}(A,B)$ such that for every triple of objects
    $A,B,C\in\mathscr{C}$, the composition of hom-sets:
    \begin{equation*}
      \Hom(A,B)\times\Hom(B,C)\longrightarrow\Hom(A,C)
    \end{equation*}
    is monotone. Thus if $f\leq f':A\rightarrow B, g\leq g':B\rightarrow C$
    then $g\circ f\leq g'\circ f':A\rightarrow C$.
    \begin{note}[As a 2-category]
      Exactly ONE 2-cell can be defined from $f$ to $g$ for arrows
      $f,g:A\rightarrow B$ if and only if $f\leq g$. Otherwise, no 2-cells
      from $f$ to $g$. For each pair of objects $A$ and $B$, the category
      $C(\Hom(A,B))$ can be constructed as a poset.
    \end{note}
  }
}

  description={is a poset-enriched category in which each poset $\Hom(A,B)$
    is a strict $\omega$-CPO where every pair of elements has a supremum
  (least upper bound) and an infimum (greatest lower bound).}
}

\longnewglossaryentry{product}
{
  name=Product,
  text=product,
  symbol={$\times$}
} {
  Let $A,B$ be objects in a category $\mathscr{C}$. A \textbf{product} of $A$ and
  $B$ is an object $A\times B$ together with projections/arrows
  $p_1:A\times B\rightarrow A$ and $p_2:A\times B\rightarrow B$ such that
  for any object $D$ and arrows
  $q_1:D\rightarrow A$, $q_2:D\rightarrow B$, there is an unique arrow
  $q:D\rightarrow A\times B$ and that the following diagram commutes:
  \begin{mytikzcd}
      \& D \ar[ld,"q_1"'] \ar[d,"q"] \ar[rd,"q_2"] \\
    A \&
    {A\times B} \ar[l,"p_1"] \ar[r,"p_2"'] \&
    B
  \end{mytikzcd}
  ie, $p_1\circ q=q_1$ and $p_2\circ q=q_2$.\\

  The concept of a product is defined up to an unique isomorphism and is
  defined by a \textit{universal mapping property}. Thus, all products of
  the objects $A$ and $B$ are isomorphic and any object isomorphic to
  $A\times B$ is a product of $A$ and $B$.\\

  Since for each pair of arrows $(q1,q2)$ produces an unique arrow $q$,
  there is a natural isomorphism:
  \begin{equation*}
    \pi:\Hom_\mathscr{C}(-,A)\times\Hom_\mathscr{C}(-,B)\rightarrow
    \Hom_\mathscr{C}(-,A\times B)
  \end{equation*}
  such that $\pi D(q_1,q_2)=q$ and is denoted as $\langle q_1,q_2\rangle$.

  \begin{note}
    $(p1,p2)$ is an \textit{universal element} of the functor
    $\Hom(-,A)\times\Hom(-,B)$.
  \end{note}
}

\longnewglossaryentry{pointwiseproducts}
{
  name=Pointwise Products,
  text=pointwise products
} {
  For categories $\mathscr{C,D}$, if $\mathscr{D}$ has products
  then $\mathbf{Func}(\mathscr{C},\mathscr{D})$ has \textbf{pointwise
  products} defined as:
  \begin{itemize}
    \item $(F\times G)(C)=F(C)\times G(C)$ for any object $C\in\mathscr{C}$
    \item $(F\times G)(f)=F(f)\times G(f)$ for any arrow $f\in\mathscr{C}$
  \end{itemize}
  where $F,G:\mathscr{C}\rightarrow\mathscr{D}$ are functors.
}

\longnewglossaryentry{productofarrows}
{
  name=Product of Arrows,
  text=product of arrows
} {
  For arrows $f:S\rightarrow S',t:T\rightarrow T'$, the \textbf{product of
  arrows}, $f\times g$ is defined such that the following diagram commutes:
  \begin{mytikzcd}
       S \ar[d,"f"']
    \& {S\times T} \ar[l,"p1"'] \ar[r,"p2"] \ar[d,"f\times g"]
    \& T \ar[d,"g"] \\
       S'
    \& {S'\times T'} \ar[l,"p1'"] \ar[r,"p2'"']
    \& T'
  \end{mytikzcd}
  where $p1,p2$ and $p1',p2'$ are the projections of $S\times T$ to $S,T$ and
  $S'\times T'$ to $S',T'$ respectively.

  \begin{note}
    Products are distributed through composition, for arrows
    $f_i:A_i\rightarrow B_i,g_i:B_i\rightarrow C_i$ for $i=1,2$:
    \begin{equation*}
      (g_1\circ f_1)\times(g_2\circ f_2)=(g_1\times g_2)\circ(f_1\times f_2)
    \end{equation*}
  \end{note}
}

\longnewglossaryentry{naryproduct}
{
  name=n-Ary Product,
  text=n-ary product
} {
  is the product of a list of $n$ objects $A_1,\dots,A_n$ with
  projectiions $p_i:\Pi^n_{i=1}A_i\rightarrow A_i$ and the property that
  given any object $B$ and arrows $f_i:B\rightarrow A_i$, there is an unique
  arrow $\langle f_1,\dots,f_n\rangle:B\rightarrow\Pi^n_{i=1}A_i$ for which
  $p_i\circ\langle f_1,\dots,f_n\rangle=f_i$.\\

  Let the product of $n$ number of the object $A$, $A\times\dots\times A$ be
  denoted as $A_n$:
  \begin{itemize}[align=left,nosep]
    \item [\textit{Nullary Product}] denoted as $A_0$, is the terminal
      object.
    \item [\textit{Unary Product}] denoted as $A_1$, is effectively a
      product ''wrapper`` around $A$, where the only projection arrow is
      $id_A$. (Analogous to characters vs strings of length 1).
  \end{itemize}

  \begin{note}
    \textit{Binary products} (2-ary product) can be used to construct
    \textit{Ternary products} (3-ary products) as
    \begin{equation*}
      A\times B\times C\cong(A\times B)\times C\cong A\times(B\times C)
    \end{equation*}
  \end{note}
}

\longnewglossaryentry{strictinitialobject}
{
  name=Strict Initial Object,
  text=strict initial object
} {
  Let $\mathscr{C}$ be a category with products. An initial product, $0$, in
  $\mathscr{C}$ is a \textbf{strict initial object} if it has one of the
  following equivalent properties:
  \begin{itemize}
    \item $\forall A\in\mathscr{C}$, if there is an $u:A\rightarrow 0$, then
      $A\cong 0$.
    \item $\forall A\in\mathscr{C}, 0\times A\cong 0$.
  \end{itemize}
}

\newglossaryentry{infiniteproduct}
{
  name=Infinite Product,
  text=infinite product,
  description={is a product of infinitely many objects. Let $I$ be a set and
    $\{A_i|i\in I\}$ be an indexed set of objects in the category
    $\mathscr{C}$. An \textbf{infinite product} $\Pi_{i=1}A_i$ of an indexed set
    is an object $P$ with an indexed set of arrows $p_i:P\rightarrow A_i$ such
    that for any object $A\in\mathscr{C}$ with arrows $q_i:A\rightarrow A_i$,
    there is an unique arrow $q=\langle q_i\rangle:A\rightarrow P$ and $p_i\circ
  q=q_i$ for all $i\in I$.}
}
