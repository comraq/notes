\newglossaryentry{homfunction}
{
  name=Hom Function,
  text=hom function,
  description={
    Let $S$ be an object and $f:T\rightarrow V$ an arrow in a
    category, a \textit{covariant} \textbf{hom function}
    $\Hom(S,f):\Hom(S,T)\rightarrow\Hom(S,V)$, if defined by:
    \begin{center}
      $\Hom(S,f)(g)=f\circ g$
    \end{center}
    for some $g\in\Hom(S,T)$.\\
%
    Similarily, a \textit{contravariant} \textbf{hom function}
    $\Hom(f,S):\Hom(V,S)\rightarrow\Hom(T,S)$, if defined by:
    \begin{center}
      $\Hom(f,S)(g)=g\circ f$
    \end{center}
    for some $g\in\Hom(V,S)$.
  }
}

\longnewglossaryentry{category}
{
  name=Category,
  text=category
} {
  A \textbf{category} $\mathscr{C}$ contains a collection of
  \textit{objects} (denoted as $\mathscr{C}_0$) and a collection of
  \textit{morphisms} (denoted as $\mathscr{C}_1$) where every morphism has
  $source, target:\mathscr{C}_1\rightarrow\mathscr{C}_0$ functions mapping
  each arrow to its \textit{source} and \textit{target} objects respectively
  into the collection of objects.\\

  For any two arrows $f,g$ where the $target(f)=source(g)$, the
  \textit{composition} of $f$ and $g$ are written as $g\circ
  f:source(f)\rightarrow target(g)\in\mathscr{C}_1$. Composition is
  associative, $(h\circ g)\circ f=h\circ (g\circ f)$ when either side is
  defined.\\

  $\forall A\in\mathscr{C}_0$, there exists the \textit{identity} arrow,
  $id_A:A\rightarrow A$ and for any $f:A\rightarrow B$, $f\circ id_A=id_B\circ
  f=f$.\\

  A category is \textit{small} if its objects and arrows constitute sets;
  otherwise it is \textit{large}.
}

\longnewglossaryentry{subcategory}
{
  name=Subcategory,
  text=subcategory
} {
  A \textbf{subcategory} $\mathscr{D}$ of a category
  $\mathscr{C}$ is a category for which:
  \begin{itemize}
    \item All objects and arrows in $\mathscr{D}$ are objects and arrows in
      $\mathscr{C}$ (ie: $\mathscr{D}_0\subseteq\mathscr{C}_0$ and
      $\mathscr{D}_1\subseteq\mathscr{C}_1$).

    \item The source and target of an arrow in $\mathscr{D}$ are the same as
      its souce and target in $\mathscr{C}$ (ie: source and target maps in
      $\mathscr{D}$ are restrictions of those in $\mathscr{C}$). Thus,
      $\forall A,B\in\mathscr{C}_0$,
      $\Hom_{\mathscr{D}}(A,B)\subseteq\Hom_{\mathscr{C}}(A,B)$.

    \item If $A$ is an object in $\mathscr{C}$, its identity arrow $id_A$ in
      $\mathscr{C}$ is in $\mathscr{D}$.

    \item If $f:A\rightarrow B$ and $g:B\rightarrow C$ in $\mathscr{D}$,
      then the composite $g\circ f\in\mathscr{C}_1$ is also the composite in
      $\mathscr{D}$.
  \end{itemize}
}

\newglossaryentry{productofcategories}
{
  name=Product of Categories,
  text=product of categories,
  description={If $\mathscr{C}$ and $\mathscr{D}$ are categories their
    \textbf{product} $\mathscr{C}\times\mathscr{D}$ is the category whose
    objects are all ordered pairs $(C,D)$ where $C\in\mathscr{C}_0,
    D\in\mathscr{D}_0$ and arrows are $(f,g):(C,D)\rightarrow (C',D')$ where
    $f:C\rightarrow C'\in\mathscr{C}_1, g:D\rightarrow D'\in\mathscr{D}_1$.
    The identity arrow of $(C,D)$ is $(id_C,id_D)$ and composition of arrows is
  defined component wise.}
}

\longnewglossaryentry{categoricaldual}
{
  name=Categorical Dual,
  text=categorical dual,
  symbol={$^{op}$}
} {
  Given any category $\mathscr{C}$, $\mathscr{C}^{op}$ is the
  \textbf{dual category} of $\mathscr{C}$ where all the arrows are reversed.
  $\mathscr{C}^{op}$ is defined as follows:
  \begin{itemize}
    \item The objects and arrows of $\mathscr{C}^{op}$ are the objects and
      arrows of $\mathscr{C}$.

    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $f:B\rightarrow A$ in
      $\mathscr{C}^{op}$.

    \item If $h=g\circ f$ in $\mathscr{C}$, then $h=f\circ g$ in
      $\mathscr{C}^{op}$.
  \end{itemize}
  Since identity arrows have the same source and target objects, identity
  arrows in $\mathscr{C}$ are the same as in $\mathscr{C}^{op}$ and thus
  $\mathscr{C}^{op}$ is also a category.
}

\longnewglossaryentry{slicecategory}
{
  name=Slice Category,
  text=slice category,
  symbol={$/$}
} {
  If $\mathscr{C}$ is a category, for any object $A\in\mathscr{C}_0$, the
  \textbf{slice category} $\mathscr{C}/A$ is defined as follows:
  \begin{itemize}
    \item An object of $\mathscr{C}/A$ is an arrow $f:C\rightarrow A$ in
      $\mathscr{C}$ for some $C\in\mathscr{C}_0$.

    \item An arrow of $\mathscr{C}/A$ from $f:C\rightarrow A$ to
      $f':C'\rightarrow A$ is an arrow $h:C\rightarrow C'$ such that
      $f=f'\circ h$.

    \item The composite of $h:f\circ f'$ and $h':f'\circ f''$ is $h'\circ
      h$.
  \end{itemize}

  \begin{note}
    Since the same $h$ can satisfy both $f=f'\circ h$ and $g=g'\circ h$
    where $f\neq g$ or $f'\neq g'$, $h:f\rightarrow f'$ and $h:g\rightarrow
    g'$ are regarded as different arrows in $\mathscr{C}/A$.
  \end{note}

  \begin{note}
    The \textit{indexed function} of an \textit{indexed set} indexed
    by $S$, is precisely an arrow in the slice category \textbf{Set}$/S$.
  \end{note}
}

\newglossaryentry{monomorphism}
{
  name=Monomorphism,
  text=monomorphism,
  description={is an arrow $f:A\rightarrow B$ such that for any object $T$
    and arrows $x,y:T\rightarrow A$, if $x\neq y$, then $f\circ x\neq f\circ
    y$. Monomorphisms can be denoted by $f:A\rightarrowtail B$.
    \begin{note}
      In this context, $x,y:T\rightarrow A$ may be regarded as
      \textit{variable elements} and can be written as if $x\neq y$, then
      $f(x)\neq f(y)$.
    \end{note}
    \begin{note}[Properties]
      Let $f:A\rightarrow B$ and $g:B\rightarrow C$ be arrows:
      \begin{itemize}
        \item If $f$ and $g$ are monomorphisms, so is $g\circ f$.
        \item If $g\circ f$ is a monomorphism, so is $f$.
      \end{itemize}
    \end{note}
    \begin{note}
      In \textbf{Set}, a function is injective if and only if it is a
      monomorphism.
    \end{note}
    \begin{note}
      In the category determined by a Monoid, a monic element is left
      cancellable.
    \end{note}
  },
  symbol={$\rightarrowtail$}
}

\newglossaryentry{epimorphism}
{
  name=Epimorphism,
  text=epimorphism,
  description={is the dual to monomorphisms. An arrow $f:S\rightarrow T$ is
    an \textbf{epimorphism} if for any arrows $g,h:T\rightarrow X$, $g\circ
    f=h\circ f$ implies $g=h$. Epimorphisms can be denoted by
    $f:S\twoheadrightarrow T$.
    \begin{note}[Properties]
      Let $f:A\rightarrow B$ and $g:B\rightarrow C$ be arrows:
      \begin{itemize}
        \item If $f$ and $g$ are epimorphisms, so is $g\circ f$.
        \item If $g\circ f$ is an epimorphism, so is $g$.
      \end{itemize}
    \end{note}
    \begin{note}
      In \textbf{Set}, a function is surjective if and only if it is an
      epimorphism.
    \end{note}
    \begin{note}
      In the category determined by a Monoid, an epic element is right
      cancellable.
    \end{note}
  },
  symbol={$\twoheadrightarrow$}
}

\longnewglossaryentry{subobject}
{
  name=Subobject,
  text=subobject
} {
  In a category $\mathscr{C}$, a \textbf{subobject} of an object
  $C\in\mathscr{C}$ is an equivalence class of monomorphisms under the
  relation $\sim$. The subobject is a \textbf{proper subobject} if it does not
  contain $id_C$. The equivalence relation $\sim$ is defined as follows:
  \begin{itemize}
    \item For monomorphisms $f_0:C_0\rightarrowtail C$ and $f_1:C_1\rightarrowtail
      C$, $f_0\sim f_1$ if both $f_0$ and $f_1$ factors through each other.
  \end{itemize}

  \begin{note}
    A preorder for subobjects $(i:A\rightarrow C)\leq(j:B\rightarrow C)$ is
    defined by the condition that there exists $k:A\rightarrow B$ such that
    $i=j\circ k$. Since $j$ is monic, $k$ is unique. And $i$ is monic
    implies $k$ is monic. In \textbf{Set}, $A\subseteq B\subseteq C$
    where $i$ is the inclusion of $A$ to $C$, $j$ the inclusion of $B$ to
    $C$ and $k$ the inclusion of $A$ to $B$.
  \end{note}
}

\longnewglossaryentry{functor}
{
  name=Functor,
  text=functor
} {
  is a ``structure preserving'' map between categories, similar to
  homomorphisms between groups, graphs and etc$\dots$ A \textbf{functor}
  $F:\mathscr{C}\rightarrow\mathscr{D}$ from the category $\mathscr{C}$ to
  the category $\mathscr{D}$ defined as follows:
  \begin{itemize}
    \item $\forall A\in\mathscr{C}$, $F(A)\in\mathscr{D}$.
    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $F(f):F(A)\rightarrow
      F(B)$ is an arrow in $\mathscr{D}$ such that:
      \begin{itemize}
        \item $F(id_A)=id_{F(A)}$ for every object $A\in\mathscr{C}$.
        \item $F(g\circ f)=F(g)\circ F(f)$ for all arrows $f:X\rightarrow
          Y, g:Y\rightarrow Z$ in $\mathscr{C}$.
      \end{itemize}
  \end{itemize}

  \begin{note}
    All functors preserve isomorphisms.
  \end{note}
}

\longnewglossaryentry{categoryexamples}
{
  name={Examples of Categories:}
} {
  \begin{itemize}[label={}]
    \item
  \begin{itemize}[align=left]
    \item[\textbf{Set}]
      is the category whose objects are sets and arrows are functions
      between sets. Composition is function composition and the identity
      arrow is the identity function $id_S$ for a set $S$.

    \item[\textbf{Fin}]
      is the category whose objects are finite sets and arrows are all the
      functions between finite sets.

    \item[\textbf{Pfn}]
      is the category whose objects are sets but all arrows are all partial
      functions. If $f:S\rightarrow T, g:T\rightarrow V$ are partial
      functions with $f$ defined on $S_0\subseteq S$ and $g$ defined on
      $T_0\subseteq T$, then the composite $g\circ f:S\rightarrow V$ is the
      partial function from $S$ to $V$ defined on the subset $\{x\in
      S_0|f(x)\in T_0\}$ of $S$ by the requirement $(g\circ f)(x)=g(f(x))$.

    \item[\textbf{Rel}]
      is the category whose objects are sets and arrows are relations
      relations between sets. The composite $\beta\circ\alpha$ of relations
      $\alpha$ from sets $S$ to $T$ and $\beta$ from $T$ to $U$ is a
      relation from $S$ to $U$ defined as follows:
      \begin{center}
        If $x\in S$ and $z\in U$, $(x,z)\in\beta\circ\alpha$ if and only if
        there is an element $y\in T$ for which $(x,y)\in\alpha$ and
        $(y,z)\in\beta$.
      \end{center}
      The identity arrow for a set $S$ is the \textit{diagonal relation}
      $\Delta_S=\{(x,x)|x\in S\}$

    \item [\textbf{Poset}]
      is the category whose objects are elements of a poset $P$ and arrows
      are the partially ordered relation between those elements. The
      reflexive and transitive properties provides the identity and
      composition of arrows.
      \begin{note}
        Every hom-set in this category has at most ONE element.
        Equivalently, any category where every hom-set has at most one element
        is a poset.
      \end{note}

    \item [\textbf{Monoid}]
      is the category  whose only object is the Monoid $M$ itself and arrows
      are the elements of the monoid (this category is denoted as $C(M)$).
      The identity arrow is the identity element and composition being the
      monoid binary operation.
      \begin{note}
        Any category with one object is a monoid.
      \end{note}

    \item [\textbf{Group}]
      is the category whose only object is the Group itself and arrows are
      elements of the group, same as the category determined by a Monoid.
      \begin{note}
        All arrows in this category are isomorphisms.
      \end{note}

    \item [\textbf{Grf}]
      is the category whose objects are graphs and arrows are graph
      homomorphisms. Identity arrow being the identity homormophism and
      composition being composition of homomorphisms.

    \item [\textbf{Posets and Monotone Functions}]
      is the category whose objects are posets and arrows are monotone
      functions between posets.

    \item [\textbf{Sem}]
      is the category whose objects are semigroups and arrows are semigroup
      homomorphisms.
      \begin{note}
        This can be extended to monoids, denoted as \textbf{Mon}.
      \end{note}

    \item[\textbf{$\omega$-CPOs and Scott-continuous Functions}]
      is the category whose objects are $\omega$-CPOs and arrows are
      Scott-continuous functions.
      \begin{note}
        A category is also formed with \textit{strict} $\omega$-CPOs and
        \textit{strict} Scott-continous functions.
      \end{note}

    \item[\textbf{Path Category}]
      is the free category $F(\mathscr{G})$ generated by a graph
      $\mathscr{G}$ whose objects are the nodes of $\mathscr{G}$ and arrows
      are paths in $\mathscr{G}$. For each node/object $A$, the identity
      arrow is the empty path from $A$ to $A$. Composition is defined by
      $(f_1,\dots,f_k)\circ(f_{k+1},\dots,f_n)=(f_1,\dots,f_n)$

    \item[\textbf{Cat}]
      is the category whose objects are small categories and arrows are
      functors between small categories.

    \item[$M$-\textbf{Act}]
      is the category whose objects are monoid actions and arrows are
      equivariant maps. Since equivariant maps are functions, the identity
      arrow is the identity equivariant map and composition is function
      composition.

    \item[\textbf{FBool}]
      is the category whose objects are finite Boolean algebras and arrows
      are homormophisms between them.

    \item[\textbf{Pts}]
      is the category whose objects are pointed sets and arrows are
      functions that preserves the pointed object.

    \item[$\mathbf{Func}(\mathscr{C},\mathscr{D})$]
      is the category whose objects are functors from categories
      $\mathscr{C}$ to $\mathscr{D}$ and arrows are natural transformations.

    \item[$cone(D)$]
      is the category whose objects are cones with base $D$ and arrows are
      morphisms between cones.

    \item[$\mathbf{SO}(\mathscr{C})$]
      is the category whose objects are split opfibrations of the category
      $\mathscr{C}$ and arrows are homomorphisms of split opfibrations.

    \item[$\mathbf{End}(M)$]
      is the category of endomorphisms for the monoid $M$.

    \item[$(F:\mathscr{C})$]
      is the category of F-algebras and homomorphisms between F-algebras for
      the endofunctor $F:\mathscr{C}\rightarrow\mathscr{C}$.

    \item[$act(A)$]
      is the category of algebras for the functor $A\times -$ where $A$ is
      an object in $\mathscr{C}$. Thus, objects of $act(A)$ are $x:A\times
      X\rightarrow X$ for each object $X$ in $\mathscr{C}$ and arrows are
      homomorphisms between these algebras.

    \item[$\mathscr{K}(M)$]
      is the \textit{Kleisli} category for a monad $(M,\eta,\mu)$ in a
      category $\mathscr{C}$. $\mathscr{K}(M)$ has the same objects as
      $\mathscr{C}$ but an arrow from $A$ to $B$ is an arrow of the form
      $f:A\rightarrow M(B)$ where $A$ and $B$ are objects in $\mathscr{C}$.
      Let $\ast$ denote composition in $\mathscr{K}(M)$, for another arrow
      $g:B\rightarrow M(C)$, composition is defined as $g\ast f=\mu C\circ
      M(g)\circ f:A\rightarrow M(C)$. For any object $A$, the identity arrow
      is $\eta A:A\rightarrow M(A)$.
      \begin{note}
        There are functors $U:\mathscr{K}(M)\rightarrow\mathscr{C}$ and
        $F:\mathscr{C}\rightarrow\mathscr{K}(M)$ defined by $U(A)=M(A)$,
        $U(f:A\rightarrow B)=\mu B\circ M(f)$ and $F(A)=A$,
        $F(g:A\rightarrow B)=M(g)\circ\eta A$. Then $F\dashv U$ and
        $M=U\circ F$.
      \end{note}

    \item[$\mathbf{Psh}(\mathscr{C})$]
      is the category where objects are presheaves on $\mathscr{C}$ and
      arrows are natural transformations between presheaves.
      \begin{note}
        The category of presheaves on a category $\mathscr{C}$ forms a
        topos.
      \end{note}
  \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{functorexamples}
{
  name={Examples of Functors:}
} {
  \begin{itemize}[label={}]
    \item
  \begin{itemize}[align=left]
    \item[\textbf{Monoid Homomorphism}]
      $f:M\rightarrow N$ where $M$ and $N$ are monoids is a functor from the
      category $C(M)$ to the category $C(N)$.

    \item[\textbf{Monotone Map}]
      between two posets is a functor between the category determined by the
      posets.

    \item[\textbf{Projection}]
      The first and second projections from the product category to its
      first or second parts,
      $P_1:\mathscr{C}\times\mathscr{D}\rightarrow\mathscr{C}$ and
      $P_2:\mathscr{C}\times\mathscr{D}\rightarrow\mathscr{D}$ for
      categories $\mathscr{C}$ and $\mathscr{D}$ are functors.

    \item[\textbf{Inclusion Map}]
      of a subcategory is a functor.
      \begin{note}
        By the definition of \textit{subcategory}, objects and arrows of the
        subcategory need not be objects and arrows in the bigger category.
      \end{note}
      \begin{note}[Example]
        \textbf{Set} is a subcategory of \textbf{Rel} and the functor is
        defined by taking sets to the sets as objects in \textbf{Rel} and
        set functions $f:S\rightarrow T$ to $\{(s,t)|t=f(s)\}$.
      \end{note}

    \item[\textbf{Monoid Actions}]
      For an action $\alpha$ of a monoid $M$ acting on a set $S$, let $C(M)$
      be the category determined by $M$ and the action $\alpha$ determines a
      functor $F_\alpha:C(M)\rightarrow\mathbf{Set}$ defined as:
      \begin{itemize}[nosep, label={-}]
        \item $F_\alpha(*)=S$ where $*$ denotes the single object in $C(M)$.
        \item $F_\alpha(m)=s\mapsto\alpha(m,s)$ for $m\in M$, $s\in S$.
      \end{itemize}

    \item[\textbf{Quotient Functor}]
      $Q:\mathscr{C}\rightarrow\mathscr{C}/\sim$\\
      is the functor thats takes a category to its quotient category by
      where $Q(A)=A$ for any object $A$ and $Q(f)=[f]$ for arrows $f$ in the
      category $\mathscr{C}$.

    \item[\textbf{Pullback Functor}]
      Let $\mathscr{C}$ be a category with pullbacks and $f:A\rightarrow B$
      is an arrow in $\mathscr{C}$. The \textbf{pullback functor}
      $f^*:\mathscr{C}/B\rightarrow\mathscr{C}/A$ is defined as:
      \begin{itemize}
        \item For an object $x:X\rightarrow B$, $f^*(x)=p_1:P\rightarrow A$
          of the following pullback diagram:
          \begin{mytikzcd}
            P \ar[r,"p_2"] \ar[d,"p_1"'] \& X \ar[d,"x"] \\
            A \ar[r,"f"'] \& B
          \end{mytikzcd}
        \item For an arrow $t:x\rightarrow x'$ and objects $x:X\rightarrow
          B$, $x':X'\rightarrow B$, $x'\circ t=x$ and $f^*(t)=s$ of the
          following pullback diagram:
          \begin{mytikzcd}
            P \ar[rd,"s"] \ar[rdd,bend right=25,"p_1"']
                          \ar[rrd,bend left=25,"t"] \\
            \& P' \ar[r,"p_2'"] \ar[d,"p_1'"'] \& X' \ar[d,"x'"] \\
            \& A \ar[r,"f"'] \& B
          \end{mytikzcd}
          where $s:P\rightarrow P'$ is the unique arrow by the definition of
          a pullback.
      \end{itemize}

  \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{forgetfulfunctor}
{
  name=Forgetful/Underlying Functor,
  text=forgetful functor
} {
  is the functor which ``forgets'' some of the structure in a
  category of structures and structure-preserving functions.
  \begin{itemize}[label={}, align=left, nolistsep]
    \item[\textit{Examples}:]
    \item
    \begin{itemize}

      \item $U:\mathbf{Sem}\rightarrow\mathbf{Set}$

      \item $U:\mathbf{Grf}\rightarrow\mathbf{Set}\times\mathbf{Set}$\\

        $A:\mathbf{Grf}\rightarrow\mathbf{Set}$ and nodeset functor

        $O:\mathbf{Cat}\rightarrow\mathbf{Set}$ which takes a category to

      \item $U:\mathscr{C}/A\rightarrow\mathscr{C}$\\
        for an object $A$ in the category $\mathscr{C}$.
        \begin{note}
          If $\mathscr{C}$ is \textbf{Set}, the underlying functor

  \end{itemize}

\longnewglossaryentry{freefunctor}
{
  name=Free Functor,
  text=free functor
} {
  Informally, a \textbf{free functor} is left adjoint to a \textbf{forgetful
  functor}.
  \begin{itemize}[label={}, align=left, nolistsep]
    \item[\textit{Examples}:]
    \item
    \begin{itemize}
      \item $F:\mathbf{Set}\rightarrow\mathbf{Mon}$\\
        is the free monoid functor that takes a set $A$ to the free monoid
        $F(A)$, the Kleene closure $A^*$ with concatenation as the monoid
        operation. $F$ takes set functions to the Kleene closure induced
        homormophism.
        \begin{note}
          The Kleene closure itself is a functor from \textbf{Set} to
          \textbf{Set}, which is the composition of $U\circ F$, the
          underlying functor $U:\mathbf{Mon}\rightarrow\mathbf{Set}$ and the
          free monoid functor.
        \end{note}

      \item $F:\mathbf{Grf}\rightarrow\mathbf{Cat}$\\
        takes graphs to the path category of the graph as objects in
        \textbf{Cat} and graph homomorphisms to arrows (functors between
        path categories). For a graph homomorphism
        $\phi:\mathscr{G}\rightarrow\mathscr{H}$,
        $F(\phi)(f_n,f_{n-1},\dots,f_1)=(\phi(f_n),\phi(f_{n-1}),\dots,\phi(f_1))$,
        which is clearly a path in $\mathscr{H}$ (and thus an arrow in the
        path category of $\mathscr{H}$).

      \item $F:\mathbf{Set}\rightarrow\mathscr{L}$\\
        takes a set $S$ to the category $\mathscr{L}$ of join-semilattice
        and homomorphisms where semilattices are non-empty subsets of $S$
        with inclusion as ordering. If a join-semilattice has bottom, the
        empty $\emptyset$ is the bottom element of the semi-lattice.
        \begin{note}
          Meet-semilattices can be constructed by dualizing the inclusion
          for ordering.
        \end{note}
    \end{itemize}
  \end{itemize}
}

\newglossaryentry{contravariantfunctor}
{
  name=Contravariant Functor,
  text=contravariant functor,
  description={is a functor $F:\mathscr{C}^{op}\rightarrow\mathscr{D}$ for
    categories $\mathscr{C}$ and $\mathscr{D}$. In contrast to
    \textit{covariant} functors, for an arrow $f$ in $\mathscr{C}$
    \begin{center}
      $F(f:B\rightarrow A)=F(A)\rightarrow F(B)$
    \end{center}
    is an arrow in $\mathscr{D}$. This follows from the fact that $f$ is an
  arrow from $A$ to $B$ in $\mathscr{C}^{op}$.}
}

\longnewglossaryentry{powersetfunctor}
{
  name=Powerset Functor,
  text=powerset functor
} {
  is a functor that takes a set $S$ to its powerset
  $\mathcal{P}(S)$. There are three different powerset functors:

  \begin{itemize}[label={}, nosep]
    \item
      \begin{itemize}[align=left]
        \item[\textbf{Inverse Image}]
          $\mathscr{P}:\mathbf{Set}^{op}\rightarrow\mathbf{Set}$\\
          is a contravariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to its inverse image
          $f:\mathcal{P}(B)\rightarrow\mathcal{P}(A)$.

        \item[\textbf{Direct/Existential Image}]
          is a covariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to the function
          $f_*:\mathcal{P}(A)\rightarrow\mathcal{P}(B)$ where
          $f_*(A_0)=\{f(x)|x\in A_0\}$.

        \item[\textbf{Universal Image}]
          is a covariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to the function
          $f_!:\mathcal{P}(A)\rightarrow\mathcal{P}(B)$ where
          $f_!(A_0)=\{y\in B|(f(x)=y)\Rightarrow(x\in A_0)\}$.
      \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{homfunctor}
{
  name=Hom Functor,
  text=hom functor
} {
  is a functor that takes a category to a hom-set. For any category
  $\mathscr{C}$ with an object $C$, there are three different hom functors:

  \begin{itemize}[label={}, nosep]
    \item
      \begin{itemize}[align=left]
        \item[\textbf{Covariant Hom Functor}]
          $\Hom(C,-):\mathscr{C}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(C,-)(A)=\Hom(C,A)$ for each object
              $A\in\mathscr{C}$.
            \item $\Hom(C,-)(f)=\Hom(C,f):\Hom(C,A)\rightarrow\Hom(C,B)$ for
              an arrow $f:A\rightarrow B$ in $\mathscr{C}$.
          \end{itemize}

        \item[\textbf{Contravariant Hom Functor}]
          $\Hom(-,C):\mathscr{C}^{op}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(-,C)(A)=\Hom(A,C)$ for each object
              $A\in\mathscr{C}$.
            \item $\Hom(-,C)(f)=\Hom(f,C):\Hom(B,C)\rightarrow\Hom(A,C)$ for
              an arrow $f:A\rightarrow B$ in $\mathscr{C}$.
          \end{itemize}

        \item[\textbf{Two-Variable Hom Functor}]
          $\Hom(-,-):\mathscr{C}^{op}\times\mathscr{C}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(-,-)((A,B))=\Hom(A,B)$ for each object
              $(A,B)\in\mathscr{C}^{op}\times\mathscr{C}$.
            \item $\Hom(-,-)((f,g))=\Hom(f,g):\Hom(B,C)\rightarrow\Hom(A,D)$ for
              an arrow $(f:A\rightarrow B, g:C\rightarrow D)$ in
              $\mathscr{C}^{op}\times\mathscr{C}$.
          \end{itemize}

      \end{itemize}
  \end{itemize}

  \begin{note}
    All covariant hom functors preserves products.
  \end{note}
}

\newglossaryentry{faithfunctor}

  name=Faithful,
  text=faithful,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$ is

    \begin{center}
      $\Hom_\mathscr{C}(A,B)\rightarrow\Hom_\mathscr{D}(F(A),F(B))$
    \end{center}
    for each pair of objects $A$ and $B$ in $\mathscr{C}$ induced by $F$ is

%
    \begin{note}

    \end{note}
  }

\newglossaryentry{reflect}
{
  name=Reflect,
  text=reflect,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$
    \textbf{reflects} a property $P$ of arrows if whenever $F(f)$ has property
    $P$ then so does $f$ (where $f$ is any arrow for which $F$ takes to
    $F(f)$).
    \begin{note}[Theorem]
      Consider a full and faithful functor
      $F:\mathscr{C}\rightarrow\mathscr{D}$. Let $A$ and $B$ be objects in
      $C$ and $u:F(A)\rightarrow F(B)$ is an isomorphism in $\mathscr{D}$,
      then there is an unique isomorphism $f:A\rightarrow
      B$ in $\mathscr{C}$ for which $F(f)=u$.
    \end{note}
    \begin{note}[Corollary]
      A full and faithful functor reflects isomorphisms.
    \end{note}
    \begin{note}[Corollary]
      For a full an faithful functor $F$, if $F(A)=F(B)$ for objects $A$ and
      $B$ in the domain of $F$, then $A\cong B$.
    \end{note}
  }
}

\longnewglossaryentry{categoryequivalence}
{
  name=Category Equivalence,
  text=category equivalence,
} {
  A functor $F:\mathscr{C}\rightarrow\mathscr{D}$ is an \textbf{equivalence
  of categories} if there is a functor $G:\mathscr{D}\rightarrow\mathscr{C}$
  such that:
  \begin{itemize}[nosep]
    \item A family $u_C:C\rightarrow G(F(C))$ of isomorphisms of
      $\mathscr{C}$ indexed by the objects of $\mathscr{C}$ with the
      property that for every arrow $f:C\rightarrow C'$ in
      $\mathscr{C}$, $G(F(f))=u_{C'}\circ f\circ u_C^{-1}$.

    \item A family $v_D:D\rightarrow F(G(D))$ of isomorphisms of
      $\mathscr{D}$ indexed by the objects of $\mathscr{D}$ with the
      property that for every arrow $g:D\rightarrow D'$ in $\mathscr{D}$,
      $F(G(g))=v_{D'}\circ g\circ v_D^{-1}$.
  \end{itemize}

  \begin{note}
    In this case, the functor $G$ is called a \textit{pseudo-inverse} of $F$
    and $u,v$ are \textit{natural isomorphisms}.
  \end{note}

  \begin{note}
    $F$ and $G$ are both full and faithful.
  \end{note}
}

\longnewglossaryentry{congruence}
{
  name=Congruence Relation,
  text=congruence relation,
  symbol={$\sim$}
} {
  is an equivalence relation $\sim$ on the arrows of a category
  $\mathscr{C}$ if:
  \begin{itemize}
    \item Whenever $f\sim g$, then $f$ and $g$ have the same domain and the
      same codomain.
    \item In the diagram below:
      \begin{mytikzcd}
        A \ar[r,"h"] \& B \ar[r,"f",shift left] \ar[r,"g"',shift right]
                     \& C \ar[r,"k"]
                     \& D
      \end{mytikzcd}
      If $f\sim g$, then $f\circ h\sim g\circ h$ and $k\circ f\sim
      k\circ g$ for arrows $h,k$ where $target(h)=source(f)$ and
      $source(k)=target(f)$.
  \end{itemize}
  \begin{note}
    The congruence class containing the arrow $f$ is denoted by $[f]$.
  \end{note}
}

\longnewglossaryentry{quotientcategory}
{
  name=Quotient Category,
  text=quotient category,
  symbol={$/\sim$}
} {
  For a congruence relation $\sim$ on the arrows of a category
  $\mathscr{C}$, the \textbf{quotient category} $\mathscr{C}/\sim$ is
  defined as follows:
  \begin{itemize}
    \item The objects of $\mathscr{C}/\sim$ are the objects of
      $\mathscr{C}$.
    \item The arrows of $\mathscr{C}/\sim$ are the congruence classes of
      arrows of $\mathscr{C}$.
    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $[f]:A\rightarrow B$
      in $\mathscr{C}/\sim$.
    \item If $f:A\rightarrow B$ and $g:B\rightarrow C$ in $\mathscr{C}$,
      then $[g]\circ[f]=[g\circ f]:A\rightarrow C$ in $\mathscr{C}$.
  \end{itemize}

  \begin{note}
    For any functor $F:\mathscr{C}\rightarrow\mathscr{D}$ with the property
    that if $f\sim g$ then $F(f)=F(g)$, then there is an unique functor
    $F_0:\mathscr{C}/\sim\rightarrow\mathscr{D}$ for which $F_0\circ Q=F$.
    Here, $Q$ is the quotient functor,
    $Q:\mathscr{C}\rightarrow\mathscr{C}/\sim$.
  \end{note}
}

\newglossaryentry{naturalisomorphism}
{
  name=Natural Isomorphism,
  text=natural isomorphism,
  description={is a natural transformation $\alpha:F\rightarrow G$ in which
    there is a natural transformation $\beta:G\rightarrow F$ is an inverse
    to $\alpha$ in $\mathbf{Func}(\mathscr{C},\mathscr{D})$.
    \begin{note}
      Natural isomorphisms are also known as \textit{natural equivalences}.
    \end{note}
    \begin{note}[Theorem]
      Suppose $F:\mathscr{C}\rightarrow\mathscr{D},
      G:\mathscr{D}\rightarrow\mathscr{C}$ are functors and
      $\alpha:F\rightarrow G$ is a natural transformation between them. Then
      $\alpha$ is a \textit{natural isomorphism} if and only if for each
      object $C\in\mathscr{C}$, $\alpha C:F(C)\rightarrow G(C)$ is an
      isomorphism in $\mathscr{D}$.
    \end{note}
  }
}

\newglossaryentry{subfunctor}
{
  name=Subfunctor,
  text=subfunctor,
  description={A \textbf{subfunctor} of a functor
    $F:\mathscr{C}\rightarrow\mathbf{Set}$ is a functor
    $G:\mathscr{C}\rightarrow\mathbf{Set}$ with the property that for each
    object $C\in\mathscr{C}$, $G(C)\subseteq F(C)$ and for each arrow
    $f:C\rightarrow C'$ and element $x\in G(C)$, $G(f)(x)=F(f)(x)$.
    \begin{note}
      The inclusion function $i_C:G(C)\rightarrow F(C)$ that takes the
      the subfunctor $G$ to the functor $F$ is a natural transformation.
    \end{note}
  }
}

\longnewglossaryentry{factorizationsystem}
{
  name=Factorization System,
  text=factorization system,
} {
  A \textbf{factorization system} in a category $\mathscr{C}$ consists of
  two subclasses $\mathscr{E}$ and $\mathscr{M}$ of the arrows of
  $\mathscr{C}$ such that:
  \begin{itemize}
    \item If $\mathscr{I}$ is the class of isomorphisms, then
      $\mathscr{M}\circ\mathscr{I}\subseteq\mathscr{M}$ and
      $\mathscr{I}\circ\mathscr{E}\subseteq\mathscr{E}$.

    \item Every arrow $f\in\mathscr{C}$ factors as $f=m\circ e$ where
      $m\in\mathscr{M}$ and $e\in\mathscr{E}$.

    \item In any commutative square
      \begin{mytikzcd}
        A \ar[r,"e",twoheadrightarrow] \ar[d,"f"'] \& B \ar[d,"g"] \\
        C \ar[r,"m"',rightarrowtail]               \& D
      \end{mytikzcd}
      with $e\in\mathscr{E}$ and $m\in\mathscr{M}$, there is a unique
      $h:B\rightarrow C$ such that $h\circ e=f$ and $m\circ h=g$.
  \end{itemize}
}

\longnewglossaryentry{naturaltransformation}
{
  name=Natural Transformation,
  text=natural transformation
} {
  Let $F,G:\mathscr{C}\rightarrow\mathscr{D}$ be 2 functors with the same
  domain and codomain. A \textbf{natural transformation}
  $\alpha:F\rightarrow G$ is given by a family of arrows $\alpha C$ for each
  object $C\in\mathscr{C}$ such that:
  \begin{itemize}
    \item $\alpha C:F(C)\rightarrow G(C)$ for each object $C\in\mathscr{C}$,
      also say that $\alpha$ is natural in $C$.
    \item For any arrow $f:C\rightarrow D$ in $\mathscr{C}$, the following
      diagram commutes.
      \begin{mytikzcd}
        F(C) \ar[r, "\alpha C"]  \ar[d, "F(f)"'] \& G(C) \ar[d, "G(f)"] \\
        F(D) \ar[r, "\alpha D"']                 \& G(D)
      \end{mytikzcd}
  \end{itemize}

  \begin{note}[Composition with Functors]
    Let $H:\mathscr{B}\rightarrow\mathscr{C}$ be a functor and
    $\alpha:F\rightarrow G$ where $F,G:\mathscr{A}\rightarrow\mathscr{B}$ be
    are functors. Then, postcomposing $\alpha$ with the functor $H$ is a
    natural transformation $H\alpha:H\circ F\rightarrow H\circ G$ defined by
    $(H\alpha)A=H(\alpha A)$ for any object $A\in\mathscr{A}$.\\
    A similar result can be shown when precomposing a natural transformation
    with a functor.
  \end{note}
}

\newglossaryentry{verticalcomposition}
{
  name=Vertical Composition,
  text=vertical composition,
  description={Composition of natural transformations $\alpha:F\rightarrow
    G$ and $\beta:G\rightarrow H$ is $\beta\circ\alpha:F\rightarrow H$ and
    is known as \textbf{vertical composition}, where $F, G$ and $H$ are
    functors from the categories $\mathscr{C}$ to $\mathscr{D}$. This
    follows from the fact that the outer rectangle of the following diagram
    is commutative:
    \begin{mytikzcd}
         F(A) \ar[r, "\alpha A"] \ar[d, "F(f)"']
      \& G(A) \ar[r, "\beta A"]  \ar[d, "G(f)"]
      \& H(A)                    \ar[d, "H(f)"] \\
         F(B) \ar[r, "\alpha B"']
      \& G(B) \ar[r, "\beta B"']
      \& H(B)
    \end{mytikzcd}
    for each arrow $f:A\rightarrow B$.
    \begin{note}
      The vertical composite of natural transformations is also a natural
      transformation.
      \begin{equation*}
        \begin{tikzcd}[ampersand replacement=\&]
             \mathscr{C} \ar[r, bend left=70, "F"] \ar[r, phantom, bend left=70, ""{name=F, below}]
                         \ar[r, "G"']              \ar[r, phantom, ""{name=G, above}]
                         \ar[Rightarrow, from=F, to=G, "\alpha"]
          \& \mathscr{D}
        \end{tikzcd}
        \circ
        \begin{tikzcd}[ampersand replacement=\&]
             \mathscr{C} \ar[r, bend right=70, "H"'] \ar[r, phantom, bend right=70, ""{name=H, above}]
                         \ar[r, "G"]                 \ar[r, phantom, ""{name=G, below}]
                         \ar[Rightarrow, from=G, to=H, "\beta"]
          \& \mathscr{D}
        \end{tikzcd}
        \longmapsto
        \begin{tikzcd}[ampersand replacement=\&]
             \mathscr{C} \ar[r, bend left=70, "F"]   \ar[r, phantom, bend left=70, ""{name=F, below}]
                         \ar[r, bend right=70, "H"'] \ar[r, phantom, bend right=70, ""{name=H, above}]
                         \ar[Rightarrow, from=F, to=H, "\beta\circ\alpha"]
          \& \mathscr{D}
        \end{tikzcd}
      \end{equation*}
    \end{note}
  },
  symbol={$\circ$}
}

\newglossaryentry{horizontalcomposition}
{
  name=Horizontal Composition,
  text=horizontal composition,
  description={Let $\mathscr{A}, \mathscr{B}$ and $\mathscr{C}$ be
    categories, and $F,F':\mathscr{A}\rightarrow\mathscr{B},
    G,G':\mathscr{B}\rightarrow\mathscr{C}$ be functors. \textbf{Horizontal
    composition} between natural transformation
    transformations $\alpha:F\rightarrow F'$ and $\beta:G\rightarrow G'$ is
    $\beta\ast\alpha:G\circ F\rightarrow G'\circ F'$.
    \begin{note}
      The horizontal composite of natural transformations is also a natural
      transformation.
      \begin{equation*}
        \begin{tikzcd}[ampersand replacement=\&]
             \mathscr{A} \ar[r, bend left=70, "F"]    \ar[r, phantom, bend left=70, ""{name=F, below}]
                         \ar[r, bend right=70, "F'"'] \ar[r, phantom, bend right=70, ""{name=F', above}]
                         \ar[Rightarrow, from=F, to=F', "\alpha"]
          \& \mathscr{B}
        \end{tikzcd}
        \ast
        \begin{tikzcd}[ampersand replacement=\&]
             \mathscr{B} \ar[r, bend left=70, "G"]    \ar[r, phantom, bend left=70, ""{name=G, below}]
                         \ar[r, bend right=70, "G'"'] \ar[r, phantom, bend right=70, ""{name=G', above}]
                         \ar[Rightarrow, from=G, to=G', "\beta"]
          \& \mathscr{C}
        \end{tikzcd}
        \longmapsto
        \begin{tikzcd}[ampersand replacement=\&]
             \mathscr{A} \ar[r, bend left=70, "G\circ F"]     \ar[r, phantom, bend left=70, ""{name=GF, below}]
                         \ar[r, bend right=70, "G'\circ F'"'] \ar[r, phantom, bend right=70, ""{name=G'F', above}]
                         \ar[Rightarrow, from=GF, to=G'F', "\beta\ast\alpha"]
          \& \mathscr{C}
        \end{tikzcd}
      \end{equation*}
    \end{note}
  }
}

\longnewglossaryentry{yonedaembedding}
{
  name=Yoneda Embedding,
  text=yoneda embedding
} {
  For any category $\mathscr{C}$, there is a contravariant \textit{Yoneda
  Functor} $Y:\mathscr{C}^{op}\rightarrow\mathbf{Func}(C,\mathbf{Set})$
  defined as:
  \begin{itemize}
    \item for objects $C\in\mathscr{C}, Y(C)=\Hom(C,-)$
    \item for an arrow $f:D\rightarrow C$ and an object $A\in\mathscr{C},
      Y(f)=\Hom(f,-)$ and $Y(f)(A):\Hom(C,A)\rightarrow\Hom(D,A)$
  \end{itemize}

  Thus, the following diagram commutes
  \begin{equation*}\begin{tikzcd}[column sep=huge, ampersand replacement=\&]
    {\Hom(D,A)} \ar[r, "{\Hom(D,k)}"] \& {\Hom(D,B)} \\
    {\Hom(C,A)} \ar[u, "Y(f)A"] \ar[r, "{\Hom(C,k)}"'] \& {\Hom(C,B)} \ar[u, "Y(f)B"']
  \end{tikzcd}\end{equation*}
  for an arrow $k:A\rightarrow B$.\\

  \begin{note}
    $Y(f):\Hom(C,-)\rightarrow\Hom(D,-)$ is the induced natural
    transformation between hom functors corresponding to $f$. Generalizing
    $\Hom(D,-)$ to an arbitary set valued functor, by the \textit{Yoneda
    Lemma}, $Y$ is both full and faithful.
  \end{note}

  \begin{note}
    By the contravariant \textit{Yoneda Lemma}, there also exists the
    covariant functor
    $J:\mathscr{C}\rightarrow\mathbf{Func}(\mathscr{C}^{op},\mathbf{Set})$.
  \end{note}
}

\longnewglossaryentry{godementcalculus}
{
  name=Godement Calculus,
  text=Godement Calculus
} {
  Let $\mathscr{A,B,C,D,E}$ be categories, $E, F_1, F_2,
  F_3, G_1, G_2, G_3, H$ be functors and $\alpha,\beta,\gamma,\delta$ be
  natrual morphisms as shown in the following diagram:
  \begin{mytikzcd}
       \mathscr{A} \ar[r,"E"]
    \& \mathscr{B} \ar[r,bend left=70,"F_1"]
                   \ar[r,phantom,bend left=70,""{name=F1,below}]
                   \ar[r,"F_2" description]
                   \ar[r,phantom,""{name=F2A,above}]
                   \ar[r,phantom,""{name=F2B,below}]
                   \ar[r,bend right=70,"F_3"']
                   \ar[r,phantom,bend right=70,""{name=F3,above}]
                   \ar[Rightarrow,from=F1,to=F2A,"\alpha"]
                   \ar[Rightarrow,from=F2B,to=F3,"\beta"]
    \& \mathscr{C} \ar[r,bend left=70,"G_1"]
                   \ar[r,phantom,bend left=70,""{name=G1,below}]
                   \ar[r,"G_2" description]
                   \ar[r,phantom,""{name=G2A,above}]
                   \ar[r,phantom,""{name=G2B,below}]
                   \ar[r,bend right=70,"G_3"']
                   \ar[r,phantom,bend right=70,""{name=G3,above}]
                   \ar[Rightarrow,from=G1,to=G2A,"\gamma"]
                   \ar[Rightarrow,from=G2B,to=G3,"\delta"]
    \& \mathscr{D} \ar[r,"H"]
    \& \mathscr{E}
  \end{mytikzcd}
  There are 5 laws/equational rules between composition of functors and
  natural transformations:
  \begin{enumerate}
    \item
      $(\delta\circ\gamma)\ast(\beta\circ\alpha)=(\delta\ast\beta)\circ(\gamma\circ\alpha)$
    \item $(H\circ G_1)\alpha=H(G_1\alpha)$
    \item $\gamma(F_1\circ E)=(\gamma F_1)E$
    \item $G_1(\beta\circ\alpha)E=(G_1\beta E)\circ(G_1\alpha E)$
    \item
      $\gamma\ast\alpha=(\gamma F_2)\circ(G_1\alpha)=(G_2\alpha)\circ(\gamma F_1)$
  \end{enumerate}
  \begin{note}
    The first rule is also known as the \textit{Interchange Law}.
  \end{note}
}

\longnewglossaryentry{yonedalemma}
{
  name=Yoneda Lemma,
  text=yoneda lemma
} {
  Let $F:\mathscr{C}\rightarrow\mathbf{Set}$ be a \textit{set-valued
  functor} and $C$ is an object in $C$. There is a one-to-one correspondance
  between elements of $F(C)$ and $h^C:\Hom(C,-)\rightarrow F$, that is:
  \begin{equation*}
    Nat(h^C,F)\cong F(C)
  \end{equation*}
  Thus, this is an isomorphism that is both an injective and surjective
  mapping between elements of $F(C)$ and $h^C$.

  \begin{note}
    $h^C(X)=\Hom(C,X)\rightarrow F(X)$ for an object $X\in\mathscr{C}$
    is defined as $h^C(X)(f)=F(f)(c)$ where $f:C\rightarrow X$ is an arrow
    and $c\in F(C)$ is an element.
  \end{note}
  \begin{note}
    There is also a contravariant version of Yoneda Lemma involving the
    contravariant hom functor $h_C:\Hom(-,C)\rightarrow F$.
  \end{note}
}

\longnewglossaryentry{universalelement}
{
  name=Universal Element,
  text=universal element
} {
  By the \textit{Yoneda Lemma}, an element $c\in F(C)$ is given by
  $c=h^C(id_C)$. $c$ is an \textbf{universal element} of a functor
  $F:\mathscr{C}\rightarrow\mathbf{Set}$ if and only if the induced
  $h^C:Hom(C,-)\rightarrow F$ is a \textit{natural isomorphism}.
  Consequently, for any object $C'\in\mathscr{C}$ and any element $x\in
  F(C')$, there is an unique arrow $f:C\rightarrow C'$ in $\mathscr{C}$ for
  which $x=F(f)(c)$.

  \begin{note}
    There is a one-to-one correspondance between representations and
    universal elements of $F$.
  \end{note}
  \begin{note}
    A similar argument for contravariant set valued functors of the form
    $F:\mathscr{C}^{op}\rightarrow\mathbf{Set}$ can be made where $c\in
    F(C)$ is an \textbf{universal element} if for any object
    $C'\in\mathscr{C}$ and any element $x\in F(C')$, there is an unique
    arrow $f:C'\rightarrow C$ for which $x=F(f)(c)$.
  \end{note}
}

\longnewglossaryentry{2category}
{
  name=2-Category,
  text=2-category,
} {
  is a category \textbf{C} with the following additional structure:
  \begin{itemize}
    \item A set of objects of \textbf{C} (called \textit{0-cells}).

    \item $\forall A,B\in\mathbf{C}_0$ (for every pair of 0-cells), a small
      category $\mathbf{C}(A,B)$ whose objects are arrows from $A$ to $B$ in
      \textbf{C} (objects and arrows in $\mathbf{C}(A,B)$ are called
      \textit{1-cells} and \textit{2-cells} respectively).

    \item $\forall A,B,C\in\mathbf{C}_0$ (for every triple of 0-cells),
      there is a composition functor:
      \begin{equation*}
        comp:\mathbf{C}(B,C)\times\mathbf{C}(A,B)\longrightarrow\mathbf{C}(A,C)
      \end{equation*}
      and that the following diagram commutes:
      \begin{equation*}\begin{tikzcd}[column sep=huge,ampersand replacement=\&]
           {\mathbf{C}(C,D)\times\mathbf{C}(B,C)\times\mathbf{C}(A,B)}
             \ar[r,"comp\times id"]
             \ar[d,"id\times comp"']
        \& {\mathbf{C}(B,D)\times\mathbf{C}(A,B)}
             \ar[d,"comp"]
        \\
           {\mathbf{C}(C,D)\times\mathbf{C}(A,C)}
             \ar[r,"comp"']
        \& {\mathbf{C}(A,D)}
      \end{tikzcd}\end{equation*}
      ie: $comp$ is associative

    \item For each object $A\in\mathbf{C}$, there is the identity functor:
      \begin{equation*}
        unit_A:1\rightarrow\mathbf{C}(A,A)
      \end{equation*}
      where $1$ is the terminal object in \textbf{Cat} (ie: the category
      with one object denoted as $0$ and its identity arrow) defined as
      $unit_A(0)=id_A$ and that the following diagram commutes:
      \begin{equation*}\begin{tikzcd}[column sep=huge,ampersand replacement=\&]
           \& {\mathbf{C}(A,B)} \ar[ld,"unit_B\times id"']
                               \ar[d,"id"]
                               \ar[rd,"id\times unit_A"]
        \\    {\mathbf{C}(B,B)\times\mathbf{C}(A,B)} \ar[r,"comp"']
           \& {\mathbf{C}(A,B)}
           \& {\mathbf{C}(A,B)\times\mathbf{C}(A,A)} \ar[l,"comp"']
      \end{tikzcd}\end{equation*}
      ie: $unit$ is the identity functor with respect to $comp$

  \end{itemize}

  The structure of a 2-category can be summarized as follows:
  \begin{itemize}
    \item A \textit{horizontal category} whose objects are 0-cells, arrows are
      2-cells.
    \item A \textit{vertical category} whose objects are 1-cells, arrows are
      2-cells.
    \item A \textit{base category} (that can be derived from the horizontal
      and vertical categories) whose objects are 0-cells, arrows are
      1-cells.
  \end{itemize}

  \begin{note}[Visual Depictions]
     \item[]
     \begin{itemize}
       \item[0-cells] $x,y$
       \item[1-cells] $x\rightarrow y$
       \item[2-cells]
         \begin{tikzcd}[ampersand replacement=\&]
              x \ar[r,bend left=70,""{name=F,below}]
                \ar[r,bend right=70,""{name=G,above}]
                \ar[Rightarrow,from=F,to=G]
           \& y
         \end{tikzcd}
     \end{itemize}
  \end{note}

  \textbf{Cat} as a 2-category:
  \begin{itemize}
    \item 0-cells are small categories.
    \item 1-cells are functors.
    \item 2-cells are natural transformations.
    \item $comp$ on 1-cells is composition of functors:
      \begin{equation*}
        comp(G,F)=G\circ F
      \end{equation*}
      for arrows $F:A\rightarrow B,G:B\rightarrow C$ in \textbf{Cat}.
    \item $comp$ on 2-cells is horizontal composition of natural
      transformations:
      \begin{equation*}
        comp(\beta,\alpha)=\beta\ast\alpha
      \end{equation*}
      where $F,F':A\rightarrow B$ and $G,G':B\rightarrow C$ are functors
      $\alpha:F\rightarrow F',\beta:G\rightarrow G'$ are natural
      transformations between them.
    \item By the \textit{Interchange Law}
      \begin{equation*}
        comp((\beta',\alpha')\circ(\beta,\alpha))=comp(\beta',\alpha')\circ comp(\beta,\alpha)
      \end{equation*}
  \end{itemize}
}

\longnewglossaryentry{product}
{
  name=Product,
  text=product,
  symbol={$\times$}
} {
  Let $A,B$ be objects in a category $\mathscr{C}$. A \textbf{product} of $A$ and
  $B$ is an object $A\times B$ together with projections/arrows
  $p_1:A\times B\rightarrow A$ and $p_2:A\times B\rightarrow B$ such that
  for any object $D$ and arrows
  $q_1:D\rightarrow A$, $q_2:D\rightarrow B$, there is an unique arrow
  $q:D\rightarrow A\times B$ and that the following diagram commutes:
  \begin{mytikzcd}
      \& D \ar[ld,"q_1"'] \ar[d,"q"] \ar[rd,"q_2"] \\
    A \&
    {A\times B} \ar[l,"p_1"] \ar[r,"p_2"'] \&
    B
  \end{mytikzcd}
  ie, $p_1\circ q=q_1$ and $p_2\circ q=q_2$.\\

  The concept of a product is defined up to an unique isomorphism and is
  defined by a \textit{universal mapping property}. Thus, all products of
  the objects $A$ and $B$ are isomorphic and any object isomorphic to
  $A\times B$ is a product of $A$ and $B$.\\

  Since for each pair of arrows $(q1,q2)$ produces an unique arrow $q$,
  there is a natural isomorphism:
  \begin{equation*}
    \pi:\Hom_\mathscr{C}(-,A)\times\Hom_\mathscr{C}(-,B)\rightarrow
    \Hom_\mathscr{C}(-,A\times B)
  \end{equation*}
  such that $\pi D(q_1,q_2)=q$ and is denoted as $\langle q_1,q_2\rangle$.

  \begin{note}
    $(p1,p2)$ is an \textit{universal element} of the functor
    $\Hom(-,A)\times\Hom(-,B)$.
  \end{note}
}

\longnewglossaryentry{productofarrows}

  name=Product of Arrows,
  text=product of arrows

  For arrows $f:S\rightarrow S',t:T\rightarrow T'$, the \textbf{product of
  arrows}, $f\times g$ is defined such that the following diagram commutes:
  \begin{mytikzcd}
       S \ar[d,"f"']
    \& {S\times T} \ar[l,"p1"'] \ar[r,"p2"] \ar[d,"f\times g"]
    \& T \ar[d,"g"] \\

    $f_i:A_i\rightarrow B_i,g_i:B_i\rightarrow C_i$ for $i=1,2$:

      (g_1\circ f_1)\times(g_2\circ f_2)=(g_1\times g_2)\circ(f_1\times f_2)
    \end{equation*}
  \end{note}

\longnewglossaryentry{naryproduct}

  name=n-Ary Product,

  projectiions $p_i:\Pi^n_{i=1}A_i\rightarrow A_i$ and the property that

  \begin{itemize}[align=left,nosep]

    \textit{Ternary products} (3-ary products) as

    \end{equation*}

\longnewglossaryentry{coproduct}
{
  name=Coproduct/Sum,
  text=coproduct,
  symbol={$+$}
} {
  is the categorial dual of \textit{product}. A \textbf{coproduct} $A+B$ of
  two objects $A$ and $B$ in a category consists of an object $A+B$ together
  with arrows/\textit{coprojections} $i_1:A\rightarrow A+B$ and
  $i_2:B\rightarrow A+B$ such that given any arrows $f:A\rightarrow C$ and
  $g:B\rightarrow C$, the following diagram commutes:
  \begin{mytikzcd}
       A \ar[r,"f"] \ar[rd,"i_1"']
    \& C
    \& B \ar[l,"g"'] \ar[ld,"i_2"] \\
    \& {A+B} \ar[u,"\langle f|g\rangle"]
  \end{mytikzcd}
  ie, $\langle f|g\rangle\circ i_1=f$ and $\langle f|g\rangle\circ i_2=g$.\\

  \begin{note}
    The arrows $i_1,i_2$ are also known as \textit{canonical injections} or
    \textit{inclusions}.
  \end{note}

  \begin{note}
    The functor $\Hom(A,-)\times\Hom(B,-)$ is represented by $A+B$ and the
    universal element is the pair of canonical injections.
  \end{note}

\longnewglossaryentry{distributivecategory}
{
  name=Distributive Category,
  text=distributive category
} {
  is a category with both finite products and coproducts. Thus for all
  objects $A,B$ and $C$, the following diagram commutes
  \begin{mytikzcd}
       {A\times B} \ar[r,"i_1'"] \ar[rd,"A\times i_1"']
    \& {A\times B+A\times C} \ar[d,"d"]
    \& {A\times C} \ar[l,"i_2'"'] \ar[ld,"A\times i_2"] \\
    \& {A\times(B+C)}
  \end{mytikzcd}
  and $d$ is an isomorphism.\\

  In a distributive category $\mathscr{C}$, all canonical projections are
  monic and for any object $A\in\mathscr{C}$, the arrow $\langle
  id_0,!\rangle:0\rightarrow 0\times A$ is an ismorphism (ie: $0\cong
  0\times A$ and thus all distributive categories has a strict initial
  object).

  \begin{note}[Example]
    A \textit{Boolean Algebra} is a distributive category with $\wedge$ as
    product and $\vee$ as coproduct.
  \end{note}
}

\longnewglossaryentry{cartesianclosedcategory}

  text=cartesian closed category
} {
  is a cartesian monoidal category $\mathscr{C}$ such that for every pair of

  \end{equation*}
}

\newglossaryentry{exponentialobject}
{
  name=Exponential Object,
  text=exponential object,
  description={An \textbf{exponential object} $B^A$ is an \textit{internal
    hom}$[A,B]$ in a \textit{cartesian closed category}. $B^A$ is equipped
    with the arrow $eval:B^A\times A\rightarrow B$ and is the universal
    element of the functor $\Hom(-\times A,B)$. This means for any other
    object $C$ and the $eval':C\times A\rightarrow B$, there is an unique
    arrow from $f:C\rightarrow B^A$ such that
    \begin{equation*}
      eval'=eval\circ(f\times id_A)
    \end{equation*}
    An exponential object is also known as \textit{internal hom}.
    \begin{note}
      Since $eval$ is an universal element of the functor $\Hom(-\times
      A,B)$, the process of \textit{currying}, which takes the function
      $f:C\times A\rightarrow B$ to $f':C\rightarrow B^A$ forms a natural
      isomorphism of functors
      \begin{equation*}
        \Hom(-\times A,B)\cong\Hom(-,B^A)
      \end{equation*}
    \end{note}
    \begin{note}
      Consider the sets $A,B$ where $|A|$ is the cardinality of $A$. The
      exponential object $B^A$ is a set of functions from $A$ to $B$.
      $\forall f\in B^A$, $f$ maps each element of $A$ to an element of $B$,
      which can be described as $|A|$-tuple of elements of $B$. Hence, the
      notation for $B^A$ which is an $|A|$ product of $B$.
    \end{note}
    \begin{note}
      Any cartesian closed category with finite coproducts is a
      distributive category.
    \end{note}
  }
}

\longnewglossaryentry{cccexamples}
{
  name={Examples of Cartesian Closed Categories:}
} {
  \begin{itemize}[label={}]
    \item
  \begin{itemize}[align=left]
    \item [\textbf{Set}]
      For any objects $A$ and $B$, the exponential object $B^A$ is the set
      of functions from $A$ to $B$, effectively the same as $\Hom(A,B)$.
      $eval:B^A\times A\rightarrow B$ is the application of $f$ to $a$ in
      $(f,a)\in B^A\times A$, ie: $eval(f,a)=f(a)$.

    \item [\textit{Boolean Algebra}]
      In addition to being a distributive category, the exponential object
      $q^p$ is $\neg p\vee q$. $eval:q^p\times p\rightarrow q$ follows from
      the distributive law, $(\neg p\vee q)\wedge p\Rightarrow q$.

    \item [\textit{Heyting Algebra}]
      The meet operation is product, implication is the exponential object
      and the arrow $eval$ is \textit{modus ponens}.

    \item [\textbf{Grf}]
      Let No be the graph with 1 node and no arrows and Ar be the graph with
      2 nodes and 1 arrow between them.  The exponential object between two
      graphs $\mathscr{G}$ and $\mathscr{H}$, $\mathscr{H}^\mathscr{G}$, is
      a graph which has a set of nodes
      $\Hom(\mathscr{G}\times\text{No},\mathscr{H})$ and a set of arrows
      $\Hom(\mathscr{G}\times\text{Ar},\mathscr{H})$. Both sets of nodes and
      arrows are sets of graph homorphisms.

    \item [\textbf{Func($\mathscr{C}$,\textbf{Set})}]
      For any category $\mathscr{C}$,
      \textbf{Func($\mathscr{C}$,\textbf{Set})} is a cartesian closed
      category. For any two functors
      $F,G:\mathscr{C}\rightarrow\mathbf{Set}$, the exponential object $G^F$
      is a functor defined as follows:
      \begin{itemize}[label={$-$}]
        \item For any object $C$, $G^F(C)=(\Hom(C,-)\times F)\rightarrow G$
          is a set of natural transformations.
        \item For any arrow $f:C\rightarrow D$, $G^F(f)$ takes a natural
          transformation $\alpha:\Hom(C,-)\times F\rightarrow G$ to a
          natural tranformation (with component at an object $A$) that is
          $G^F(f)(\alpha):\Hom(D,A)\times F(A)\rightarrow G(A)$, defined as
          $G^F(f)(\alpha)(g,a)=\alpha A(v\circ f,x)$ for an arrow
          $g:D\rightarrow A$ and an object $a\in F(A)$.
      \end{itemize}

    \item [\textbf{Cat}]
      The exponential object of any two categories $\mathscr{C}$ and
      $\mathscr{D}$ is
      $\mathscr{D}^\mathscr{C}=\mathbf{Func}(\mathscr{C},\mathscr{D})$.

    \item[\textbf{$\omega$-CPOs and Scott-continuous Functions}]
    \item[\textbf{Functional Programming Language}]
      The exponential object is the function type and the $eval$ arrow is
      simply the function application function.

    \item[\textbf{Typed Lambda Calculus}]
      Objects are types and an arrow from an object $A$ to and object $B$ is
      an equivalence class of terms of type $B$ with one free variable of
      type $A$. The exponential object $B^A$ is the lambda abstraction with
      a bound variable of type $A$ and a body of body $B$.

  \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{internalhomfunctors}
{
  name=Internal Hom Functors,
  text=internal hom functors
} {
  In any Cartesian Closed Category $\mathscr{C}$, for an object $A$, there
  are two \textbf{internal hom functors}
  $F:\mathscr{C}\rightarrow\mathscr{C}$ and
  $G:\mathscr{C}^{op}\rightarrow\mathscr{C}$ defined as:
  \begin{itemize}
    \item For any object $B$, $F(B)=B^A$ and $G(B)=A^B$

  \end{itemize}

\newglossaryentry{partiallyorderedobject}
{
  name=Partially Ordered Object,
  text=partially ordered object,
  description={$D$ is a \textbf{partially ordered object} in a cartesian
    closed category if for every $\Hom(A,D)$ where $A$ is any object, there
    is a partial order relation such that for any $f:B\rightarrow A$ and
    $g\leq h$ in $\Hom(A,D)$, $g\circ f\leq h\circ f$ in $\Hom(B,D)$.
    \begin{note}
      $D$ is an $\omega$-CPO object if for every $\Hom(A,D)$ for any object
      $A$, $\Hom(A,D)$ is an $\omega$-CPO (ie: $\Hom(A,D)$ is $\omega$-CPO).
    \end{note}
    \begin{note}
      $D$ is \textit{strict} if there is an arrow $\bot:1\rightarrow D$ such
      that for any object $A$ and any $f\in\Hom(A,D)$,
      $\bot\circ\langle\rangle\leq f$.
    \end{note}
    \begin{note}
      If $D$ is a strict $\omega$-CPO and $f:D\rightarrow D$ is an
      $\omega$-continuous arrow, then $fix(f):1\rightarrow D$ is the least
      element with the property $f\circ fix(f)=fix(f)$.
    \end{note}
  }
}

\longnewglossaryentry{equalizer}

    equation or set of equations is true.
  \end{note}

\longnewglossaryentry{cone}
{
  name=Cone,
  text=cone,
} {
  A \textbf{cone} over a diagram
  $D:\mathbb{I}\rightarrow\mathscr{C}$ (also known as the base) with an
  object $U\in\mathscr{C}$ (also known as the vertex) is a family of arrows
  of in $\mathscr{C}$ from indexed by $D(a)$ for each object
  $a\in\mathbb{I}$, where each arrow has the form $p_a:U\rightarrow D(a)$.


  A cone is \textit{commutative} if for any arrow $s:a\rightarrow
  b$ in $\mathbb{I}$, $D(s)\circ p_a=p_b$, ie: the following diagram
  commutes:
  \begin{mytikzcd}
       \& U \ar[ld,"p_a"'] \ar[rd,"p_b"]
    \\ D(a) \ar[rr,"D(s)"']
       \& \& D(b)
  \end{mytikzcd}
  \begin{note}
    A commutative cone over $D$ with vertex $U$ is precisely a natural
    transformation from the constant functor $\Delta_U$ (the functor that
    takes every object to $U$ and morphism to $id_U$) to $D$, where the
    commutativity is given by naturality.
  \end{note}
  \begin{note}
    A cone over a discrete diagram is vacuously commutative.
  \end{note}
  \begin{note}
    A \textit{morphism} between two cones $\alpha:\Delta_U\rightarrow D$ and
    $\alpha':\Delta_{U'}\rightarrow D$ is an arrow $f:U\rightarrow U'$ such
    that for every object $a\in\mathbb{I}$, the following diagram commutes:
    \begin{mytikzcd}
      U \ar[rr,"f"] \ar[rd,"\alpha_a"'] \&\& U' \ar[ld,"\alpha'_a"] \\
      \& D(a)
    \end{mytikzcd}
  \end{note}
}

\longnewglossaryentry{limit}
{
  name=Limit,
  text=limit,
  symbol={$\underset{\longleftarrow}{\lim}$}
} {
  A \textbf{limit} is an universal commutative cone
  $\alpha:\Delta_U\rightarrow D$ such that for every other cone over the
  same diagram $D:\mathbb{I}\rightarrow\mathscr{C}$, there exists an unique
  arrow to it. Thus:
  \begin{itemize}
    \item For every arrow $s:a\rightarrow b\in\mathbb{I}$,
      $D(s)\circ\alpha_a=\alpha_b$.
    \item If $\alpha':\Delta_U'\rightarrow D$ is another commutative cone,
      there exists an unique arrow $f:U'\rightarrow U$ such that for each
      object $a\in\mathbb{I}$, $\alpha_a\circ f=\alpha'_a$.
  \end{itemize}

  \begin{note}
    A limit can also be defined as a terminal object in the category $cone(D)$.
  \end{note}

  \begin{note}
    A limit is also an universal element of the functor
    $cone(-,D):\mathscr{C}^{op}\rightarrow\mathbf{Set}$, where $cone(U,D)$
    is the set of commutative cones with vertex $U$ over the diagram $D$,
    and $cone(f:U'\rightarrow U,D):cone(U,D)\rightarrow cone(U',D)$ defined
    as $cone(f,D)(\alpha_a:\Delta_{Ua}\rightarrow D_a)=\alpha_a\circ f$
    where $a$ is some object in the domain of $D$. A universal element of
    such a functor is precisely a cone with which there is an unique arrow
    to it from any other cone with the same base.
  \end{note}
}

\longnewglossaryentry{pullback}
{
  name=Pullback,
  text=pullback
} {
  For the diagram:
  \begin{mytikzcd}
    B \ar[r,"g"] \& C \& A \ar[l,"f"']
  \end{mytikzcd}
  a \textbf{pullback} is its limit:
  \begin{mytikzcd}
    P \ar[r,"p_1"] \ar[d,"p_2"'] \& A \ar[d,"f"] \\
    B \ar[r,"g"']                \& C
  \end{mytikzcd}
  This is known as the pullback diagram and the composite arrow $f\circ
  p_1=g\circ p_2:P\rightarrow C$ is omitted by convention.

  \begin{note}[Other Names]
    $P$ together with $p_1$ and $p_2$ is a pullback or \textit{fiber
    product} of $f$ and $g$.  $p_2$ is the pullback of $f$ along $g$, $p_1$
    is the pullback of $g$ along $f$. $P$ is sometimes be denoted as
    $A\times_C B$.
  \end{note}

  \begin{note}
    In \textbf{Set}, $P=\{(a,b)|\forall a\in A,\forall b\in B,f(a)=g(b)\}$.
    $p_1$ and $p_2$ are just the coordinate projections of $P$.
  \end{note}
}

\newglossaryentry{coequalizer}
{
  name=Coequalizer,
  text=coequalizer,
  description={is the dual of \textit{equalizer}. For a parallel pair of
    arrows $f,g:A\rightarrow B$, an arrow $h:B\rightarrow C$ is the
    coequalizer of $f$ and $g$ provided $h\circ f=h\circ g$ and for any
    arrow $k:B\rightarrow D$ where $k\circ f=k\circ g$, there is an unique
    arrow $l:C\rightarrow D$ such that $k=l\circ h$.
    \begin{note}
      Coequalizers can be thought of as the least destructive identification
      necessary (by quotienting into equivalence classes) to force an
      equation or set of equations to be true.
    \end{note}
    \begin{note}[Example]
      The \textit{unification} of expressions by finding a common
      substitution is an example of a coequalizer.
    \end{note}
  }
}

\longnewglossaryentry{pushout}
{
  name=Pushout,
  text=pushout
} {
  is the dual of pullback, such that
  \begin{mytikzcd}
    C \ar[r,"f"] \ar[d,"g"'] \& A \ar[d,"q_1"] \\
    B \ar[r,"q_2"']          \& Q
  \end{mytikzcd}
  is a commutative square for any object $R$ and arrows
  $r_1:A\rightarrow R,r_2:B\rightarrow R,r_1\circ f=r_2\circ g$, there is an

\longnewglossaryentry{disjointcoproduct}
{
  name=Disjoint Coproduct,
  text=disjoint coproduct
} {
  A coproduct $A+B$ in a category is \textbf{disjoint} if the injections
  $A\rightarrow A+B$ and $B\rightarrow A+B$ are monic and the pullback of
  the injections along each other is the initial object (since the
  injections are monic, these injections are subobjects and their
  intersection is the initial object). ie:
  \begin{mytikzcd}
    0 \ar[r] \ar[d] \& A \ar[d] \\
    B \ar[r]        \& {A+B}
  \end{mytikzcd}
  is a pullback diagram (this is also a pushout diagram by the definition of
  coproducts).

  \begin{note}
    In \textbf{Set}, this precisely captures the idea of a disjoint union
    between sets $A$ and $B$ where $A\cap B=\emptyset$.
  \end{note}

  \begin{note}
    If $A+B$ is disjoint, the subobjects $A$ and $B$ are known as
    \textit{complemented subobjects} and $A$ is the complement of $B$ in $C$
    and conversely.
  \end{note}
}

\longnewglossaryentry{intersection}
{
  name=Intersection,
  text=intersection
} {
  If $C$ is an object of $\mathscr{C}$ with subobjects $C_1$ and $C_2$, the
  \textbf{intersection} of $C_1$ and $C_2$, if it exists, is the pullback:
  \begin{mytikzcd}
    C_0 \ar[r] \ar[d] \& C_1 \ar[d] \\
    C_2 \ar[r]        \& C
  \end{mytikzcd}

  \begin{note}
    A category has \textit{wide intersections} if every object $C$ and every
    class of subobjects $\{C_i\}$ of $C$, there is an intersection.
  \end{note}

  \begin{note}
    If exists, the intersection is the meet of the poset of subobjects of
    $C$.
  \end{note}
}

\longnewglossaryentry{kernelpair}
{
  name=Kernel Pair,
  text=kernel pair,
} {
  A \textbf{kernel pair} of an arrow $f:X\rightarrow Y$ is a
  parallel pair of arrows $d,e$ from any object $R$ to $X$ that forms a
  limit:
  \begin{mytikzcd}
    R \ar[r,"d"] \ar[d,"e"'] \& X \ar[d,"f"] \\
    X \ar[r,"f"']        \& Y
  \end{mytikzcd}
  This is also the fiber product $X\times_Y X$, or the pullback of $f$
  along itself.

  \begin{note}
    Dually, a \textit{cokernel pair} of $f$ is the pushout of $f$ along
    itself.
  \end{note}
}

\longnewglossaryentry{cartesianmorphism}
{
  name=Cartesian Morphism,
  text=cartesian morphism
} {
  Let $P:\mathscr{E}\rightarrow\mathscr{C}$ be a functor, $f:C\rightarrow
  D$ is an arrow in $\mathscr{C}$ and $P(Y)=D$ for some object
  $Y\in\mathscr{E}$. A morphism/arrow $u:X\rightarrow Y$ in $\mathscr{E}$ is
  \textbf{cartesian} for $f$ and $Y$ if:

  \begin{itemize}
    \item $P(u)=f$.
    \item $\forall v:Z\rightarrow Y$ in $\mathscr{E}$, $\forall
      h:P(Z)\rightarrow C$ in $\mathscr{C}$ where $f\circ h=P(v)$, $\exists
      w:Z\rightarrow X$ in $\mathscr{E}$ such that $u\circ w=v$ and
      $P(w)=h$.
  \end{itemize}

  \begin{note}
    A cartesian arrow is universal for $f$ and $Y$ as for any other arrow
    $v\in\mathscr{E}$, $f\circ h\in\mathscr{C}$ where $target(v)=Y$,
    $P(v)=f\circ h$, there is an unique factorization $w,v$ factors through
    $u$.
  \end{note}

  \begin{note}
    Dually, $u:X\rightarrow Y$ is \textbf{opcatesian/cocartesian} for
    $f:C\rightarrow D$ and $P(X)=C$ if $P(u)=f$ and $\forall v:X\rightarrow
    Z$, $\forall g:D\rightarrow P(Z)$ where $P(v)=g\circ f$,
    there exists an unique arrow $w:Y\rightarrow Z$ such that $P(w)=g$.
  \end{note}
}

\longnewglossaryentry{fibration}
{
  name=Fibration,
  text=fibration
} {
  A functor $P:\mathscr{E}\rightarrow \mathscr{C}$ is a \textbf{fibration}
  if there is a cartesian arrow for every $f:C\rightarrow D$ in
  $\mathscr{C}$ and object $Y\in\mathscr{E}$ such that $P(Y)=D$. Dually, $P$
  is an \textbf{opfibration} if there is an opcartesian arrow for every $f$
  and object $X\in\mathscr{E}$ such that $P(X)=C$.

  \begin{note}
    $P$ is a fibration if and only if
    $P^{op}:\mathscr{E}^{op}\rightarrow\mathscr{C}^{op}$ is an opfibration.
  \end{note}
  \begin{note}
    If $P:\mathscr{E}\rightarrow\mathscr{C}$ is a functor, then
    $\mathscr{E}$ is \textit{fibered over} $\mathscr{C}$. $\mathscr{C}$ is
    also known as the \textit{base category}, $\mathscr{E}$ as the
    \textit{total category}.
  \end{note}
}

\longnewglossaryentry{cleavage}
{
  name=Cleavage,
  text=cleavage
} {
  A \textbf{cleavage} for a fibration $P:\mathscr{E}\rightarrow\mathscr{C}$
  is a function $\gamma$ that takes an arrow $f:C\rightarrow D$ in
  $\mathscr{C}$ and each object $Y\in\mathscr{E}$ where $P(Y)=D$ to an arrow
  $\gamma(f,Y)\in\mathscr{E}$ that is cartesian for $f$ and $Y$. Dually, an
  \textbf{opcleavage} $\kappa$ takes $f$ and each object $X\in\mathscr{E}$
  where $P(X)=C$ to the arrow $\kappa(f,X)\in\mathscr{E}$ that is
  opcartesian for $f$ and $X$.

  \begin{note}
    A cleavage is \textit{normal} if $\gamma(id_D,Y)=id_Y$.
  \end{note}

  \begin{note}
    A cleavage is \textit{splitting} if it is normal and the composite of
    cartesian arrows is a cartesian arrow of the composite, thus
    $\gamma(g,Z)\circ\gamma(f,Y)=\gamma(g\circ f,Z)$ where $g:D\rightarrow
    E$ is in $\mathscr{C}$ and $P(Z)=E$. Note that this is well defined as
    as the domain of $\gamma(g,Z)$ is $D$, and $P(Y)=D$.
  \end{note}

  \begin{note}
    A fibration is \textit{split} if it has a splitting. The same can be
    said for opfibrations and opcleavages that are splitting.
  \end{note}
}

\longnewglossaryentry{fiber}
{
  name=Fiber,
  text=fiber
} {
  For any functor $P:\mathscr{E}\rightarrow\mathscr{C}$, the \textbf{fiber}
  over an object $C\in\mathscr{C}$ is the collection of objects $X$ and
  arrows $f$ where $P(X)=C$ and $P(f)=id_C$. It can be shown that this fiber
  is a subcategory of $\mathscr{E}$.

  \begin{note}
    The fibers of a functor forms an ``indexed category''.  A functor
    $F:\mathscr{C}\rightarrow\mathbf{Cat}$ can be defined for an opfibration
    $P:\mathscr{E}\rightarrow\mathscr{C}$ with opcleavage $\kappa$, where
    $\forall C\in\mathscr{C}, F(C)$ is the fiber over $C$ and for any arrow
    $f:C\rightarrow D$ in $\mathscr{C}$, $F(f)$ is the functor between the
    fiber over $C$ and $D$. Dually, a contravariant functor
    $F:\mathscr{C}^{op}\rightarrow\mathbf{Cat}$ can be defined for a
    fibration $P:\mathscr{E}\rightarrow\mathscr{C}$ with cleavage $\gamma$.
  \end{note}
}


\longnewglossaryentry{shapefunctor}
{
  name=Shape Functor,
  text=shape functor
} {
  Let $\mathscr{A},\mathscr{B}$ be categories,
  $G:\mathscr{A}\rightarrow\mathbf{Cat}$ be a functor.
  $S(G,\mathscr{B}):\mathscr{A}^{op}\rightarrow\mathbf{Cat}$ is a
  \textbf{shape functor} such that:
  \begin{itemize}[align=left]
    \item[On \underline{objects},] $\forall A\in\mathscr{A}$,
      $S(G,\mathscr{B})(A)=\mathbf{Func}(G(A),\mathscr{B})$.

    \item[On \underline{arrows},] for any arrow $f:A\rightarrow A'$ in
      $\mathscr{A}$,
      $S(G,\mathscr{B})(f):\mathbf{Func}(G(A'),\mathscr{B})\rightarrow\mathbf{Func}(G(A),\mathscr{B})$
      is a functor. On objects/functors,
      $S(G,\mathscr{B})(f)(H:G(A')\rightarrow\mathscr{B})=H\circ G(f)$. On
      arrows/natural transformations,
      $S(G,\mathscr{B})(f)(\alpha:H\rightarrow
      H':G(A')\rightarrow\mathscr{B})=\alpha G(f):H\circ G(f)\rightarrow
      H'\circ G(f)$. The component at an object $X\in G(A)$ for
      $S(G,\mathscr{B})(f)(\alpha)$ is the component of $\alpha$ for
      $G(f)(X)$, ie: $\alpha G(f)X:H[G(f)(X)]\rightarrow H'[G(f)(X)]$.
  \end{itemize}

  \begin{note}
    $S(G,\mathscr{B})(A)$ is the category of diagrams (diagram functors) of
    shape $G(A)$ in $\mathscr{B}$, where arrows are diagram homomorphisms
    (natural transformations between these diagram functors).
  \end{note}
}

\longnewglossaryentry{grothendieckconstruction}
{
  name=Grothendieck Construction,
  text=grothendieck construction
} {
  For any category valued functor, $F:\mathscr{C}\rightarrow\mathbf{Cat}$,
  the \textbf{Grothendieck Construction} constructs a split opfibration from
  $\mathbf{G}(\mathscr{C},F)$ to $\mathscr{C}$ which is induced by $F$. The category
  $\mathbf{G}(\mathscr{C},F)$ is defined as:
  \begin{itemize}
    \item Objects are pairs $(x,C)$, where $C$ is an object in $\mathscr{C}$
      and $x$ is an object in $F(C)$.
    \item Arrows are pairs $(u,f):(x,C)\rightarrow(x',C')$ where
      $f:C\rightarrow C'$ is an arrow in $\mathscr{C}$ and
      $u:F(f)(x)\rightarrow x'$ is an arrow in $F(C')$.
    \item Composition of $(u,f):(x,C)\rightarrow(x',C')$ and
      $(v,g):(x',C')\rightarrow(x'',C'')$ is
      $(v,g):(x,C)\rightarrow(x'',C'')$ defined by $(v,g)\circ(u,f)=(v\circ
      F(g)(u),g\circ f)$
  \end{itemize}

  The corresponding opfibration is the second projection functor of the
  pairs and has a splitting
  $\kappa(f,X)=(id_{F(f)(x)},f):(x,C)\rightarrow(F(f)(x),C')$ where
  $f:C\rightarrow C'$ is an arrow in $\mathscr{C}$ and $(x,C)$ is an object
  in $\mathbf{G}(\mathscr{C},F)$.

  \begin{note}
    On set-valued functors $F:\mathscr{C}\rightarrow\mathbf{Set}$, the
    Grothendieck construction constructs a \textit{split discrete
    opfibration} where the fibers are discrete categories (sets can be
    viewed as discrete categories where elements are objects and the only
    arrows are the identity arrows).
  \end{note}

  \begin{note}
    The monoid homomorphism for a semidirect product of $M$ and $T$ is
    $\phi:M\rightarrow\mathbf{End}(T)$, which is also a category valued
    functor. By the Grothendieck construction, the category
    $\mathbf{G}(M,\phi)$ is exactly the semidirect product of $M$ and $T$.
  \end{note}
}

\longnewglossaryentry{grothendieckfunctor}
{
  name=Grothendieck Functor,
  text=grothendieck functor
} {
  is a functor
  $\mathbf{G}:\mathbf{Func}(\mathscr{C},\mathbf{Cat})\rightarrow\mathbf{SO}(\mathscr{C})$
  defined as:
  \begin{itemize}[align=left]
    \item[On \underline{objects},]
      $\mathbf{G}(F:\mathscr{C}\rightarrow\mathbf{Cat})=\mathbf{G}(\mathscr{C},F)$.

    \item[On \underline{arrows},]
      $\mathbf{G}(\alpha:F\rightarrow
      G:\mathscr{C}\rightarrow\mathbf{Cat}):\mathbf{G}(F)\rightarrow\mathbf{G}(G)$
      is a functor. Let $\mathbf{G}\alpha=\mathbf{G}(\alpha)$, on objects,
      $\mathbf{G}\alpha(x,C)=(\alpha C(x),C)$. On arrows,
      $\mathbf{G}\alpha((u,f):(x,C)\rightarrow(x',C'))=(\alpha C'(u),f)$.
      \begin{note}
        $\alpha C:F(C)\rightarrow G(C)$ is a natural transformation
        indexed at $C$ but is also functor because $F(C)$ and $G(C)$ are
        categories.
      \end{note}
  \end{itemize}
}

\longnewglossaryentry{sohomormophism}

  name=Homomorphism of Split Opfibrations,
  text=homomorphism of split opfibrations
} {

  \begin{mytikzcd}
    \mathscr{E} \ar[rr,"\zeta"] \ar[rd,"P"'] \& \&

    \& \mathscr{C}
  \end{mytikzcd}
}

\longnewglossaryentry{wreathproductinducedaction}
{
  name=Action Induced by Wreath Product,
  text=action induced by wreath product
} {
  Let $\mathscr{A}$, $\mathscr{B}$ be small categories,
  $G:\mathscr{A}\rightarrow\mathbf{Cat}$,
  $H:\mathscr{B}\rightarrow\mathbf{Cat}$ be functors. There is an
  \textbf{induced functor} $G\text{ wr }H:\mathscr{A}\text{ wr}^G\text{
  }\mathscr{B}\rightarrow\mathbf{Cat}$ defined by:
  \begin{itemize}
    \item For an object $(A,P)$ in $\mathscr{A}\text{ wr}^G\text{
      }\mathscr{B}$, $(G\text{ wr }H)(A,P)$ is the split opfibration induced
      by $H\circ P:G(A)\rightarrow\mathbf{Cat}$. Thus, objects in this
      category are $(t,x)$ where $x$ is an object in $G(A)$ and $t$ is an
      object in $(H\circ P)(x)$. Arrows are $(u,f):(t,x)\rightarrow(t',x')$
      where $f:x\rightarrow x'$ is an arrow in $G(A)$ and $u:(H\circ
      P)(f)(x)\rightarrow x'$ is an arrow in $G(A')$.

    \item For an arrow $(h,\lambda):(G\text{ wr }H)(A,P)\rightarrow(G\text{
      wr }H)(A',P')$ in $\mathscr{A}\text{ wr}^G\text{
      }\mathscr{B}$ is a functor where $h:A\rightarrow A'$ is an arrow in
      $\mathscr{A}$ and $\lambda:P\rightarrow(P'\circ G(h))$ ($\lambda$ is a
      natural transformation). For objects $(t,x)$ in $(G\text{ wr
      }H)(A,P)$, $(G\text{ wr }H)(h,\lambda)(t,x)=(H(\lambda x)(t),G(h)(x))$
      where $\lambda x$ is a the natural transformation $\lambda$ indexed by
      $x$ in $G(A)$, $\lambda x:P(x)\rightarrow P'[G(h)(x)]$ and $H(\lambda
      x):H[P(x)]\rightarrow H(P'[G(h)(x)])$ is a natural transformation
      $H\lambda$ indexed at $x$. For arrows $(u,f):(t,x)\rightarrow(t',x')$
      in $(G\text{ wr }H)(A,P)$, $(G\text{ wr }H)(h,\lambda)(h,f)=(H(\lambda
      x')(u),G(h)(f))$.

  \end{itemize}
}

\longnewglossaryentry{adjunction}
{
  name=Adjunction,
  text=adjunction,
  symbol={$\dashv$}
} {
  Let $GF$ denote $G\circ F$ and $GFA$ denote $(G\circ F)(A)$.  Between
  functors $F:\mathscr{A}\rightarrow\mathscr{B}$ and
  $G:\mathscr{B}\rightarrow\mathscr{A}$, $F$ and $G$ form an
  \textbf{adjunction} (denoted as $F\dashv G$) provided that there is a
  natural transformation $\eta:id\rightarrow GF$ such that for any objects
  $A\in\mathscr{A}$, $B\in\mathscr{B}$ and arrow $f:A\rightarrow GB$, there
  is an unique arrow $g:FA\rightarrow B$ such that the following diagram
  commutes:
  \begin{mytikzcd}
    A \ar[r,"\eta A"] \ar[rd,"f"'] \& GFA
    \ar[d,"Gg"] \\ \& GB
  \end{mytikzcd}

  \begin{note}
    Dually, for any adjunction $F\dashv G$, there is a natural
    transformation $\epsilon:FG\rightarrow id$ such that for any
    $g:FA\rightarrow B$, there exists an unique arrow $f:A\rightarrow GB$
    such that the following diagram commutes:
    \begin{mytikzcd}
      \& FGB \ar[d,"\epsilon B"] \\
      FA \ar[ru,"Ff"] \ar[r,"g"'] \& B
    \end{mytikzcd}
  \end{note}

  \begin{note}
    If $F\dashv G$, then $F$ is \textit{left adjoint} to $G$ and $G$ is
    \textit{right adjoint} to $F$. $G$ preserves limits and $F$ preserves
    colimits.
  \end{note}

  \begin{note}
    $\eta$ and $\epsilon$ are called the \textit{unit} and \textit{counit}
    of the adjunction.
  \end{note}

  \begin{note}[Theorem]
    Let $F:\mathscr{A}\rightarrow\mathscr{B}$ and
    $G:\mathscr{B}\rightarrow\mathscr{A}$ be functors, then $F\dashv G$ if
    and only if $\Hom(F-,-)$ and $\Hom(-,U-)$ are naturally isomorphic as
    functors of $\mathscr{A}^{op}\times\mathscr{B}\rightarrow\mathbf{Set}$.
  \end{note}

  \begin{note}
    Adjoint functors are unique up to natural isomorphisms.
  \end{note}
}

  }

\longnewglossaryentry{algebraforanendofunctor}

\longnewglossaryentry{recursivecategory}
{
  name=Recursive Category,
  text=recursive category
}
{
  is a category $\mathscr{C}$ if for every object $A$, the
  underlying functor $U_A:act(A)\rightarrow\mathscr{C}$ has a left adjoint
  $F_A:\mathscr{C}\rightarrow act(A)$.

  \begin{note}
    $U_A:act(A)\rightarrow\mathscr{C}$ is defined as $U_A(x:A\times
    X\rightarrow X)=X$ and $U_A(f_A:(x:A\times X\rightarrow X)\rightarrow

  \end{note}

  \begin{note}
    For an object $B$ and arrow $f:B\rightarrow C$, let
    $rec(A,B)=U_A(F_A(B))$ and $rec(A,f)=U_A(F_A(f)):rec(A,B))\rightarrow
    rec(A,C)$. $F_A$ is defined as $F_A(B)=A\times rec(A,B)\rightarrow
    rec(A,B)$.
  \end{note}

  \begin{note}
    If for all $A$ and $B$, $\langle
    rec(A,\langle\rangle),a\rangle:rec(A,B)\rightarrow rec(A,1)\times B$ is
    an ismorphism, then $\mathscr{C}$ has \textit{local recursion}. A
    recursive category with local recursion for which every slice category
    $\mathscr{C}/C$ is recursive is known as a \textit{locally recursive
    category}.
  \end{note}

  \begin{note}
    A locally recursive category is a \textit{locos} if it is coherent.
  \end{note}
}

\longnewglossaryentry{monad}
{
  name=Monad,
  text=monad
} {
  is a triple $(M,\eta,\mu)$ in a category $\mathscr{C}$, where
  $M:\mathscr{C}\rightarrow\mathscr{C}$ is an endofunctor,
  $\eta:id\rightarrow M$ and $\mu:M^2\rightarrow M$ are natural
  transformations such that the following diagrams commute:
  \begin{mytikzcd}
    M \ar[r,"\eta M"] \ar[rd,"id"'] \& M^2 \ar[d,"\mu"]
    \& M \ar[l,"M\eta"'] \ar[ld,"id"] \\
    \& M
  \end{mytikzcd}

  \begin{mytikzcd}
    M^3 \ar[r,"M\mu"] \ar[d,"\mu M"'] \& M^2 \ar[d,"\mu"] \\
    M^2 \ar[r,"\mu"'] \& M
  \end{mytikzcd}
  $\eta$ is called \textit{unit} and $\mu$ is \textit{multiplication}, the
  diagrams depict unitary identities and associativity of multiplication.

  \begin{note}
    Dually, a \textit{comonad} $(M',\epsilon,\delta)$ in a
    category $\mathscr{C}$ is a monad in $\mathscr{C}^{op}$. Thus, $M'$ is
    an endofunctor in $\mathscr{C}$ and $\epsilon:M'\rightarrow id$,
    $\delta:M'\rightarrow M'^2$ are natural transformations where the
    corresponding dual diagrams commute.
  \end{note}

  \begin{note}
    Given functors $F:\mathscr{A}\rightarrow\mathscr{B}$ and
    $G:\mathscr{B}\rightarrow\mathscr{A}$ where $F\dashv G$,
    $(GF,\eta,G\epsilon F)$ is a monad in $\mathscr{A}$ where $\eta$ and
    $\epsilon$ are the unit and counit of the adjunction. Dually,
    $(FG,\epsilon,F\eta G)$ is a comonad.
  \end{note}
}

\longnewglossaryentry{eilenbergmoorecategory}
{
  name=Eilenberg-Moore Category,
  text=eilenberg-moore category
} {
  is the category of $M$-Algebras denoted as $\mathscr{C}^M$ for a monad
  $M:\mathscr{C}\rightarrow\mathscr{C}$. Thus, objects are $(A,v)$
  where $A$ is an object in $\mathscr{C}$ and $v:T(A)\rightarrow A$. Arrows
  are homomorphims between such $M$-algebras such that the following
  diagrams commute:

  \begin{multicols}{2}
    \begin{mytikzcd}
      {T^2(A)} \ar[r,"\mu A"] \ar[d,"T(v)"'] \& T(A) \ar[d,"v"] \\
      T(A) \ar[r,"v"'] \& A
    \end{mytikzcd}

    \begin{mytikzcd}
      {A} \ar[r,"\eta A"] \ar[rd,"id"'] \& T(A) \ar[d,"v"] \\
      \& A
    \end{mytikzcd}
  \end{multicols}

  \begin{note}
    It is apparent that there is a forgetful functor
    $U:\mathscr{C}^M\rightarrow\mathscr{C}$ and also a free functor
    $F:\mathscr{C}\rightarrow\mathscr{C}^M$ that is left
    adjoint to $U$. It can be shown that the monad induced by $U\circ F$ is

  \end{note}

  \begin{note}
    The Kleisli category over $M$ is a full subcategory of $\mathscr{C}^M$.
  \end{note}

\longnewglossaryentry{subobjectfunctor}
{
  name=Subobject Functor,
  text=subobject functor
} {
  is a functor $Sub:\mathscr{C}^{op}\rightarrow\mathbf{Set}$ defined as:
  \begin{itemize}[label={}]
    \item On objects $C$ in $\mathscr{C}$, $Sub(C)$ is the set of
      subobjects of $C$.

    \item On arrows $k:C'\rightarrow C$, $Sub(k):Sub(C)\rightarrow Sub(C')$
      is a set function that takes a subobject $f_0:C_0\rightarrowtail C$ to
      $f_0'$ in the
      following pullback diagram:
      \begin{mytikzcd}
        C_0' \ar[r,"k_0"] \ar[d,rightarrowtail,"f_0'"'] \& C_0
        \ar[d,rightarrowtail,"f_0"] \\
        C' \ar[r,"k"'] \& C
      \end{mytikzcd}
      \begin{note}
        $f_0'$ is a monic due to being a pullback of a monomorphism.
      \end{note}

  \end{itemize}
}

\longnewglossaryentry{subobjectclassifier}
{
  name=Subobject Classifier,
  text=subobject classifier
} {
  is the object representing a subobject functor, usually denoted $\Omega$
  where an universal element is a subobject
  $true:\Omega_0\rightarrowtail\Omega$ and for any other subobject
  $A_0\subseteq A$, there is an unique arrow $\chi:A\rightarrow\Omega$ such
  that the following is a pullback diagram:
  \begin{mytikzcd}
    A_0 \ar[r] \ar[d,rightarrowtail] \& \Omega_0
    \ar[d,rightarrowtail,"true"] \\
    A \ar[r,"\chi"'] \& \Omega
  \end{mytikzcd}
  It can be proven that $\Omega_0$ is a terminal object and thus the arrow
  from $A_0$ to $\Omega_0$ is the unique arrow by the definition of a
  terminal object.

  \begin{note}
    The arrow $\chi$ is called the \textit{characteristic arrow} of the
    subobject. Since the subobject functor $Sub$ is represented by
    $\Omega$, there is a natural isomorphism from $Sub(-)$ to
    $\Hom(-,\Omega)$, which takes a subobject to its characteristic
    function.
  \end{note}
}

\newglossaryentry{topos}
{
  name=Topos,
  text=topos,
  description={is a cartesian closed category which has finite limits and
    has a representable subobject functor.
    \begin{note}
      $\mathbf{Set}$ is a topos where the subobject classifier is a
      two-element set, $\Omega=\{True,False\}$. For any set $S_0\subseteq
      S$, the characteristic function $\chi:S\rightarrow\Omega$ is defined
      as:
      \begin{equation*}
        \chi(x)=
        \begin{cases}
          True & \text{if } x\in S_0 \\
          False & \text{if } x\notin S_0
        \end{cases}
      \end{equation*}
    \end{note}
    \begin{note}
      The exponential object $\Omega^A$ has the property that:
      \begin{equation*}
        \Hom(B,\Omega^A)\cong\Hom(A\times B,\Omega)\cong Sub(A\times B)
      \end{equation*}
      In topos theory, $\Omega^A$ is often called the \textit{power object}
      of $A$ and is denoted $\mathscr{P}A$
    \end{note}
    \begin{note}
      A topos $\mathscr{E}$ has finite colimits, disjoint sums, universal
      sums, every epimorphism is regular and $\mathscr{E}$ is a regular

    \end{note}
  }
}

\longnewglossaryentry{equivalencerelation}
{
  name=Equivalence Relation,
  text=equivalence relation
} {
  For $d,e:E\rightarrow A$ a parallel pair of arrows, for an object $B$, the
  function
  $\langle\Hom(B,d),\Hom(B,e)\rangle:\Hom(B,E)\rightarrow\Hom(B,A)\times\Hom(B,A)$
  sends a function $f:B\rightarrow E$ to $(d\circ f,e\circ f)$. If for each
  object $B$, $\langle\Hom(B,d),\Hom(B,e)\rangle$ is an isomorphism and
  there is an equivalence relation on the set $\Hom(B,A)$, then $E$ is an
  \textbf{equivalence relation} on $A$ (note that the image of
  $\langle\Hom(B,d),\Hom(B,e)\rangle$ is a subset of
  $\Hom(B,A)\times\Hom(B,A)$ and is an equivalence relation).

  \begin{note}
    An equivalence relation of $E$ on $A$ is \textit{effective} if it is the
    kernel pair of some arrow from $A$.
  \end{note}

  \begin{note}
    A topos has effective equivalence relations as every equivalence
    relation is effective.
  \end{note}
}

\newglossaryentry{partialmorphism}
{
  name=Partial Morphism,
  text=partial morphism,
  description={A \textbf{partial morphism} from objects $A$ to $B$ is a
    subobject $A_0\subseteq A$ and an arrow from $f:A_0\rightarrow B$. If
    $g:A_1\rightarrow B$ is another partial morphism, $f\leq g$ if
    $A_0\subseteq A_1$ and $f$ is the restriction to $A_0$ of $g$, ie: $f=
    g\circ i$ where $i:A_0\rightarrow A_1$ is the inclusion.
    \begin{note}
      If both $f\leq g$ and $g\leq f$, then $f$ and $g$ have the same
      subobject as their domain.
    \end{note}
    \begin{note}
      Partial arrows from $A$ to $B$ are \textit{representable} if
      $\widetilde{B}$ is an object where $B\rightarrowtail\widetilde{B}$ and
      $\Hom(A,\widetilde{B})$ is isomorphic to the partial arrows from $A$
      to $B$. This one to one correspondence is given by the pullback:
      \begin{mytikzcd}
        A_0 \ar[r,rightarrowtail] \ar[d,"f"'] \& A \ar[d,"g"] \\
        B \ar[r,rightarrowtail] \& \widetilde{B}
      \end{mytikzcd}
      where there is a one to one correspondence between
      $f\in\Hom(A,\widetilde{B})$ and $g$, which is an arrow from $A$ to
      $B$.
    \end{note}
  }
}

\newglossaryentry{presheaf}
{
  name=Presheaf,
  text=presheaf,
  description={is a contravariant set-valued functor
    $E:\mathscr{C}^{op}\rightarrow\mathbf{Set}$ on a category $\mathscr{C}$.
    \begin{note}
      A presheaf $E:\mathscr{C}^{op}\rightarrow\mathbf{Set}$ is
      \textit{constant} if $E$ is a constant functor. Thus, for any object
      $x$ in $\mathscr{C}^{op}$, $E(x)$ is the same set and for any arrow
      $f$ in $\mathscr{C}^{op}$, $E(f)=id_{E(x)}$.
    \end{note}
  }
}

\longnewglossaryentry{sheaf}
{
  name=Sheaf,
  text=sheaf
} {
  is a presheaf $E$ over a complete Heyting algebra such that:
  \begin{equation*}
    E(\bigvee x_i)\xrightarrow{e}
    \underset{i}{\prod}E(x_i)
    \parrightarrow[c][d]
    \underset{ij}{\prod}E(x_i\wedge x_j)
  \end{equation*}
  the arrow $e$ is an equalizer.

  \begin{note}
    Let $x=\bigvee x_i$ for all $i$.
    $e:E(x)\xrightarrow{e}\underset{i}{\prod}E(x_i)$ is induced by universal
    property of the categorical product where since forall $i,E(x\geq
    x_i):E(x)\rightarrow E(x_i)$, there exists $\underset{i}{\prod}E(x_i)$
    such that $E(x\geq x_i)=proj_i\circ e$. The arrows $c$ and $d$ are
    induced by the arrows $E(x_i\geq x_i\wedge x_j):E(x_i)\rightarrow
    E(x_i\wedge x_j)$, $E(x_j\geq x_i\wedge x_j):E(x_j)\rightarrow
    E(x_i\wedge x_j)$, $proj_i:\underset{i}{\prod}E(x_i)\rightarrow E(x_i)$,
    $proj_j:\underset{i}{\prod}E(x_i)\rightarrow E(x_i)$ and thus there
    exists $\underset{ij}{\prod}E(x_i\wedge x_j)$ such that $proj_{ij}\circ
    c=E(x_i\geq x_i\wedge x_j)\circ proj_i$ and $proj_{ij}\circ d=E(x_i\geq
    x_j\wedge x_j)\circ proj_j$.
  \end{note}

  \begin{note}
    The category of sheaves on a Heyting algebra is a topos.
  \end{note}
}

\longnewglossaryentry{monoidalcategory}
{
  name=Monoidal Category,
  text=monoidal category
} {
  is a category $\mathscr{C}$ equipped with an object $\top$ and a functor
  $\otimes:\mathscr{C}\times\mathscr{C}\rightarrow\mathscr{C}$ with natural
  isomorphisms:
  \begin{itemize}
    \item $a(A,B,C):A\otimes(B\otimes C)\rightarrow(A\otimes B)\otimes C$
    \item $lA:\top\otimes A\rightarrow A$
    \item $rA:A\otimes\top\rightarrow A$
  \end{itemize}

  such that the following diagrams commute:
  \begin{mytikzcd}
    {A\otimes(\top\otimes B)} \ar[rr,"{a(A,\top,B)}"]
                              \ar[rd,"A\otimes l(B)"']
      \& \& {(A\otimes\top)\otimes B} \ar[ld,"r(A)\otimes B"] \\
    \& {A\otimes B}
  \end{mytikzcd}

  \begin{mytikzcd}
    \& {A\otimes(B\otimes(C\otimes D))}
      \ar[ld,"{A\otimes a(B,C,D)}"']
      \ar[rd,"{a(A,B,C\otimes D)}"] \\
      {A\otimes((B\otimes C)\otimes D)} \ar[d,"{a(A,B\otimes C,D)}"']
      \& \&
      {(A\otimes B)\otimes(C\otimes D)} \ar[d,"{a(A\otimes B,C,D)}"] \\
      {(A\otimes(B\otimes C))\otimes D} \ar[rr,"{a(A,B,C)\otimes D}"']
      \& \& {((A\otimes B)\otimes C)\otimes D}
  \end{mytikzcd}
  illustrating the unit $\top$ with respect to $\otimes$ and the
  associativity of $\otimes$.

  \begin{note}
    A monoidal category is \textit{symmetric} if it has an additional
    natural isomorphism $s(A,B):A\otimes B\rightarrow B\otimes A$ such that
    the following diagrams commute:
    \begin{mytikzcd}
      {A\otimes B} \ar[rr,"{s(A,B)}"] \ar[rd,"id"']
      \& \& {B\otimes A} \ar[ld,"{s(B,A)}"] \\
      \& {A\otimes B}
    \end{mytikzcd}

    \begin{mytikzcd}
      {A\otimes\top} \ar[rr,"{s(A,\top)}"] \ar[rd,"r(A)"']
      \& \& {\top\otimes A} \ar[ld,"l(A)"] \\
      \& {A}
    \end{mytikzcd}

    \begin{mytikzcd}
    {A\otimes(B\otimes C)} \ar[rr,"{a(A,B,C)}"] \ar[d,"{A\otimes s(B,C)}"']
      \& \& {(A\otimes B)\otimes C} \ar[d,"{s(A\otimes B,C)}"] \\
      {A\otimes(C\otimes B)} \ar[d,"{a(A,C,B)}"']
      \& \& {C\otimes(A\otimes B)} \ar[d,"{a(C,A,B)}"] \\
      {(A\otimes C)\otimes B} \ar[rr,"{s(A,C)\otimes B}"']
      \& \& {(C\otimes A)\otimes B}
    \end{mytikzcd}
  \end{note}

  \begin{note}
    $\otimes$ is often known as the tensor product and
    $A\otimes -:\mathscr{C}\rightarrow\mathscr{C}$ forms an endofunctor.
  \end{note}

  \begin{note}
    A monoidal category $\mathscr{C}$ is \textit{closed} if for each object
    $A$, the functor $A\otimes -$ has a right adjoint
    $A\multimap-:\mathscr{C}\rightarrow\mathscr{C}$ where for any object $C$ in
    $\mathscr{C}$, $A\multimap C$ is the internal hom from $A$ to $C$.


    Thus by the definition of adjoints,
    \begin{equation*}
      \Hom(A\otimes B,C)\xrightarrow{\cong}\Hom(B,A\multimap C)
    \end{equation*}
    is a natural isomorphism that is natural in $B$ and $C$.  It follows
    that in a symmetric monoidal closed category,
    \begin{equation*}
      \Hom(A\otimes B,C)\xrightarrow{\cong}\Hom(A,B\multimap C)
    \end{equation*}
    is also a natural isomorphism and
    \begin{equation*}
      (A\otimes B)\multimap C\cong A\multimap(B\multimap C)
    \end{equation*}

    The counit of this adjunction is the evaluation map of the internal hom
    object, $e:A\otimes(A\multimap B)\rightarrow B$.
  \end{note}
}

\longnewglossaryentry{starautonomouscategory}
{
  name={$\ast$-Autonomous Category},
  text={$\ast$-autonomous category}
} {
  is a symmetric monoidal closed category $\mathscr{C}$ with a duality
  functor $(-)^*:\mathscr{C}^{op}\rightarrow\mathscr{C}$ such that there is
  an natural isomorphism $\Hom(A\otimes B,C^*)\cong\Hom(A,(B\otimes A)^*)$.

  \begin{note}
    $(-)^*$ is related to the functors $\otimes$ and $\multimap$ by the
    isomorphisms $A\otimes B\cong(A\multimap B^*)^*$ and $A\multimap
    B\cong(A\otimes B^*)^*$
  \end{note}

  \begin{note}
    There is a second monoidal structure in a $\ast$-autonomous category with
    $\oplus:\mathscr{C}\times\mathscr{C}\rightarrow\mathscr{C}$ and
    $\bot=\top^*$ as unit where $A\oplus B=(A^*\otimes B^*)^*$.
  \end{note}

  \begin{note}
    There are isomorphisms:
    \begin{equation*}
      \Hom(A\otimes B, C)\cong\Hom(B,A\multimap C)\cong\Hom(B,(A\otimes
      C^*)^*)\cong\Hom(B,A^*\oplus C)
    \end{equation*}
    \begin{equation*}
      A^*\cong\top\multimap A^*\cong A\multimap\top^*\cong
      A\multimap\bot
    \end{equation*}

    By the second isomorphism, the duality functor is completely determined
    by $\multimap$ and $\bot$ and thus $\bot$ is also known as the
    \textit{dualizing object}.
  \end{note}
}

\newglossaryentry{retract}
{
  name=Retract,
  text=retract,
  description={is an object $A$ for an object $B$ when there are arrows
    $i:A\rightarrow B$ and $r:B\rightarrow A$ such that $r\circ i=id_A$. $r$
    is also called a \textbf{retraction} of $B$ onto $A$ and $i$ is a
    section of $r$.
    \begin{note}
      Retraction is a split epimorphisms.
    \end{note}
  }
}

\newglossaryentry{section}
{
  name=Section,
  text=section,
  description={is an arrow that is also a right-inverse, ie: a split
    monomorphism.
    \begin{note}
      The left inverse of a section is also called a retraction or
      \textit{cosection}.
    \end{note}
  }
}

\newglossaryentry{zeroobject}
{
  name=Zero Object,
  text=zero object,
  description={is an object in a category that is both an initial and a
    terminal object.
    \begin{note}
      A zero object is also as a \textit{null object}.
    \end{note}
    \begin{note}
      A category with a zero object is sometimes called a \textit{pointed
      category}.
    \end{note}
    \begin{note}
      In a category with a zero object $0$, between any two objects $A$ and
      $B$, there exists a canonical morphism given by the composition
      $A\rightarrow 0\rightarrow B$.
    \end{note}
  }
}

\longnewglossaryentry{commacategory}
{
  name=Comma Category,
  text=comma category
} {
  Given functors $f:\mathscr{C}\rightarrow\mathscr{E}$ and
  $g:\mathscr{D}\rightarrow\mathscr{E}$, their \textbf{comma category}
  $(f/g)$ is defined as:
  \begin{itemize}
    \item An object is a triple $(c,d,a)$ where $c,d$ are objects in
      $\mathscr{C,D}$ respectively and $a:f(c)\rightarrow g(d)$ is an arrow
      in $\mathscr{E}$

    \item An arrow is a pair
      $(\beta,\gamma):(c_1,d_1,a_1)\rightarrow(c_2,d_2,a_2)$ where
      $\beta:c_1\rightarrow c_2$, $\gamma:d_1\rightarrow d_2$ are arrows in
      $\mathscr{C},\mathscr{D}$ such that $a_2\circ f(\beta)=g(\gamma)\circ
      a_1$:
      \begin{mytikzcd}
        f(c_1) \ar[r,"f(\beta)"] \ar[d,"a_1"'] \& f(c_2) \ar[d,"a_2"] \\
        g(d_1) \ar[r,"g(\gamma)"'] \& g(d_2)
      \end{mytikzcd}

    \item Composition of arrows
      $(\beta,\gamma):(c_1,d_1,a_1)\rightarrow(c_2,d_2,a_2)$ and
      $(\beta',\gamma'):(c_2,d_2,a_2)\rightarrow(c_3,d_3,a_3)$ is
      $(\beta'\circ\beta,\gamma'\circ\gamma):(c_1,d_1,a_1)\rightarrow(c_3,d_3,a_3)$

  \end{itemize}
}
