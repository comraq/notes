{
  name=Hom-Set,
  text=hom-set,
  description={A \textbf{hom-set} $\Hom(S,T)$ is the set of all morphisms

\newglossaryentry{homfunction}
{
  name=Hom Function,
  text=hom function,
  description={
    Let $S$ be an object and $f:T\rightarrow V$ an arrow in a
    category, a \textit{covariant} \textbf{hom function}
    $\Hom(S,f):\Hom(S,T)\rightarrow\Hom(S,V)$, if defined by:
    \begin{center}
      $\Hom(S,f)(g)=f\circ g$
    \end{center}
    for some $g\in\Hom(S,T)$.\\
%
    Similarily, a \textit{contravariant} \textbf{hom function}
    $\Hom(f,S):\Hom(V,S)\rightarrow\Hom(T,S)$, if defined by:
    \begin{center}
      $\Hom(f,S)(g)=g\circ f$
    \end{center}
    for some $g\in\Hom(V,S)$.
  }
}

\longnewglossaryentry{category}
{
  name=Category,
  text=category
} {
  A \textbf{category} $\mathscr{C}$ contains a collection of
  \textit{objects} (denoted as $\mathscr{C}_0$) and a collection of
  \textit{morphisms} (denoted as $\mathscr{C}_1$) where every morphism has
  $source, target:\mathscr{C}_1\rightarrow\mathscr{C}_0$ functions mapping
  each arrow to its \textit{source} and \textit{target} objects respectively
  into the collection of objects.\\

  For any two arrows $f,g$ where the $target(f)=source(g)$, the
  \textit{composition} of $f$ and $g$ are written as $g\circ
  f:source(f)\rightarrow target(g)\in\mathscr{C}_1$. Composition is
  associative, $(h\circ g)\circ f=h\circ (g\circ f)$ when either side is
  defined.\\

  $\forall A\in\mathscr{C}_0$, there exists the \textit{identity} arrow,
  $id_A:A\rightarrow A$ and for any $f:A\rightarrow B$, $f\circ id_A=id_B\circ
  f=f$.\\

  A category is \textit{small} if its objects and arrows constitute sets;
  otherwise it is \textit{large}.
}

\longnewglossaryentry{subcategory}
{
  name=Subcategory,
  text=subcategory
} {
  A \textbf{subcategory} $\mathscr{D}$ of a category
  $\mathscr{C}$ is a category for which:
  \begin{itemize}
    \item All objects and arrows in $\mathscr{D}$ are objects and arrows in
      $\mathscr{C}$ (ie: $\mathscr{D}_0\subseteq\mathscr{C}_0$ and
      $\mathscr{D}_1\subseteq\mathscr{C}_1$).

    \item The source and target of an arrow in $\mathscr{D}$ are the same as
      its souce and target in $\mathscr{C}$ (ie: source and target maps in
      $\mathscr{D}$ are restrictions of those in $\mathscr{C}$). Thus,
      $\forall A,B\in\mathscr{C}_0$,
      $\Hom_{\mathscr{D}}(A,B)\subseteq\Hom_{\mathscr{C}}(A,B)$.

    \item If $A$ is an object in $\mathscr{C}$, its identity arrow $id_A$ in
      $\mathscr{C}$ is in $\mathscr{D}$.

    \item If $f:A\rightarrow B$ and $g:B\rightarrow C$ in $\mathscr{D}$,
      then the composite $g\circ f\in\mathscr{C}_1$ is also the composite in
      $\mathscr{D}$.
  \end{itemize}
}

\newglossaryentry{productofcategories}
{
  name=Product of Categories,
  text=product of categories,
  description={If $\mathscr{C}$ and $\mathscr{D}$ are categories their
    \textbf{product} $\mathscr{C}\times\mathscr{D}$ is the category whose
    objects are all ordered pairs $(C,D)$ where $C\in\mathscr{C}_0,
    D\in\mathscr{D}_0$ and arrows are $(f,g):(C,D)\rightarrow (C',D')$ where
    $f:C\rightarrow C'\in\mathscr{C}_1, g:D\rightarrow D'\in\mathscr{D}_1$.
    The identity arrow of $(C,D)$ is $(id_C,id_D)$ and composition of arrows is
  defined component wise.}
}

\longnewglossaryentry{categoricaldual}
{
  name=Categorical Dual,
  text=categorical dual,
  symbol={$^{op}$}
} {
  Given any category $\mathscr{C}$, $\mathscr{C}^{op}$ is the
  \textbf{dual category} of $\mathscr{C}$ where all the arrows are reversed.
  $\mathscr{C}^{op}$ is defined as follows:
  \begin{itemize}
    \item The objects and arrows of $\mathscr{C}^{op}$ are the objects and
      arrows of $\mathscr{C}$.

    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $f:B\rightarrow A$ in
      $\mathscr{C}^{op}$.

    \item If $h=g\circ f$ in $\mathscr{C}$, then $h=f\circ g$ in
      $\mathscr{C}^{op}$.
  \end{itemize}
  Since identity arrows have the same source and target objects, identity
  arrows in $\mathscr{C}$ are the same as in $\mathscr{C}^{op}$ and thus
  $\mathscr{C}^{op}$ is also a category.
}

\longnewglossaryentry{slicecategory}
{
  name=Slice Category,
  text=slice category,
  symbol={$/$}
} {
  If $\mathscr{C}$ is a category, for any object $A\in\mathscr{C}_0$, the
  \textbf{slice category} $\mathscr{C}/A$ is defined as follows:
  \begin{itemize}
    \item An object of $\mathscr{C}/A$ is an arrow $f:C\rightarrow A$ in
      $\mathscr{C}$ for some $C\in\mathscr{C}_0$.

    \item An arrow of $\mathscr{C}/A$ from $f:C\rightarrow A$ to
      $f':C'\rightarrow A$ is an arrow $h:C\rightarrow C'$ such that
      $f=f'\circ h$.

    \item The composite of $h:f\circ f'$ and $h':f'\circ f''$ is $h'\circ
      h$.
  \end{itemize}

  \begin{note}
    Since the same $h$ can satisfy both $f=f'\circ h$ and $g=g'\circ h$
    where $f\neq g$ or $f'\neq g'$, $h:f\rightarrow f'$ and $h:g\rightarrow
    g'$ are regarded as different arrows in $\mathscr{C}/A$.
  \end{note}

  \begin{note}
    The \textit{indexed function} of an \textit{indexed set} indexed
    by $S$, is precisely an arrow in the slice category \textbf{Set}$/S$.
  \end{note}
}

  }

\newglossaryentry{monomorphism}
{
  name=Monomorphism,
  text=monomorphism,
  description={is an arrow $f:A\rightarrow B$ such that for any object $T$
    and arrows $x,y:T\rightarrow A$, if $x\neq y$, then $f\circ x\neq f\circ
    y$. Monomorphisms can be denoted by $f:A\rightarrowtail B$.
    \begin{note}
      In this context, $x,y:T\rightarrow A$ may be regarded as
      \textit{variable elements} and can be written as if $x\neq y$, then
      $f(x)\neq f(y)$.
    \end{note}
    \begin{note}[Properties]
      Let $f:A\rightarrow B$ and $g:B\rightarrow C$ be arrows:
      \begin{itemize}
        \item If $f$ and $g$ are monomorphisms, so is $g\circ f$
        \item If $g\circ f$ is a monomorphism, so is $f$
      \end{itemize}
    \end{note}
    \begin{note}
      In \textbf{Set}, a function is injective if and only if it is a
      monomorphism.
    \end{note}
    \begin{note}
      In the category determined by a Monoid, a monic element is left
      cancellable.
    \end{note}
  },
  symbol={$\rightarrowtail$}
}

\newglossaryentry{epimorphism}
{
  name=Epimorphism,
  text=epimorphism,
  description={is the dual to monomorphisms. An arrow $f:S\rightarrow T$ is
    an \textbf{epimorphism} if for any arrows $g,h:T\rightarrow X$, $g\circ
    f=h\circ f$ implies $g=h$. Epimorphisms can be denoted by
    $f:S\twoheadrightarrow T$.
    \begin{note}[Properties]
      Let $f:A\rightarrow B$ and $g:B\rightarrow C$ be arrows:
      \begin{itemize}
        \item If $f$ and $g$ are epimorphisms, so is $g\circ f$
        \item If $g\circ f$ is an epimorphism, so is $g$
      \end{itemize}
    \end{note}
    \begin{note}
      In \textbf{Set}, a function is surjective if and only if it is an
      epimorphism.
    \end{note}
    \begin{note}
      In the category determined by a Monoid, an epic element is right
      cancellable.
    \end{note}
  },
  symbol={$\twoheadrightarrow$}
}

\longnewglossaryentry{subobject}
{
  name=Subobject,
  text=subobject
} {
  In a category $\mathscr{C}$, a \textbf{subobject} of an object
  $C\in\mathscr{C}$ is an equivalence class of monomorphisms under the
  relation $\sim$. The subobject is a \textbf{proper subobject} if it does not
  contain $id_C$. The equivalence relation $\sim$ is defined as follows:
  \begin{center}
    For monomorphisms $f_0:C_0\rightarrowtail C$ and $f_1:C_1\rightarrowtail
    C$, $f_0\sim f_1$ if both $f_0$ and $f_1$ factors through each other.
  \end{center}
}

\longnewglossaryentry{functor}
{
  name=Functor,
  text=functor
} {
  is a ``structure preserving'' map between categories, similar to
  homomorphisms between groups, graphs and etc$\dots$ A \textbf{functor}
  $F:\mathscr{C}\rightarrow\mathscr{D}$ from the category $\mathscr{C}$ to
  the category $\mathscr{D}$ defined as follows:
  \begin{itemize}
    \item $\forall A\in\mathscr{C}$, $F(A)\in\mathscr{D}$
    \item If $f:A\rightarrow B$ in $\mathscr{C}$, then $F(f):F(A)\rightarrow
      F(B)$ is an arrow in $\mathscr{D}$ such that:
      \begin{itemize}
        \item $F(id_A)=id_{F(A)}$ for every object $A\in\mathscr{C}$
        \item $F(g\circ f)=F(g)\circ F(f)$ for all arrows $f:X\rightarrow
          Y, g:Y\rightarrow Z$ in $\mathscr{C}$
      \end{itemize}
  \end{itemize}

  \begin{note}
    All functors preserve isomorphisms.
  \end{note}
}

\longnewglossaryentry{categoryexamples}
{
  name={Examples of Categories:}
} {
  \begin{itemize}[label={}]
    \item
  \begin{itemize}[align=left]
    \item[\textbf{Set}]
      is the category whose objects are sets and arrows are functions
      between sets. Composition is function composition and the identity
      arrow is the identity function $id_S$ for a set $S$.

    \item[\textbf{Fin}]
      is the category whose objects are finite sets and arrows are all the
      functions between finite sets.

    \item[\textbf{Pfn}]
      is the category whose objects are sets but all arrows are all partial
      functions. If $f:S\rightarrow T, g:T\rightarrow V$ are partial
      functions with $f$ defined on $S_0\subseteq S$ and $g$ defined on
      $T_0\subseteq T$, then the composite $g\circ f:S\rightarrow V$ is the
      partial function from $S$ to $V$ defined on the subset $\{x\in
      S_0|f(x)\in T_0\}$ of $S$ by the requirement $(g\circ f)(x)=g(f(x))$.

    \item[\textbf{Rel}]
      is the category whose objects are sets and arrows are relations
      relations between sets. The composite $\beta\circ\alpha$ of relations
      $\alpha$ from sets $S$ to $T$ and $\beta$ from $T$ to $U$ is a
      relation from $S$ to $U$ defined as follows:
      \begin{center}
        If $x\in S$ and $z\in U$, $(x,z)\in\beta\circ\alpha$ if and only if
        there is an element $y\in T$ for which $(x,y)\in\alpha$ and
        $(y,z)\in\beta$.
      \end{center}
      The identity arrow for a set $S$ is the \textit{diagonal relation}
      $\Delta_S=\{(x,x)|x\in S\}$

    \item [\textbf{Poset}]
      is the category whose objects are elements of a poset $P$ and arrows
      are the partially ordered relation between those elements. The
      reflexive and transitive properties provides the identity and
      composition of arrows.
      \begin{note}
        Every hom-set in this category has at most ONE element.
        Equivalently, any category where every hom-set has at most one element
        is a poset.
      \end{note}

    \item [\textbf{Monoid}]
      is the category  whose only object is the Monoid $M$ itself and arrows
      are the elements of the monoid (this category is denoted as $C(M)$).
      The identity arrow is the identity element and composition being the
      monoid binary operation.
      \begin{note}
        Any category with one object is a monoid.
      \end{note}

    \item [\textbf{Group}]
      is the category whose only object is the Group itself and arrows are
      elements of the group, same as the category determined by a Monoid.
      \begin{note}
        All arrows in this category are isomorphisms.
      \end{note}

    \item [\textbf{Grf}]
      is the category whose objects are graphs and arrows are graph
      homomorphisms. Identity arrow being the identity homormophism and
      composition being composition of homomorphisms.

    \item [\textbf{Posets and Monotone Functions}]
      is the category whose objects are posets and arrows are monotone
      functions between posets.

    \item [\textbf{Sem}]
      is the category whose objects are semigroups and arrows are semigroup
      homomorphisms.
      \begin{note}
        This can be extended to monoids, denoted as \textbf{Mon}.
      \end{note}

    \item[\textbf{$\omega$-CPOs and Scott-continuous Functions}]
      is the category whose objects are $\omega$-CPOs and arrows are
      Scott-continuous functions.
      \begin{note}
        A category is also formed with \textit{strict} $\omega$-CPOs and
        \textit{strict} Scott-continous functions.
      \end{note}

    \item[\textbf{Path Category}]
      is the free category $F(\mathscr{G})$ generated by a graph
      $\mathscr{G}$ whose objects are the nodes of $\mathscr{G}$ and arrows
      are paths in $\mathscr{G}$. For each node/object $A$, the identity
      arrow is the empty path from $A$ to $A$. Composition is defined by
      $(f_1,\dots,f_k)\circ(f_{k+1},\dots,f_n)=(f_1,\dots,f_n)$

    \item[\textbf{Cat}]
      is the category whose objects are small categories and arrows are
      functors between small categories.

    \item[$M$-\textbf{Act}]
      is the category whose objects are monoid actions and arrows are
      equivariant maps. Since equivariant maps are functions, the identity
      arrow is the identity equivariant map and composition is function
      composition.

  \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{functorexamples}
{
  name={Examples of Functors:}
} {
  \begin{itemize}[label={}]
    \item
  \begin{itemize}[align=left]
    \item[\textbf{Monoid Homomorphism}]
      $f:M\rightarrow N$ where $M$ and $N$ are monoids is a functor from the
      category $C(M)$ to the category $C(N)$.

    \item[\textbf{Monotone Map}]
      between two posets is a functor between the category determined by the
      posets.

    \item[\textbf{Projection}]
      The first and second projections from the product category to its
      first or second parts,
      $P_1:\mathscr{C}\times\mathscr{D}\rightarrow\mathscr{C}$ and
      $P_2:\mathscr{C}\times\mathscr{D}\rightarrow\mathscr{D}$ for
      categories $\mathscr{C}$ and $\mathscr{D}$ are functors.

    \item[\textbf{Inclusion Map}]
      of a subcategory is a functor.
      \begin{note}
        By the definition of \textit{subcategory}, objects and arrows of the
        subcategory need not be objects and arrows in the bigger category.
      \end{note}
      \begin{note}[Example]
        \textbf{Set} is a subcategory of \textbf{Rel} and the functor is
        defined by taking sets to the sets as objects in \textbf{Rel} and
        set functions $f:S\rightarrow T$ to $\{(s,t)|t=f(s)\}$.
      \end{note}

    \item[\textbf{Monoid Actions}]
      For an action $\alpha$ of a monoid $M$ acting on a set $S$, let $C(M)$
      be the category determined by $M$ and the action $\alpha$ determines a
      functor $F_\alpha:C(M)\rightarrow\mathbf{Set}$ defined as:
      \begin{itemize}[nosep, label={-}]
        \item $F_\alpha(*)=S$ where $*$ denotes the single object in $C(M)$
        \item $F_\alpha(m)=s\mapsto\alpha(m,s)$ for $m\in M$, $s\in S$
      \end{itemize}

  \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{forgetfulfunctor}
{
  name=Forgetful/Underlying Functor,
  text=forgetful functor
} {
  is the functor which ``forgets'' some of the structure in a
  category of structures and structure-preserving functions.
  \begin{itemize}[label={}, align=left, nolistsep]
    \item[\textit{Examples}:]
    \item
    \begin{itemize}
      \item $U:\mathbf{Mon}\rightarrow\mathbf{Sem}$

      \item $U:\mathbf{Sem}\rightarrow\mathbf{Set}$

      \item $U:\mathbf{Grf}\rightarrow\mathbf{Set}\times\mathbf{Set}$\\
        that takes a graph $\mathscr{G}$ to the set of nodes and set of
        arrows, $U(\mathscr{G})=(\mathscr{G}_0,\mathscr{G}_1)$. It follows
        that there is also an arrowset functor
        $A:\mathbf{Grf}\rightarrow\mathbf{Set}$ and nodeset functor
        $N:\mathbf{Grf}\rightarrow\mathbf{Set}$.

      \item $U:\mathbf{Cat}\rightarrow\mathbf{Grf}$\\
        by forgetting the identity arrows and composition, the remains of a
        category becomes a graph.
        \begin{note}
          If $F$ is a functor, then $F$ is a graph homomorphism. Note that
          the converse is not true.
        \end{note}
        Simiarily, there is the functor
        $A:\mathbf{Cat}\rightarrow\mathbf{Set}$ and
        $O:\mathbf{Cat}\rightarrow\mathbf{Set}$ which takes a category to
        its set of arrows and objects respectively.

      \item $U:\mathscr{C}/A\rightarrow\mathscr{C}$\\
        for an object $A$ in the category $\mathscr{C}$.
        \begin{note}
          If $\mathscr{C}$ is \textbf{Set}, the underlying functor
          $U:\mathbf{Set}/S\rightarrow\mathbf{Set}$ forgets the indexing of
          $S$-indexed sets.
        \end{note}

    \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{freefunctor}
{
  name=Free Functor,
  text=free functor
} {
  Informally, a \textbf{free functor} is left adjoint to a \textbf{forgetful
  functor}.
  \begin{itemize}[label={}, align=left, nolistsep]
    \item[\textit{Examples}:]
    \item
    \begin{itemize}
      \item $F:\mathbf{Set}\rightarrow\mathbf{Mon}$\\
        is the free monoid functor that takes a set $A$ to the free monoid
        $F(A)$, the Kleene closure $A^*$ with concatenation as the monoid
        operation. $F$ takes set functions to the Kleene closure induced
        homormophism.
        \begin{note}
          The Kleene closure itself is a functor from \textbf{Set} to
          \textbf{Set}, which is the composition of $U\circ F$, the
          underlying functor $U:\mathbf{Mon}\rightarrow\mathbf{Set}$ and the
          free monoid functor.
        \end{note}

      \item $F:\mathbf{Grf}\rightarrow\mathbf{Cat}$\\
        takes graphs to the path category of the graph as objects in
        \textbf{Cat} and graph homomorphisms to arrows (functors between
        path categories). For a graph homomorphism
        $\phi:\mathscr{G}\rightarrow\mathscr{H}$,
        $F(\phi)(f_n,f_{n-1},\dots,f_1)=(\phi(f_n),\phi(f_{n-1}),\dots,\phi(f_1))$,
        which is clearly a path in $\mathscr{H}$ (and thus an arrow in the
        path category of $\mathscr{H}$).
    \end{itemize}
  \end{itemize}
}

\newglossaryentry{contravariantfunctor}
{
  name=Contravariant Functor,
  text=contravariant functor,
  description={is a functor $F:\mathscr{C}^{op}\rightarrow\mathscr{D}$ for
    categories $\mathscr{C}$ and $\mathscr{D}$. In contrast to
    \textit{covariant} functors, for an arrow $f$ in $\mathscr{C}$
    \begin{center}
      $F(f:B\rightarrow A)=F(A)\rightarrow F(B)$
    \end{center}
    is an arrow in $\mathscr{D}$. This follows from the fact that $f$ is an
  arrow from $A$ to $B$ in $\mathscr{C}^{op}$.}
}

\longnewglossaryentry{powersetfunctor}
{
  name=Powerset Functor,
  text=powerset functor
} {
  is a functor that takes a set $S$ to its powerset
  $\mathcal{P}(S)$. There are three different powerset functors:

  \begin{itemize}[label={}, nosep]
    \item
      \begin{itemize}[align=left]
        \item[\textbf{Inverse Image}]
          $\mathscr{P}:\mathbf{Set}^{op}\rightarrow\mathbf{Set}$\\
          is a contravariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to its inverse image
          $f:\mathcal{P}(B)\rightarrow\mathcal{P}(A)$.

        \item[\textbf{Direct/Existential Image}]
          is a covariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to the function
          $f_*:\mathcal{P}(A)\rightarrow\mathcal{P}(B)$ where
          $f_*(A_0)=\{f(x)|x\in A_0\}$.

        \item[\textbf{Universal Image}]
          is a covariant functor that takes sets $S$ to its powerset
          $\mathcal{P}(S)$ and functions $f:A\rightarrow B$ to the function
          $f_!:\mathcal{P}(A)\rightarrow\mathcal{P}(B)$ where
          $f_!(A_0)=\{y\in B|(f(x)=y)\Rightarrow(x\in A_0)\}$.
      \end{itemize}
  \end{itemize}
}

\longnewglossaryentry{homfunctor}
{
  name=Hom Functor,
  text=hom functor
} {
  is a functor that takes a category to a hom-set. For any category
  $\mathscr{C}$ with an object $C$, there are three different hom functors:

  \begin{itemize}[label={}, nosep]
    \item
      \begin{itemize}[align=left]
        \item[\textbf{Covariant Hom Functor}]
          $\Hom(C,-):\mathscr{C}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(C,-)(A)=\Hom(C,A)$ for each object $A\in\mathscr{C}$
            \item $\Hom(C,-)(f)=\Hom(C,f):\Hom(C,A)\rightarrow\Hom(C,B)$ for
              an arrow $f:A\rightarrow B$ in $\mathscr{C}$
          \end{itemize}

        \item[\textbf{Contravariant Hom Functor}]
          $\Hom(-,C):\mathscr{C}^{op}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(-,C)(A)=\Hom(A,C)$ for each object $A\in\mathscr{C}$
            \item $\Hom(-,C)(f)=\Hom(f,C):\Hom(B,C)\rightarrow\Hom(A,C)$ for
              an arrow $f:A\rightarrow B$ in $\mathscr{C}$
          \end{itemize}

        \item[\textbf{Two-Variable Hom Functor}]
          $\Hom(-,-):\mathscr{C}^{op}\times\mathscr{C}\rightarrow\mathbf{Set}$
          \begin{itemize}[nosep]
            \item $\Hom(-,-)((A,B))=\Hom(A,B)$ for each object
              $(A,B)\in\mathscr{C}^{op}\times\mathscr{C}$
            \item $\Hom(-,-)((f,g))=\Hom(f,g):\Hom(B,C)\rightarrow\Hom(A,D)$ for
              an arrow $(f:A\rightarrow B, g:C\rightarrow D)$ in
              $\mathscr{C}^{op}\times\mathscr{C}$
          \end{itemize}

      \end{itemize}
  \end{itemize}
}

\newglossaryentry{fullfunctor}
{
  name=Full,
  text=full,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$ is
    \textbf{full} if the set mapping:
    \begin{center}
      $\Hom_\mathscr{C}(A,B)\rightarrow\Hom_\mathscr{D}(F(A),F(B))$
    \end{center}
    for each pair of objects $A$ and $B$ in $\mathscr{C}$ induced by $F$ is
    \textit{surjective}.
%
    \begin{note}
      A full functor need not be surjective on objects or arrows.
    \end{note}
  }
}

\newglossaryentry{faithfunctor}
{
  name=Faithful,
  text=faithful,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$ is
    \textbf{faithful} if the set mapping:
    \begin{center}
      $\Hom_\mathscr{C}(A,B)\rightarrow\Hom_\mathscr{D}(F(A),F(B))$
    \end{center}
    for each pair of objects $A$ and $B$ in $\mathscr{C}$ induced by $F$ is
    \textit{injective}.
%
    \begin{note}
      If $f:A\rightarrow B$ and $g:C\rightarrow D$, as long as either $A\neq
      C$ or $C\neq D$, $F(f)=F(g)$ may hold and $F$ can still be faithful.
    \end{note}
%
    \begin{note}
      A faithful functor need not be injective on objects or arrows.
    \end{note}
  }
}

\newglossaryentry{reflect}
{
  name=Reflect,
  text=reflect,
  description={A functor $F:\mathscr{C}\rightarrow\mathscr{D}$
    \textbf{reflects} a property $P$ of arrows if whenever $F(f)$ has property
    $P$ then so does $f$ (where $f$ is any arrow for which $F$ takes to
    $F(f)$).
    \begin{note}[Theorem]
      Consider a full and faithful functor
      $F:\mathscr{C}\rightarrow\mathscr{D}$. Let $A$ and $B$ be objects in
      $C$ and $u:F(A)\rightarrow F(B)$ is an isomorphism in $\mathscr{D}$,
      then there is an unique isomorphism $f:A\rightarrow
      B$ in $\mathscr{C}$ for which $F(f)=u$.
    \end{note}
    \begin{note}[Corollary]
      A full and faithful functor reflects isomorphisms.
    \end{note}
    \begin{note}[Corollary]
      For a full an faithful functor $F$, if $F(A)=F(B)$ for objects $A$ and
      $B$ in the domain of $F$, then $A\cong B$.
    \end{note}
  }
}
